105 QByteArrayView bav((char*)&event->fingerId, sizeof(event->fingerId));

110 }

113 }
114

154void SdlInputHandler::emulateAbsoluteFingerEvent(SDL_TouchFingerEvent* event)
155{
156 // Observations on Windows 10: x and y appear to be relative to 0,0 of the window client area.
157 // Although SDL documentation states they are 0.0 - 1.0 float values, they can actually be higher
158 // or lower than those values as touch events continue for touches started within the client area that
159 // leave the client area during a drag motion.
160 // dx and dy are deltas from the last touch event, not the first touch down.
161
162 // Ignore touch down events with more than one finger
163 if (event->type == SDL_FINGERDOWN && SDL_GetNumTouchFingers(event->touchId) > 1) {
164 return;
165 }
166
167 // Ignore touch move and touch up events from the non-primary finger
168 if (event->type != SDL_FINGERDOWN && event->fingerId != m_LastTouchDownEvent.fingerId) {
169 return;
170 }
171
172 SDL_Rect src, dst;
173 int windowWidth, windowHeight;
174
175 SDL_GetWindowSize(m_Window, &windowWidth, &windowHeight);
176
177 src.x = src.y = 0;
178 src.w = m_StreamWidth;
179 src.h = m_StreamHeight;
180
181 dst.x = dst.y = 0;
182 dst.w = windowWidth;
183 dst.h = windowHeight;
184
185 // Use the stream and window sizes to determine the video region
186 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
187
188 if (qSqrt(qPow(event->x - m_LastTouchDownEvent.x, 2) + qPow(event->y - m_LastTouchDownEvent.y, 2)) > LONG_PRESS_ACTIVATION_DELTA) {
189 // Moved too far since touch down. Cancel the long press timer.
190 SDL_RemoveTimer(m_LongPressTimer);
191 m_LongPressTimer = 0;
192 }
193
194 // Don't reposition for finger down events within the deadzone. This makes double-clicking easier.
195 if (event->type != SDL_FINGERDOWN ||
196 event->timestamp - m_LastTouchUpEvent.timestamp > DOUBLE_TAP_DEAD_ZONE_DELAY ||
197 qSqrt(qPow(event->x - m_LastTouchUpEvent.x, 2) + qPow(event->y - m_LastTouchUpEvent.y, 2)) > DOUBLE_TAP_DEAD_ZONE_DELTA) {
198 // Scale window-relative events to be video-relative and clamp to video region
199 short x = qMin(qMax((int)(event->x * windowWidth), dst.x), dst.x + dst.w);
200 short y = qMin(qMax((int)(event->y * windowHeight), dst.y), dst.y + dst.h);
201
202 // Update the cursor position relative to the video region
203 LiSendMousePositionEvent(x - dst.x, y - dst.y, dst.w, dst.h);
204 }
205
206 if (event->type == SDL_FINGERDOWN) {
207 m_LastTouchDownEvent = *event;
208
209 // Start/restart the long press timer
210 SDL_RemoveTimer(m_LongPressTimer);
211 m_LongPressTimer = SDL_AddTimer(LONG_PRESS_ACTIVATION_DELAY,
212 longPressTimerCallback,
213 nullptr);
214
215 // Left button down on finger down
216 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_LEFT);
217 }
218 else if (event->type == SDL_FINGERUP) {
219 m_LastTouchUpEvent = *event;
220
221 // Cancel the long press timer
222 SDL_RemoveTimer(m_LongPressTimer);
223 m_LongPressTimer = 0;
224
225 // Left button up on finger up
226 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_LEFT);
227
228 // Raise right button too in case we triggered a long press gesture
229 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_RIGHT);
230 }
231}
