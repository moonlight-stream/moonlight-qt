99{
100 unsigned int portFlags = LiGetPortFlagsFromTerminationErrorCode(errorCode);
101 s_ActiveSession->m_PortTestResults = LiTestClientConnectivity(CONN_TEST_SERVER, 443, portFlags);
102

106 break;

119 s_ActiveSession->m_UnexpectedTermination = true;
120 emit s_ActiveSession->displayLaunchError(tr("Your network connection isn't performing well. Reduce your video bitrate setting or try a faster connection."));
121 break;
122
123 case ML_ERROR_PROTECTED_CONTENT:
124 case ML_ERROR_UNEXPECTED_EARLY_TERMINATION:
125 s_ActiveSession->m_UnexpectedTermination = true;
126 emit s_ActiveSession->displayLaunchError(tr("Something went wrong on your host PC when starting the stream.") + "\n\n" +
127 tr("Make sure you don't have any DRM-protected content open on your host PC. You can also try restarting your host PC."));
128 break;
129
130 case ML_ERROR_FRAME_CONVERSION:
131 s_ActiveSession->m_UnexpectedTermination = true;
132 emit s_ActiveSession->displayLaunchError(tr("The host PC reported a fatal video encoding error.") + "\n\n" +
133 tr("Try disabling HDR mode, changing the streaming resolution, or changing your host PC's display resolution."));
134 break;

137 s_ActiveSession->m_UnexpectedTermination = true;

140 bool hexError = qAbs(errorCode) > 1000;
141 emit s_ActiveSession->displayLaunchError(tr("Connection terminated") + "\n\n" +
142 tr("Error code: %1").arg(errorCode, hexError ? 8 : 0, hexError ? 16 : 10, QChar('0')));

148 errorCode);

152 event.type = SDL_QUIT;

373int Session::drSubmitDecodeUnit(PDECODE_UNIT du)
374{
375 // Use a lock since we'll be yanking this decoder out
376 // from underneath the session when we initiate destruction.
377 // We need to destroy the decoder on the main thread to satisfy
378 // some API constraints (like DXVA2). If we can't acquire it,
379 // that means the decoder is about to be destroyed, so we can
380 // safely return DR_OK and wait for the IDR frame request by
381 // the decoder reinitialization code.
382
383 if (SDL_TryLockMutex(s_ActiveSession->m_DecoderLock) == 0) {
384 IVideoDecoder* decoder = s_ActiveSession->m_VideoDecoder;
385 if (decoder != nullptr) {
386 int ret = decoder->submitDecodeUnit(du);
387 SDL_UnlockMutex(s_ActiveSession->m_DecoderLock);
388 return ret;
389 }
390 else {
391 SDL_UnlockMutex(s_ActiveSession->m_DecoderLock);
392 return DR_OK;
393 }
394 }
395 else {
396 // Decoder is going away. Ignore anything coming in until
397 // the lock is released.
398 return DR_OK;
399 }
400}

402void Session::getDecoderInfo(SDL_Window* window,
403 bool& isHardwareAccelerated, bool& isFullScreenOnly,
404 bool& isHdrSupported, QSize& maxResolution)
405{
406 IVideoDecoder* decoder;
407
408 // Since AV1 support on the host side is in its infancy, let's not consider
409 // _only_ a working AV1 decoder to be acceptable and still show the warning
410 // dialog indicating lack of hardware decoding support.
411
412 // Try an HEVC Main10 decoder first to see if we have HDR support
413 if (chooseDecoder(StreamingPreferences::VDS_FORCE_HARDWARE,
414 window, VIDEO_FORMAT_H265_MAIN10, 1920, 1080, 60,
415 false, false, true, decoder)) {
416 isHardwareAccelerated = decoder->isHardwareAccelerated();
417 isFullScreenOnly = decoder->isAlwaysFullScreen();
418 isHdrSupported = decoder->isHdrSupported();
419 maxResolution = decoder->getDecoderMaxResolution();
420 delete decoder;
421
422 return;
423 }
424
425 // Try an AV1 Main10 decoder next to see if we have HDR support
426 if (chooseDecoder(StreamingPreferences::VDS_FORCE_HARDWARE,
427 window, VIDEO_FORMAT_AV1_MAIN10, 1920, 1080, 60,
428 false, false, true, decoder)) {
429 // If we've got a working AV1 Main 10-bit decoder, we'll enable the HDR checkbox
430 // but we will still continue probing to get other attributes for HEVC or H.264
431 // decoders. See the AV1 comment at the top of the function for more info.
432 isHdrSupported = decoder->isHdrSupported();
433 delete decoder;
434 }
435 else {
436 // If we found no hardware decoders with HDR, check for a renderer
437 // that supports HDR rendering with software decoded frames.
438 if (chooseDecoder(StreamingPreferences::VDS_FORCE_SOFTWARE,
439 window, VIDEO_FORMAT_H265_MAIN10, 1920, 1080, 60,
440 false, false, true, decoder) ||
441 chooseDecoder(StreamingPreferences::VDS_FORCE_SOFTWARE,
442 window, VIDEO_FORMAT_AV1_MAIN10, 1920, 1080, 60,
443 false, false, true, decoder)) {
444 isHdrSupported = decoder->isHdrSupported();
445 delete decoder;
446 }
447 else {
448 // We weren't compiled with an HDR-capable renderer or we don't
449 // have the required GPU driver support for any HDR renderers.
450 isHdrSupported = false;
451 }
452 }
453
454 // Try a regular hardware accelerated HEVC decoder now
455 if (chooseDecoder(StreamingPreferences::VDS_FORCE_HARDWARE,
456 window, VIDEO_FORMAT_H265, 1920, 1080, 60,
457 false, false, true, decoder)) {
458 isHardwareAccelerated = decoder->isHardwareAccelerated();
459 isFullScreenOnly = decoder->isAlwaysFullScreen();
460 maxResolution = decoder->getDecoderMaxResolution();
461 delete decoder;
462
463 return;
464 }
465
466
467#if 0 // See AV1 comment at the top of this function
468 if (chooseDecoder(StreamingPreferences::VDS_FORCE_HARDWARE,
469 window, VIDEO_FORMAT_AV1_MAIN8, 1920, 1080, 60,
470 false, false, true, decoder)) {
471 isHardwareAccelerated = decoder->isHardwareAccelerated();
472 isFullScreenOnly = decoder->isAlwaysFullScreen();
473 maxResolution = decoder->getDecoderMaxResolution();
474 delete decoder;
475
476 return;
477 }
478#endif
479
480 // If we still didn't find a hardware decoder, try H.264 now.
481 // This will fall back to software decoding, so it should always work.
482 if (chooseDecoder(StreamingPreferences::VDS_AUTO,
483 window, VIDEO_FORMAT_H264, 1920, 1080, 60,
484 false, false, true, decoder)) {
485 isHardwareAccelerated = decoder->isHardwareAccelerated();
486 isFullScreenOnly = decoder->isAlwaysFullScreen();
487 maxResolution = decoder->getDecoderMaxResolution();
488 delete decoder;
489
490 return;
491 }
492
493 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
494 "Failed to find ANY working H.264 or HEVC decoder!");
495}

604bool Session::initialize()
605{
606#ifdef Q_OS_DARWIN
607 if (qEnvironmentVariableIntValue("I_WANT_BUGGY_FULLSCREEN") == 0) {
608 // If we have a notch and the user specified one of the two native display modes
609 // (notched or notchless), override the fullscreen mode to ensure it works as expected.
610 // - SDL_HINT_VIDEO_MAC_FULLSCREEN_SPACES=0 will place the video underneath the notch
611 // - SDL_HINT_VIDEO_MAC_FULLSCREEN_SPACES=1 will place the video below the notch
612 bool shouldUseFullScreenSpaces = m_Preferences->windowMode != StreamingPreferences::WM_FULLSCREEN;
613 SDL_DisplayMode desktopMode;
614 SDL_Rect safeArea;
615 for (int displayIndex = 0; StreamUtils::getNativeDesktopMode(displayIndex, &desktopMode, &safeArea); displayIndex++) {
616 // Check if this display has a notch (safeArea != desktopMode)
617 if (desktopMode.h != safeArea.h || desktopMode.w != safeArea.w) {
618 // Check if we're trying to stream at the full native resolution (including notch)
619 if (m_Preferences->width == desktopMode.w && m_Preferences->height == desktopMode.h) {
620 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
621 "Overriding default fullscreen mode for native fullscreen resolution");
622 shouldUseFullScreenSpaces = false;
623 break;
624 }
625 else if (m_Preferences->width == safeArea.w && m_Preferences->height == safeArea.h) {
626 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
627 "Overriding default fullscreen mode for native safe area resolution");
628 shouldUseFullScreenSpaces = true;
629 break;
630 }
631 }

968 }
969 }
970 }
971 }

973 return true;

1187 if (!audioTestPassed && CHANNEL_COUNT_FROM_AUDIO_CONFIGURATION(m_StreamConfig.audioConfiguration) > 2) {
1188 audioTestPassed = testAudio(AUDIO_CONFIGURATION_STEREO);
1189 if (audioTestPassed) {
1190 m_StreamConfig.audioConfiguration = AUDIO_CONFIGURATION_STEREO;
1191 emitLaunchWarning(tr("Your selected surround sound setting is not supported by the current audio device."));
1192 }
1193 }
1194
1195 // If nothing worked, warn the user that audio will not work
1196 if (!audioTestPassed) {
1197 emitLaunchWarning(tr("Failed to open audio device. Audio will be unavailable during this session."));
1198 }
1199
1200 // Check for unmapped gamepads
1201 if (!SdlInputHandler::getUnmappedGamepads().isEmpty()) {
1202 emitLaunchWarning(tr("An attached gamepad has no mapping and won't be usable. Visit the Moonlight help to resolve this."));
1203 }
1204
1205 // If we removed all codecs with the checks above, use H.264 as the codec of last resort.
1206 if (m_SupportedVideoFormats.empty()) {
1207 m_SupportedVideoFormats.append(VIDEO_FORMAT_H264);
1208 }
1209
1210 // NVENC will fail to initialize when any dimension exceeds 4096 using:
1211 // - H.264 on all versions of NVENC
1212 // - HEVC prior to Pascal
1213 //
1214 // However, if we aren't using Nvidia hosting software, don't assume anything about
1215 // encoding capabilities by using HEVC Main 10 support. It will likely be wrong.
1216 if ((m_StreamConfig.width > 4096 || m_StreamConfig.height > 4096) && m_Computer->isNvidiaServerSoftware) {
1217 // Pascal added support for 8K HEVC encoding support. Maxwell 2 could encode HEVC but only up to 4K.
1218 // We can't directly identify Pascal, but we can look for HEVC Main10 which was added in the same generation.
1219 if (m_Computer->maxLumaPixelsHEVC == 0 || !(m_Computer->serverCodecModeSupport & SCM_HEVC_MAIN10)) {
1220 emit displayLaunchError(tr("Your host PC's GPU doesn't support streaming video resolutions over 4K."));
1221 return false;
1222 }
1223 else if ((m_SupportedVideoFormats & ~VIDEO_FORMAT_MASK_H264) == 0) {
1224 emit displayLaunchError(tr("Video resolutions over 4K are not supported by the H.264 codec."));
1225 return false;
1226 }
1227 }
1228
1229 if (m_Preferences->videoDecoderSelection == StreamingPreferences::VDS_FORCE_HARDWARE &&
1230 !(m_SupportedVideoFormats & VIDEO_FORMAT_MASK_10BIT) && // HDR was already checked for hardware decode support above
1231 getDecoderAvailability(testWindow,
1232 m_Preferences->videoDecoderSelection,
1233 m_SupportedVideoFormats.front(),
1234 m_StreamConfig.width,
1235 m_StreamConfig.height,
1236 m_StreamConfig.fps) != DecoderAvailability::Hardware) {
1237 if (m_Preferences->videoCodecConfig == StreamingPreferences::VCC_AUTO) {
1238 emit displayLaunchError(tr("Your selection to force hardware decoding cannot be satisfied due to missing hardware decoding support on this PC's GPU."));
1239 }
1240 else {
1241 emit displayLaunchError(tr("Your codec selection and force hardware decoding setting are not compatible. This PC's GPU lacks support for decoding your chosen codec."));
1242 }
1243
1244 // Fail the launch, because we won't manage to get a decoder for the actual stream
1245 return false;
1246 }
1247
1248 return true;
1249}

1400void Session::updateOptimalWindowDisplayMode()
1401{
1402 SDL_DisplayMode desktopMode, bestMode, mode;
1403 int displayIndex = SDL_GetWindowDisplayIndex(m_Window);
1404
1405 // Try the current display mode first. On macOS, this will be the normal
1406 // scaled desktop resolution setting.
1407 if (SDL_GetDesktopDisplayMode(displayIndex, &desktopMode) == 0) {
1408 // If this doesn't fit the selected resolution, use the native
1409 // resolution of the panel (unscaled).
1410 if (desktopMode.w < m_ActiveVideoWidth || desktopMode.h < m_ActiveVideoHeight) {
1411 SDL_Rect safeArea;
1412 if (!StreamUtils::getNativeDesktopMode(displayIndex, &desktopMode, &safeArea)) {
1413 return;
1414 }
1415 }
1416 }
1417 else {
1418 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
1419 "SDL_GetDesktopDisplayMode() failed: %s",
1420 SDL_GetError());
1421 return;
1422 }
1423
1424 // Start with the native desktop resolution and try to find
1425 // the highest refresh rate that our stream FPS evenly divides.
1426 bestMode = desktopMode;
1427 bestMode.refresh_rate = 0;
1428 for (int i = 0; i < SDL_GetNumDisplayModes(displayIndex); i++) {
1429 if (SDL_GetDisplayMode(displayIndex, i, &mode) == 0) {
1430 if (mode.w == desktopMode.w && mode.h == desktopMode.h &&
1431 mode.refresh_rate % m_StreamConfig.fps == 0) {
1432 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1433 "Found display mode with desktop resolution: %dx%dx%d",
1434 mode.w, mode.h, mode.refresh_rate);
1435 if (mode.refresh_rate > bestMode.refresh_rate) {
1436 bestMode = mode;
1437 }
1438 }
1439 }
1440 }
1441
1442 // If we didn't find a mode that matched the current resolution and
1443 // had a high enough refresh rate, start looking for lower resolution
1444 // modes that can meet the required refresh rate and minimum video
1445 // resolution. We will also try to pick a display mode that matches
1446 // aspect ratio closest to the video stream.
1447 if (bestMode.refresh_rate == 0) {
1448 float bestModeAspectRatio = 0;
1449 float videoAspectRatio = (float)m_ActiveVideoWidth / (float)m_ActiveVideoHeight;
1450 for (int i = 0; i < SDL_GetNumDisplayModes(displayIndex); i++) {
1451 if (SDL_GetDisplayMode(displayIndex, i, &mode) == 0) {
1452 float modeAspectRatio = (float)mode.w / (float)mode.h;
1453 if (mode.w >= m_ActiveVideoWidth && mode.h >= m_ActiveVideoHeight &&
1454 mode.refresh_rate % m_StreamConfig.fps == 0) {
1455 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1456 "Found display mode with video resolution: %dx%dx%d",
1457 mode.w, mode.h, mode.refresh_rate);
1458 if (mode.refresh_rate >= bestMode.refresh_rate &&
1459 (bestModeAspectRatio == 0 || fabs(videoAspectRatio - modeAspectRatio) <= fabs(videoAspectRatio - bestModeAspectRatio))) {
1460 bestMode = mode;
1461 bestModeAspectRatio = modeAspectRatio;
1462 }
1463 }
1464 }
1465 }
1466 }
1467
1468 if (bestMode.refresh_rate == 0) {
1469 // We may find no match if the user has moved a 120 FPS
1470 // stream onto a 60 Hz monitor (since no refresh rate can
1471 // divide our FPS setting). We'll stick to the default in
1472 // this case.
1473 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
1474 "No matching display mode found; using desktop mode");
1475 bestMode = desktopMode;
1476 }
1477
1478 if ((SDL_GetWindowFlags(m_Window) & SDL_WINDOW_FULLSCREEN_DESKTOP) == SDL_WINDOW_FULLSCREEN) {
1479 // Only print when the window is actually in full-screen exclusive mode,
1480 // otherwise we're not actually using the mode we've set here
1481 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1482 "Chosen best display mode: %dx%dx%d",
1483 bestMode.w, bestMode.h, bestMode.refresh_rate);
1484 }
1485
1486 SDL_SetWindowDisplayMode(m_Window, &bestMode);
1487}

1563// Called in a non-main thread
1564bool Session::startConnectionAsync()
1565{
1566 // Wait 1.5 seconds before connecting to let the user
1567 // have time to read any messages present on the segue
1568 SDL_Delay(1500);
1569
1570 // The UI should have ensured the old game was already quit
1571 // if we decide to stream a different game.
1572 Q_ASSERT(m_Computer->currentGameId == 0 ||
1573 m_Computer->currentGameId == m_App.id);
1574
1575 bool enableGameOptimizations;
1576 if (m_Computer->isNvidiaServerSoftware) {
1577 // GFE will set all settings to 720p60 if it doesn't recognize
1578 // the chosen resolution. Avoid that by disabling SOPS when it
1579 // is not streaming a supported resolution.
1580 enableGameOptimizations = false;
1581 for (const NvDisplayMode &mode : m_Computer->displayModes) {
1582 if (mode.width == m_StreamConfig.width &&
1583 mode.height == m_StreamConfig.height) {
1584 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1585 "Found host supported resolution: %dx%d",
1586 mode.width, mode.height);
1587 enableGameOptimizations = m_Preferences->gameOptimizations;
1588 break;
1589 }
1590 }
1591 }
1592 else {
1593 // Always send SOPS to Sunshine because we may repurpose the
1594 // option to control whether the display mode is adjusted
1595 enableGameOptimizations = m_Preferences->gameOptimizations;
1596 }
1597
1598 QString rtspSessionUrl;
1599
1600 try {
1601 NvHTTP http(m_Computer);
1602 http.startApp(m_Computer->currentGameId != 0 ? "resume" : "launch",
1603 m_Computer->isNvidiaServerSoftware,
1604 m_App.id, &m_StreamConfig,
1605 enableGameOptimizations,
1606 m_Preferences->playAudioOnHost,
1607 m_InputHandler->getAttachedGamepadMask(),
1608 !m_Preferences->multiController,
1609 rtspSessionUrl);
1610 } catch (const GfeHttpResponseException& e) {
1611 emit displayLaunchError(tr("Host returned error: %1").arg(e.toQString()));
1612 return false;
1613 } catch (const QtNetworkReplyException& e) {
1614 emit displayLaunchError(e.toQString());
1615 return false;
1616 }
1617
1618 QByteArray hostnameStr = m_Computer->activeAddress.address().toLatin1();
1619 QByteArray siAppVersion = m_Computer->appVersion.toLatin1();
1620
1621 SERVER_INFORMATION hostInfo;
1622 hostInfo.address = hostnameStr.data();
1623 hostInfo.serverInfoAppVersion = siAppVersion.data();
1624 hostInfo.serverCodecModeSupport = m_Computer->serverCodecModeSupport;
1625
1626 // Older GFE versions didn't have this field
1627 QByteArray siGfeVersion;
1628 if (!m_Computer->gfeVersion.isEmpty()) {
1629 siGfeVersion = m_Computer->gfeVersion.toLatin1();
1630 }
1631 if (!siGfeVersion.isEmpty()) {
1632 hostInfo.serverInfoGfeVersion = siGfeVersion.data();
1633 }
1634
1635 // Older GFE and Sunshine versions didn't have this field
1636 QByteArray rtspSessionUrlStr;
1637 if (!rtspSessionUrl.isEmpty()) {
1638 rtspSessionUrlStr = rtspSessionUrl.toLatin1();
1639 hostInfo.rtspSessionUrl = rtspSessionUrlStr.data();
1640 }
1641
1642 if (m_Preferences->packetSize != 0) {
1643 // Override default packet size and remote streaming detection
1644 // NB: Using STREAM_CFG_AUTO will cap our packet size at 1024 for remote hosts.
1645 m_StreamConfig.streamingRemotely = STREAM_CFG_LOCAL;
1646 m_StreamConfig.packetSize = m_Preferences->packetSize;
1647 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
1648 "Using custom packet size: %d bytes",
1649 m_Preferences->packetSize);
1650 }
1651 else {
1652 // Use 1392 byte video packets by default
1653 m_StreamConfig.packetSize = 1392;
1654
1655 // getActiveAddressReachability() does network I/O, so we only attempt to check
1656 // reachability if we've already contacted the PC successfully.
1657 switch (m_Computer->getActiveAddressReachability()) {
1658 case NvComputer::RI_LAN:
1659 // This address is on-link, so treat it as a local address
1660 // even if it's not in RFC 1918 space or it's an IPv6 address.
1661 m_StreamConfig.streamingRemotely = STREAM_CFG_LOCAL;
1662 break;
1663 case NvComputer::RI_VPN:
1664 // It looks like our route to this PC is over a VPN, so cap at 1024 bytes.
1665 // Treat it as remote even if the target address is in RFC 1918 address space.
1666 m_StreamConfig.streamingRemotely = STREAM_CFG_REMOTE;
1667 m_StreamConfig.packetSize = 1024;
1668 break;
1669 default:
1670 // If we don't have reachability info, let moonlight-common-c decide.
1671 m_StreamConfig.streamingRemotely = STREAM_CFG_AUTO;
1672 break;
1673 }
1674 }
1675
1676 // If the user has chosen YUV444 without adjusting the bitrate but the host doesn't
1677 // support YUV444 streaming, use the default non-444 bitrate for the stream instead.
1678 // This should provide equivalent image quality for YUV420 as the stream would have
1679 // had if the host supported YUV444 (though obviously with 4:2:0 subsampling).
1680 // If the user has adjusted the bitrate from default, we'll assume they really wanted
1681 // that value and not second guess them.
1682 if (m_Preferences->enableYUV444 &&
1683 !(m_StreamConfig.supportedVideoFormats & VIDEO_FORMAT_MASK_YUV444) &&
1684 m_StreamConfig.bitrate == StreamingPreferences::getDefaultBitrate(m_StreamConfig.width,
1685 m_StreamConfig.height,
1686 m_StreamConfig.fps,
1687 true)) {
1688 m_StreamConfig.bitrate = StreamingPreferences::getDefaultBitrate(m_StreamConfig.width,
1689 m_StreamConfig.height,
1690 m_StreamConfig.fps,
1691 false);
1692 }
1693
1694 // Set auto bitrate flag based on checkbox state
1695 m_StreamConfig.autoBitrateEnabled = m_Preferences->autoAdjustBitrate ? 1 : 0;
1696 m_StreamConfig.autoBitrateMinKbps = m_Preferences->autoBitrateMinKbps;
1697 m_StreamConfig.autoBitrateMaxKbps = m_Preferences->autoBitrateMaxKbps;
1698
1699 int err = LiStartConnection(&hostInfo, &m_StreamConfig, &k_ConnCallbacks,
1700 &m_VideoCallbacks, &m_AudioCallbacks,
1701 NULL, 0, NULL, 0);
1702 if (err != 0) {
1703 // We already displayed an error dialog in the stage failure
1704 // listener.
1705 return false;
1706 }
1707
1708 emit connectionStarted();
1709 return true;
1710}

1762void Session::exec(QWindow* qtWindow)
1763{
1764 m_QtWindow = qtWindow;
1765
1766 // Use a separate thread for the streaming session on X11 or Wayland
1767 // to ensure we don't stomp on Qt's GL context. This breaks when using
1768 // the Qt EGLFS backend, so we will restrict this to X11
1769 m_ThreadedExec = WMUtils::isRunningX11() || WMUtils::isRunningWayland();
1770
1771 if (m_ThreadedExec) {
1772 // Run the streaming session on a separate thread for Linux/BSD
1773 ExecThread execThread(this);
1774 execThread.start();
1775
1776 // Until the SDL streaming window is created, we should continue
1777 // to update the Qt UI to allow warning messages to display and
1778 // make sure that the Qt window can hide itself.
1779 while (!execThread.wait(10) && m_Window == nullptr) {
1780 QCoreApplication::processEvents(QEventLoop::ExcludeUserInputEvents);
1781 QCoreApplication::sendPostedEvents();
1782 }
1783 QCoreApplication::processEvents(QEventLoop::ExcludeUserInputEvents);
1784 QCoreApplication::sendPostedEvents();
1785
1786 // SDL is in charge now. Wait until the streaming thread exits
1787 // to further update the Qt window.
1788 execThread.wait();
1789 }
1790 else {
1791 // Run the streaming session on the main thread for Windows and macOS
1792 execInternal();
1793 }

1796void Session::execInternal()
1797{
1798 // Complete initialization in this deferred context to avoid
1799 // calling expensive functions in the constructor (during the
1800 // process of loading the StreamSegue).
1801 //
1802 // NB: This initializes the SDL video subsystem, so it must be
1803 // called on the main thread.
1804 if (!initialize()) {
1805 emit sessionFinished(0);
1806 emit readyForDeletion();
1807 return;
1808 }
1809
1810 // Wait for any old session to finish cleanup
1811 s_ActiveSessionSemaphore.acquire();
1812
1813 // We're now active
1814 s_ActiveSession = this;
1815
1816 // Initialize the gamepad code with our preferences
1817 // NB: m_InputHandler must be initialize before starting the connection.
1818 m_InputHandler = new SdlInputHandler(*m_Preferences, m_StreamConfig.width, m_StreamConfig.height);
1819
1820 AsyncConnectionStartThread asyncConnThread(this);
