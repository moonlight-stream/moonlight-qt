112IdentityManager::IdentityManager()
113{
114 QSettings settings;
115
116 m_CachedPemCert = settings.value(SER_CERT).toByteArray();
117 m_CachedPrivateKey = settings.value(SER_KEY).toByteArray();
118
119 if (m_CachedPemCert.isEmpty() || m_CachedPrivateKey.isEmpty()) {

128 qWarning() << "Private key is unreadable";
