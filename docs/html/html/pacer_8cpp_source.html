17// Limit the number of queued frames to prevent excessive memory consumption
18// if the V-Sync source or renderer is blocked for a while. It's important
19// that the sum of all queued frames between both pacing and rendering queues
20// must not exceed the number buffer pool size to avoid running the decoder
21// out of available decoding surfaces.

24// We may be woken up slightly late so don't go all the way

26// the next V-sync period. It also takes some amount of time
27// to do the render itself, so we can't render right before
28// V-sync happens.

45{

101int Pacer::vsyncThread(void *context)
102{
103 Pacer* me = reinterpret_cast<Pacer*>(context);
104
105#if SDL_VERSION_ATLEAST(2, 0, 9)
106 SDL_SetThreadPriority(SDL_THREAD_PRIORITY_TIME_CRITICAL);
107#else
108 SDL_SetThreadPriority(SDL_THREAD_PRIORITY_HIGH);
109#endif
110
111 bool async = me->m_VsyncSource->isAsync();
112 while (!me->m_Stopping) {
113 if (async) {
114 // Wait for the VSync source to invoke signalVsync() or 100ms to elapse
115 me->m_FrameQueueLock.lock();
116 me->m_VsyncSignalled.wait(&me->m_FrameQueueLock, 100);
117 me->m_FrameQueueLock.unlock();
118 }
119 else {
120 // Let the VSync source wait in the context of our thread
121 me->m_VsyncSource->waitForVsync();
122 }
123
124 if (me->m_Stopping) {
125 break;
126 }
127
128 me->handleVsync(1000 / me->m_DisplayFps);
129 }
130
131 return 0;

134int Pacer::renderThread(void* context)
135{
136 Pacer* me = reinterpret_cast<Pacer*>(context);
137
138 if (SDL_SetThreadPriority(SDL_THREAD_PRIORITY_HIGH) < 0) {
139 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
140 "Unable to set render thread to high priority: %s",
141 SDL_GetError());
142 }
143
144 while (!me->m_Stopping) {
145 // Wait for the renderer to be ready for the next frame
146 me->m_VsyncRenderer->waitToRender();
147
148 // Acquire the frame queue lock to protect the queue and
149 // the not empty condition
150 me->m_FrameQueueLock.lock();
151
152 // Wait for a frame to be ready to render

154 me->m_RenderQueueNotEmpty.wait(&me->m_FrameQueueLock);
155 }

159 me->m_FrameQueueLock.unlock();
160 break;

164 me->m_FrameQueueLock.unlock();

166 me->renderFrame(frame);
167 }
168

170 // NB: This must happen on the same thread that calls renderFrame().
171 me->m_VsyncRenderer->cleanupRenderContext();

196// Called in an arbitrary thread by the IVsyncSource on V-sync
197// or an event synchronized with V-sync
198void Pacer::handleVsync(int timeUntilNextVsyncMillis)
199{
200 // Make sure initialize() has been called
201 SDL_assert(m_MaxVideoFps != 0);
202
203 m_FrameQueueLock.lock();
204
205 // If the queue length history entries are large, be strict
206 // about dropping excess frames.
207 int frameDropTarget = 1;
208
209 // If we may get more frames per second than we can display, use
210 // frame history to drop frames only if consistently above the
211 // one queued frame mark.
212 if (m_MaxVideoFps >= m_DisplayFps) {
213 for (int queueHistoryEntry : m_PacingQueueHistory) {
214 if (queueHistoryEntry <= 1) {
215 // Be lenient as long as the queue length
216 // resolves before the end of frame history
217 frameDropTarget = 3;
218 break;
219 }
220 }
221
222 // Keep a rolling 500 ms window of pacing queue history
223 if (m_PacingQueueHistory.count() == m_DisplayFps / 2) {
224 m_PacingQueueHistory.dequeue();
225 }
226
227 m_PacingQueueHistory.enqueue(m_PacingQueue.count());
228 }
229
230 // Catch up if we're several frames ahead
231 while (m_PacingQueue.count() > frameDropTarget) {
232 AVFrame* frame = m_PacingQueue.dequeue();
233
234 // Drop the lock while we call av_frame_free()
235 m_FrameQueueLock.unlock();
236 m_VideoStats->pacerDroppedFrames++;
237 av_frame_free(&frame);
238 m_FrameQueueLock.lock();
239 }
240
241 if (m_PacingQueue.isEmpty()) {
242 // Wait for a frame to arrive or our V-sync timeout to expire
243 if (!m_PacingQueueNotEmpty.wait(&m_FrameQueueLock, SDL_max(timeUntilNextVsyncMillis, TIMER_SLACK_MS) - TIMER_SLACK_MS)) {
244 // Wait timed out - unlock and bail
245 m_FrameQueueLock.unlock();
246 return;
247 }
248
249 if (m_Stopping) {
250 m_FrameQueueLock.unlock();
251 return;
252 }
253 }
254
255 // Place the first frame on the render queue
256 enqueueFrameForRenderingAndUnlock(m_PacingQueue.dequeue());
257}

259bool Pacer::initialize(SDL_Window* window, int maxVideoFps, bool enablePacing)
260{

264

268 m_DisplayFps, m_MaxVideoFps);

271 SDL_VERSION(&info.version);

273 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
274 "SDL_GetWindowWMInfo() failed: %s",

276 return false;
277 }
278

287 break;
288 #endif
289
290 #if defined(SDL_VIDEO_DRIVER_WAYLAND) && defined(HAS_WAYLAND)
291 case SDL_SYSWM_WAYLAND:
292 m_VsyncSource = new WaylandVsyncSource(this);
293 break;
294 #endif
295
296 default:
297 // Platforms without a VsyncSource will just render frames
298 // immediately like they used to.
299 break;
300 }
301
302 SDL_assert(m_VsyncSource != nullptr || !(m_RendererAttributes & RENDERER_ATTRIBUTE_FORCE_PACING));
303

305 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
306 "Vsync source failed to initialize. Frame pacing will not be available!");
307 delete m_VsyncSource;

309 }
310 }
311 else {
312 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
313 "Frame pacing disabled: target %d Hz with %d FPS stream",
314 m_DisplayFps, m_MaxVideoFps);
315 }
316
317 if (m_VsyncSource != nullptr) {
318 m_VsyncThread = SDL_CreateThread(Pacer::vsyncThread, "PacerVsync", this);
319 }
320
321 if (m_VsyncRenderer->isRenderThreadSupported()) {
322 m_RenderThread = SDL_CreateThread(Pacer::renderThread, "PacerRender", this);
323 }
324
325 return true;

334{
335 // Count time spent in Pacer's queues
336 uint64_t beforeRender = LiGetMicroseconds();
337 m_VideoStats->totalPacerTimeUs += (beforeRender - (uint64_t)frame->pkt_dts);
338
339 // Render it
340 m_VsyncRenderer->renderFrame(frame);
341 uint64_t afterRender = LiGetMicroseconds();
342
343 m_VideoStats->totalRenderTimeUs += (afterRender - beforeRender);
344 m_VideoStats->renderedFrames++;
345 av_frame_free(&frame);
346
347 // Drop frames if we have too many queued up for a while
348 m_FrameQueueLock.lock();
349
350 int frameDropTarget;
351
352 if (m_RendererAttributes & RENDERER_ATTRIBUTE_NO_BUFFERING) {
353 // Renderers that don't buffer any frames but don't support waitToRender() need us to buffer
354 // an extra frame to ensure they don't starve while waiting to present.
355 frameDropTarget = 1;
356 }
357 else {
358 frameDropTarget = 0;
359 for (int queueHistoryEntry : m_RenderQueueHistory) {
360 if (queueHistoryEntry == 0) {
361 // Be lenient as long as the queue length
362 // resolves before the end of frame history
363 frameDropTarget = 2;
364 break;
365 }
366 }
367
368 // Keep a rolling 500 ms window of render queue history
369 if (m_RenderQueueHistory.count() == m_MaxVideoFps / 2) {
370 m_RenderQueueHistory.dequeue();
371 }
372
373 m_RenderQueueHistory.enqueue(m_RenderQueue.count());
374 }
375
376 // Catch up if we're several frames ahead
377 while (m_RenderQueue.count() > frameDropTarget) {
378 AVFrame* frame = m_RenderQueue.dequeue();
379
380 // Drop the lock while we call av_frame_free()
381 m_FrameQueueLock.unlock();
382 m_VideoStats->pacerDroppedFrames++;
383 av_frame_free(&frame);
