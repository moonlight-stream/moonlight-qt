7void SdlInputHandler::handleMouseButtonEvent(SDL_MouseButtonEvent* event)
8{
9 int button;
10
11 if (event->which == SDL_TOUCH_MOUSEID) {
12 // Ignore synthetic mouse events
13 return;
14 }
15 else if (!isCaptureActive()) {
16 if (event->button == SDL_BUTTON_LEFT && event->state == SDL_RELEASED &&
17 isMouseInVideoRegion(event->x, event->y)) {
18 // Capture the mouse again if clicked when unbound.
19 // We start capture on left button released instead of
20 // pressed to avoid sending an errant mouse button released
21 // event to the host when clicking into our window (since
22 // the pressed event was consumed by this code).
23 setCaptureActive(true);
24 }
25
26 // Not capturing
27 return;
28 }
29 else if (m_AbsoluteMouseMode && !isMouseInVideoRegion(event->x, event->y) && event->state == SDL_PRESSED) {
30 // Ignore button presses outside the video region, but allow button releases
31 return;
32 }
33
34 switch (event->button)
35 {
36 case SDL_BUTTON_LEFT:
37 button = BUTTON_LEFT;
38 break;
39 case SDL_BUTTON_MIDDLE:
40 button = BUTTON_MIDDLE;
41 break;
42 case SDL_BUTTON_RIGHT:
43 button = BUTTON_RIGHT;
44 break;
45 case SDL_BUTTON_X1:
46 button = BUTTON_X1;
47 break;
48 case SDL_BUTTON_X2:
49 button = BUTTON_X2;
50 break;
51 default:
52 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
53 "Unhandled button event: %d",
54 event->button);
55 return;
56 }
57
58 if (m_SwapMouseButtons) {
59 if (button == BUTTON_RIGHT)
60 button = BUTTON_LEFT;
61 else if (button == BUTTON_LEFT)
62 button = BUTTON_RIGHT;
63 }
64
65 LiSendMouseButtonEvent(event->state == SDL_PRESSED ?
66 BUTTON_ACTION_PRESS :
67 BUTTON_ACTION_RELEASE,
68 button);
69}

71void SdlInputHandler::handleMouseMotionEvent(SDL_MouseMotionEvent* event)
72{
73 if (!isCaptureActive()) {
74 // Not capturing
75 return;
76 }
77 else if (event->which == SDL_TOUCH_MOUSEID) {
78 // Ignore synthetic mouse events
79 return;
80 }
81
82 // Batch all pending mouse motion events to save CPU time
83 Sint32 x = event->x, y = event->y, xrel = event->xrel, yrel = event->yrel;
84 SDL_Event nextEvent;
85 while (SDL_PeepEvents(&nextEvent, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {
86 event = &nextEvent.motion;
87
88 // Ignore synthetic mouse events
89 if (event->which != SDL_TOUCH_MOUSEID) {
90 x = event->x;
91 y = event->y;
92 xrel += event->xrel;
93 yrel += event->yrel;
94 }
95 }
96
97 // We should not reference the original event anymore
98 event = nullptr;
99
100 if (m_AbsoluteMouseMode) {
101 int windowWidth, windowHeight;
102 SDL_GetWindowSize(m_Window, &windowWidth, &windowHeight);
103
104 SDL_Rect src, dst;
105 bool mouseInVideoRegion;
106
107 src.x = src.y = 0;
108 src.w = m_StreamWidth;
109 src.h = m_StreamHeight;
110
111 dst.x = dst.y = 0;
112 dst.w = windowWidth;
113 dst.h = windowHeight;
114
115 // Use the stream and window sizes to determine the video region
116 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
117
118 mouseInVideoRegion = isMouseInVideoRegion(x, y, windowWidth, windowHeight);
119
120 // Clamp motion to the video region
121 x = qMin(qMax(x - dst.x, 0), dst.w);
122 y = qMin(qMax(y - dst.y, 0), dst.h);
123
124 // Send the mouse position update if one of the following is true:
125 // a) it is in the video region now
126 // b) it just left the video region (to ensure the mouse is clamped to the video boundary)
127 // c) a mouse button is still down from before the cursor left the video region (to allow smooth dragging)
128 Uint32 buttonState = SDL_GetMouseState(nullptr, nullptr);
129 if (buttonState == 0) {
130 if (m_PendingMouseButtonsAllUpOnVideoRegionLeave) {
131 // Stop capturing the mouse now
132 SDL_CaptureMouse(SDL_FALSE);
133 m_PendingMouseButtonsAllUpOnVideoRegionLeave = false;
134 }
135 }
136 if (mouseInVideoRegion || m_MouseWasInVideoRegion || m_PendingMouseButtonsAllUpOnVideoRegionLeave) {
137 LiSendMousePositionEvent((short)x, (short)y, dst.w, dst.h);
138 }
139
140 // Adjust the cursor visibility if applicable
141 if (mouseInVideoRegion ^ m_MouseWasInVideoRegion) {
142 SDL_ShowCursor((mouseInVideoRegion && m_MouseCursorCapturedVisibilityState == SDL_DISABLE) ? SDL_DISABLE : SDL_ENABLE);
143 if (!mouseInVideoRegion && buttonState != 0) {
144 // If we still have a button pressed on leave, wait for that to come up
145 // before we stop sending mouse position events.
146 m_PendingMouseButtonsAllUpOnVideoRegionLeave = true;
147 }
148 }
149
150 m_MouseWasInVideoRegion = mouseInVideoRegion;
151 }
152 else {
153 LiSendMouseMoveEvent(xrel, yrel);
154 }
155}

238bool SdlInputHandler::isMouseInVideoRegion(int mouseX, int mouseY, int windowWidth, int windowHeight)
239{

241
242 if (windowWidth < 0 || windowHeight < 0) {

245

251 dst.w = windowWidth;
252 dst.h = windowHeight;

261void SdlInputHandler::updatePointerRegionLock()
262{
263 // Pointer region lock is irrelevant in relative mouse mode
264 if (SDL_GetRelativeMouseMode()) {
265 return;
266 }
267
268 // Our pointer lock behavior tracks with the fullscreen mode unless the user has
269 // toggled it themselves using the keyboard shortcut. If that's the case, they
270 // have full control over it and we don't touch it anymore.
271 if (!m_PointerRegionLockToggledByUser) {
272 // Lock the pointer in true full-screen mode or in any fullscreen mode when only a single monitor is present
273 Uint32 fullscreenFlags = SDL_GetWindowFlags(m_Window) & SDL_WINDOW_FULLSCREEN_DESKTOP;
274 m_PointerRegionLockActive = (fullscreenFlags == SDL_WINDOW_FULLSCREEN) ||
275 (fullscreenFlags != 0 && SDL_GetNumVideoDisplays() == 1);
276 }
277
278 // If region lock is enabled, grab the cursor so it can't accidentally leave our window.
279 if (isCaptureActive() && m_PointerRegionLockActive) {
280#if SDL_VERSION_ATLEAST(2, 0, 18)
281 SDL_Rect src, dst;
282
283 src.x = src.y = 0;
284 src.w = m_StreamWidth;
285 src.h = m_StreamHeight;
286
287 dst.x = dst.y = 0;
288 SDL_GetWindowSize(m_Window, &dst.w, &dst.h);
289
290 // Use the stream and window sizes to determine the video region
291 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
292
293 // SDL 2.0.18 lets us lock the cursor to a specific region
294 SDL_SetWindowMouseRect(m_Window, &dst);
295#elif SDL_VERSION_ATLEAST(2, 0, 15)
296 // SDL 2.0.15 only lets us lock the cursor to the whole window
297 SDL_SetWindowMouseGrab(m_Window, SDL_TRUE);
298#else
299 SDL_SetWindowGrab(m_Window, SDL_TRUE);
300#endif
301 }
302 else {
303 // Allow the cursor to leave the bounds of our video region or window
304#if SDL_VERSION_ATLEAST(2, 0, 18)
305 SDL_SetWindowMouseRect(m_Window, nullptr);
306#elif SDL_VERSION_ATLEAST(2, 0, 15)
307 SDL_SetWindowMouseGrab(m_Window, SDL_FALSE);
308#else
309 SDL_SetWindowGrab(m_Window, SDL_FALSE);
310#endif
311 }
