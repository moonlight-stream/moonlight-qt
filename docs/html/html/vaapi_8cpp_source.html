83VAAPIRenderer::openDisplay(SDL_Window* window)
84{
85 SDL_SysWMinfo info;
86 VADisplay display;
87
88 SDL_VERSION(&info.version);
89
90 if (!SDL_GetWindowWMInfo(window, &info)) {
91 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
92 "SDL_GetWindowWMInfo() failed: %s",
93 SDL_GetError());
94 return nullptr;
95 }
96
97 m_WindowSystem = info.subsystem;
98 if (info.subsystem == SDL_SYSWM_X11) {
99#ifdef HAVE_LIBVA_X11
100 m_XWindow = info.info.x11.window;
101 display = vaGetDisplay(info.info.x11.display);
102 if (display == nullptr) {
103 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
104 "Unable to open X11 display for VAAPI");
105 return nullptr;
106 }
107#else
108 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
109 "Moonlight not compiled with VAAPI X11 support!");
110 return nullptr;
111#endif
112 }
113 else if (info.subsystem == SDL_SYSWM_WAYLAND) {
114#ifdef HAVE_LIBVA_WAYLAND
115 display = vaGetDisplayWl(info.info.wl.display);
116 if (display == nullptr) {
117 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
118 "Unable to open Wayland display for VAAPI");
119 return nullptr;
120 }
121#else
122 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
123 "Moonlight not compiled with VAAPI Wayland support!");
124 return nullptr;
125#endif
126 }
127#if defined(SDL_VIDEO_DRIVER_KMSDRM) && defined(HAVE_LIBVA_DRM) && SDL_VERSION_ATLEAST(2, 0, 15)
128 else if (info.subsystem == SDL_SYSWM_KMSDRM) {
129 // It's possible to enter this function several times as we're probing VA drivers.
130 // Make sure to only duplicate the DRM FD the first time through.
131 if (m_DrmFd < 0) {
132 // Try to get the FD that we're sharing with SDL
133 bool mustCloseFd = false;
134 int fd = StreamUtils::getDrmFdForWindow(window, &mustCloseFd);
135 if (fd < 0) {
136 // Try to open any DRM render node
137 fd = StreamUtils::getDrmFd(true);
138 if (fd < 0) {
139 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
140 "Failed to open DRM render node: %d",
141 errno);
142 return nullptr;
143 }
144 }
145
146 // If the KMSDRM FD is not a render node FD, open the render node for libva to use.
147 // Since libva 2.20, using a primary node will fail in vaGetDriverNames().
148 if (drmGetNodeTypeFromFd(fd) != DRM_NODE_RENDER) {
149 char* renderNodePath = drmGetRenderDeviceNameFromFd(fd);
150 if (renderNodePath) {
151 // Don't need the primary node FD anymore
152 if (mustCloseFd) {
153 close(fd);
154 }
155
156 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
157 "Opening render node for VAAPI: %s",
158 renderNodePath);
159 m_DrmFd = open(renderNodePath, O_RDWR | O_CLOEXEC);
160 free(renderNodePath);
161 if (m_DrmFd < 0) {
162 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
163 "Failed to open render node: %d",
164 errno);
165 return nullptr;
166 }
167 }
168 else {
169 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
170 "Failed to get render node path. Using the SDL FD directly.");
171 m_DrmFd = mustCloseFd ? fd : dup(fd);
172 }
173 }
174 else {
175 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
176 "KMSDRM FD is already a render node. Using the SDL FD directly.");
177 m_DrmFd = mustCloseFd ? fd : dup(fd);
178 }
179 }
180
181 display = vaGetDisplayDRM(m_DrmFd);
182 if (display == nullptr) {
183 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
184 "Unable to open DRM display for VAAPI");
185 return nullptr;
186 }
187 }
188#endif
189 else {
190 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
191 "Unsupported VAAPI rendering subsystem: %d",
192 info.subsystem);
193 return nullptr;
194 }
195
196 return display;
197}

199VAStatus
200VAAPIRenderer::tryVaInitialize(AVVAAPIDeviceContext* vaDeviceContext, PDECODER_PARAMETERS params, int* major, int* minor)
201{
202 VAStatus status;
203
204 SDL_assert(vaDeviceContext->display == nullptr);
205
206 vaDeviceContext->display = openDisplay(params->window);
207 if (vaDeviceContext->display == nullptr) {
208 // openDisplay() logs the error
209 return VA_STATUS_ERROR_INVALID_DISPLAY;
210 }
211
212 status = vaInitialize(vaDeviceContext->display, major, minor);
213 if (status != VA_STATUS_SUCCESS) {
214 // vaInitialize() stores state into the VADisplay even on failure, so we must still
215 // call vaTerminate() even if vaInitialize() failed. Similarly, calling vaInitialize()
216 // more than once on the same VADisplay can cause resource leaks, even if it failed
217 // in the prior call. https://github.com/intel/libva/issues/741
218 vaTerminate(vaDeviceContext->display);
219 vaDeviceContext->display = nullptr;
220 }
221
222 return status;
223}

227{
228 int err;
229
230 m_Window = params->window;
231 m_VideoFormat = params->videoFormat;
232
233 m_HwContext = av_hwdevice_ctx_alloc(AV_HWDEVICE_TYPE_VAAPI);
234 if (!m_HwContext) {
235 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
236 "Failed to allocate VAAPI context");
237 return false;
238 }

241 AVVAAPIDeviceContext* vaDeviceContext = (AVVAAPIDeviceContext*)deviceContext->hwctx;
242
243 int major, minor;
244 VAStatus status;
245 bool setPathVar = false;
246
247 for (;;) {
248 // vaInitialize() will return the libva library version even if the function
249 // fails. This has been the case since libva v2.6 from 5 years ago. This
250 // doesn't seem to be documented anywhere, so we will be conservative to
251 // protect against changes in libva behavior by reinitializing major/minor
252 // each time and clamping it to the valid range of versions based upon
253 // the version of libva that we compiled with.
254 major = minor = 0;
255 status = tryVaInitialize(vaDeviceContext, params, &major, &minor);
256 if (status != VA_STATUS_SUCCESS) {
257 major = std::max(major, VA_MAJOR_VERSION);
258 minor = std::max(minor, VA_MINOR_VERSION);
259
260 // If LIBVA_DRIVER_NAME has not been set manually and we're running a
261 // version of libva less than 2.20, we'll try our own fallback names.
262 // Beginning in libva 2.20, the driver name detection code is much
263 // more robust than earlier versions and it includes DRI3 support for
264 // driver name detection under Xwayland.
265 if (!qEnvironmentVariableIsEmpty("LIBVA_DRIVER_NAME")) {
266 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
267 "Skipping VAAPI fallback driver names due to LIBVA_DRIVER_NAME");
268 }
269 else if (major > 1 || (major == 1 && minor >= 20)) {
270 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
271 "Skipping VAAPI fallback driver names on libva 2.20+");
272 }
273 else {
274 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
275 "Trying fallback VAAPI driver names");
276
277 // It would be nice to use vaSetDriverName() here, but there's no way to unset
278 // it and get back to the default driver selection logic once we've overridden
279 // the driver name using that API. As a result, we must use LIBVA_DRIVER_NAME.
280
281 if (status != VA_STATUS_SUCCESS) {
282 // The iHD driver supports newer hardware like Ice Lake and Comet Lake.
283 // It should be picked by default on those platforms, but that doesn't
284 // always seem to be the case for some reason.
285 qputenv("LIBVA_DRIVER_NAME", "iHD");
286 status = tryVaInitialize(vaDeviceContext, params, &major, &minor);
287 }
288
289 if (status != VA_STATUS_SUCCESS) {
290 // The Iris driver in Mesa 20.0 returns a bogus VA driver (iris_drv_video.so)
291 // even though the correct driver is still i965. If we hit this path, we'll
292 // explicitly try i965 to handle this case.
293 qputenv("LIBVA_DRIVER_NAME", "i965");
294 status = tryVaInitialize(vaDeviceContext, params, &major, &minor);
295 }
296
297 if (status != VA_STATUS_SUCCESS) {
298 // The RadeonSI driver is compatible with XWayland but can't be detected by libva
299 // so try it too if all else fails.
300 qputenv("LIBVA_DRIVER_NAME", "radeonsi");
301 status = tryVaInitialize(vaDeviceContext, params, &major, &minor);
302 }
303
304 if (status != VA_STATUS_SUCCESS && (m_WindowSystem != SDL_SYSWM_X11 || m_DecoderSelectionPass > 0)) {
305 // The unofficial nvidia VAAPI driver over NVDEC/CUDA works well on Wayland,
306 // but we'd rather use CUDA for XWayland and VDPAU for regular X11.
307 // NB: Remember to update the VA-API NVDEC condition below when modifying this!
308 qputenv("LIBVA_DRIVER_NAME", "nvidia");
309 status = tryVaInitialize(vaDeviceContext, params, &major, &minor);
310 }
311
312 if (status != VA_STATUS_SUCCESS) {
313 // Unset LIBVA_DRIVER_NAME if none of the drivers we tried worked. This ensures
314 // we will get a fresh start using the default driver selection behavior after
315 // setting LIBVA_DRIVERS_PATH in the code below.
316 qunsetenv("LIBVA_DRIVER_NAME");
317 }
318 }
319 }
320
321 if (status == VA_STATUS_SUCCESS) {
322 // Success!
323 break;
324 }
325
326#if defined(APP_IMAGE) || defined(USE_FALLBACK_DRIVER_PATHS)
327 // AppImages will be running with our libva.so which means they don't know about
328 // distro-specific driver paths. To avoid failing in this scenario, we'll hardcode
329 // some such paths here for common distros. Non-AppImage packaging mechanisms won't
330 // need this fallback because either:
331 // a) They are using both distro libva.so and distro libva drivers (native packages)
332 // b) They are using both runtime libva.so and runtime libva drivers (Flatpak/Snap)
333 if (qEnvironmentVariableIsEmpty("LIBVA_DRIVERS_PATH")) {
334 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
335 "Trying fallback VAAPI driver paths");
336
337 qputenv("LIBVA_DRIVERS_PATH",
338 #if Q_PROCESSOR_WORDSIZE == 8
339 "/usr/lib64/dri-nonfree:" // Fedora x86_64
340 "/usr/lib64/dri-freeworld:" // Fedora x86_64
341 "/usr/lib64/dri:" // Fedora x86_64
342 "/usr/lib64/va/drivers:" // Gentoo x86_64
343 #endif
344 "/usr/lib/dri:" // Arch i386 & x86_64, Fedora i386
345 "/usr/lib/va/drivers:" // Gentoo i386
346 #if defined(Q_PROCESSOR_X86_64)
347 "/usr/lib/x86_64-linux-gnu/dri:" // Ubuntu/Debian x86_64
348 #elif defined(Q_PROCESSOR_X86_32)
349 "/usr/lib/i386-linux-gnu/dri:" // Ubuntu/Debian i386
350 #endif
351 );
352 setPathVar = true;
353 }
354 else
355#endif
356 {
357 if (setPathVar) {
358 // Unset LIBVA_DRIVERS_PATH if we set it ourselves
359 // and we didn't find any working VAAPI drivers.
360 qunsetenv("LIBVA_DRIVERS_PATH");
361 }
362
363 // Give up
364 break;
365 }
366 }
367
368 if (status != VA_STATUS_SUCCESS) {
369 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
370 "Failed to initialize VAAPI: %d",
371 status);
372 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
373 return false;
374 }
375
376 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
377 "Initialized VAAPI %d.%d",
378 major, minor);
379
380 const char* vendorString = vaQueryVendorString(vaDeviceContext->display);
381 QString vendorStr(vendorString);
382 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
383 "Driver: %s",
384 vendorString ? vendorString : "<unknown>");
385
386 // This is the libva-vdpau-driver which is not supported by our VAAPI renderer.
387 if (vendorStr.contains("Splitted-Desktop Systems VDPAU backend for VA-API")) {
388 // Fail and let our VDPAU renderer pick this up
389 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
390 "Avoiding VDPAU wrapper for VAAPI decoding");
391 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
392 return false;
393 }
394
395 // The Snap (core22) and Focal/Jammy Mesa drivers have a bug that causes
396 // a large amount of video latency when using more than one reference frame
397 // and severe rendering glitches on my Ryzen 3300U system.
398 m_HasRfiLatencyBug = vendorStr.contains("Gallium", Qt::CaseInsensitive) && qgetenv("IGNORE_RFI_LATENCY_BUG") != "1";
399 if (m_HasRfiLatencyBug) {
400 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
401 "VAAPI driver is affected by RFI latency bug");
402 }
403
404 if (m_DecoderSelectionPass == 0 && qgetenv("FORCE_VAAPI") != "1") {
405 // Older versions of the Gallium VAAPI driver have a nasty memory leak that
406 // causes memory to be leaked for each submitted frame. I believe this is
407 // resolved in the libva2 drivers (VAAPI 1.x). We will try to use VDPAU
408 // instead for old VAAPI versions or drivers affected by the RFI latency bug
409 // as long as we're not streaming HDR (which is unsupported by VDPAU).
410 if ((major == 0 || (m_HasRfiLatencyBug && !(m_VideoFormat & VIDEO_FORMAT_MASK_10BIT))) &&
411 vendorStr.contains("Gallium", Qt::CaseInsensitive)) {
412 // Fail and let VDPAU pick this up
413 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
414 "Deprioritizing VAAPI on Gallium driver. Set FORCE_VAAPI=1 to override.");
415 return false;
416 }
417
418 // Prefer CUDA for XWayland and VDPAU for regular X11.
419 if (m_WindowSystem == SDL_SYSWM_X11 && vendorStr.contains("VA-API NVDEC", Qt::CaseInsensitive)) {
420 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
421 "Deprioritizing VAAPI for NVIDIA driver on X11/XWayland. Set FORCE_VAAPI=1 to override.");
422 return false;
423 }
424 }
425
426 if (WMUtils::isRunningWayland()) {
427 // The iHD VAAPI driver can initialize on XWayland but it crashes in
428 // vaPutSurface() so we must also not directly render on XWayland.
429 m_BlacklistedForDirectRendering = vendorStr.contains("iHD");
430 }
431
432 m_RequiresExplicitPixelFormat = vendorStr.contains("i965");
433
434 // This will populate the driver_quirks
435 err = av_hwdevice_ctx_init(m_HwContext);
436 if (err < 0) {
437 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
438 "Failed to initialize VAAPI context: %d",
439 err);
440 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
441 return false;
442 }
443
444 // Allocate mutex to synchronize overlay updates and rendering
445 m_OverlayMutex = SDL_CreateMutex();
446 if (m_OverlayMutex == nullptr) {
447 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
448 "Failed to create overlay mutex");
449 return false;
450 }
451
452 unsigned int formatCount = vaMaxNumSubpictureFormats(vaDeviceContext->display);
453 if (formatCount != 0) {
454 auto formats = new VAImageFormat[formatCount];
455 auto flags = new unsigned int[formatCount];
456
457 status = vaQuerySubpictureFormats(vaDeviceContext->display, formats, flags, &formatCount);
458 if (status == VA_STATUS_SUCCESS) {
459 for (unsigned int i = 0; i < formatCount; i++) {
460 // Format must have 32-bit color depth
461 if (formats[i].depth != 32) {
462 continue;
463 }
464
465 // Select an RGB format with alpha
466 if (formats[i].byte_order == VA_MSB_FIRST) {
467 switch (formats[i].fourcc) {
468 case VA_FOURCC_RGBA:

470 break;
471 case VA_FOURCC_ARGB:
472 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_ARGB8888;
473 break;
474 case VA_FOURCC_BGRA:
475 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_BGRA8888;
476 break;
477 case VA_FOURCC_ABGR:
478 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_ABGR8888;
479 break;
480 default:
481 continue;
482 }
483 }
484 else {
485 SDL_assert(formats[i].byte_order == VA_LSB_FIRST);
486 switch (formats[i].fourcc) {
487 case VA_FOURCC_RGBA:
488 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_ABGR8888;
489 break;
490 case VA_FOURCC_ARGB:
491 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_BGRA8888;
492 break;
493 case VA_FOURCC_BGRA:
494 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_ARGB8888;
495 break;
496 case VA_FOURCC_ABGR:
497 m_OverlaySdlPixelFormat = SDL_PIXELFORMAT_RGBA8888;
498 break;
499 default:
500 continue;
501 }
502 }
503
504 // If we made it here, we found a format that works for us
505 m_OverlayFormat = formats[i];
506
507 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
508 "Selected overlay subpicture format: %c%c%c%c8888",
509 (m_OverlayFormat.fourcc >> 0) & 0xff,
510 (m_OverlayFormat.fourcc >> 8) & 0xff,
511 (m_OverlayFormat.fourcc >> 16) & 0xff,
512 (m_OverlayFormat.fourcc >> 24) & 0xff);
513 break;
514 }
515 }
516 else {
517 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
518 "vaQuerySubpictureFormats() failed: %d",
519 status);
520 }
521
522 delete[] formats;
523 delete[] flags;
524 }
525
526 return true;

549bool
550VAAPIRenderer::isDirectRenderingSupported()
551{
552 if (qgetenv("VAAPI_FORCE_DIRECT") == "1") {
553 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
554 "Using direct rendering due to environment variable");
555 return true;
556 }
557 else if (qgetenv("VAAPI_FORCE_INDIRECT") == "1") {
558 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
559 "Using indirect rendering due to environment variable");
560 return false;
561 }
562
563 // We only support direct rendering on X11 with VAEntrypointVideoProc support
564 if (m_WindowSystem != SDL_SYSWM_X11 || m_BlacklistedForDirectRendering) {
565 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
566 "Using indirect rendering due to WM or blacklist");
567 return false;
568 }
569 else if (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) {
570 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
571 "Using indirect rendering for 10-bit video");
572 return false;
573 }
574 else if (m_VideoFormat & VIDEO_FORMAT_MASK_YUV444) {
575 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
576 "Using indirect rendering for YUV 4:4:4 video");
577 return false;
578 }
579
580 AVHWDeviceContext* deviceContext = (AVHWDeviceContext*)m_HwContext->data;
581 AVVAAPIDeviceContext* vaDeviceContext = (AVVAAPIDeviceContext*)deviceContext->hwctx;
582 std::vector<VAEntrypoint> entrypoints(vaMaxNumEntrypoints(vaDeviceContext->display));
583 int entrypointCount;
584 VAStatus status = vaQueryConfigEntrypoints(vaDeviceContext->display, VAProfileNone, entrypoints.data(), &entrypointCount);
585 if (status == VA_STATUS_SUCCESS) {
586 for (int i = 0; i < entrypointCount; i++) {
587 // Without VAEntrypointVideoProc support, the driver will crash inside vaPutSurface()
588 if (entrypoints[i] == VAEntrypointVideoProc) {
589 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
590 "Using direct rendering with VAEntrypointVideoProc");
591
592 if (m_OverlayFormat.fourcc == 0) {
593 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
594 "Unable to find supported subpicture format. Overlays will be unavailable!");
595 }
596
597 return true;
598 }
599 }
600 }
601
602 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
603 "Using indirect rendering due to lack of VAEntrypointVideoProc");
604 return false;
605}

626void VAAPIRenderer::notifyOverlayUpdated(Overlay::OverlayType type)
627{
628 AVHWDeviceContext* deviceContext = (AVHWDeviceContext*)m_HwContext->data;
629 AVVAAPIDeviceContext* vaDeviceContext = (AVVAAPIDeviceContext*)deviceContext->hwctx;
630 VAStatus status;
631
632 if (m_OverlayFormat.fourcc == 0) {
633 // We already logged for this in isDirectRenderingSupported()
634 return;
635 }
636
637 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
638 bool overlayEnabled = Session::get()->getOverlayManager().isOverlayEnabled(type);
639 if (newSurface == nullptr && overlayEnabled) {
640 // There's no updated surface and the overlay is enabled, so just leave the old surface alone.
641 return;
642 }
643
644 // Destroy the old image and subpicture
645 // NB: The mutex ensures the overlay is not currently being read for rendering.
646 // NB 2: It is safe to unlock here because this thread is the only surface producer.
647 SDL_LockMutex(m_OverlayMutex);
648 VAImageID oldImageId = m_OverlayImage[type].image_id;
649 SDL_zero(m_OverlayImage[type]);
650
651 VASubpictureID oldSubpictureId = m_OverlaySubpicture[type];
652 m_OverlaySubpicture[type] = 0;
653 SDL_UnlockMutex(m_OverlayMutex);
654
655 if (oldSubpictureId != 0) {
656 status = vaDestroySubpicture(vaDeviceContext->display, oldSubpictureId);
657 if (status != VA_STATUS_SUCCESS) {
658 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
659 "vaDestroySubpicture() failed: %d",
660 status);
661 }
662 }
663 if (oldImageId != 0) {
664 status = vaDestroyImage(vaDeviceContext->display, oldImageId);
665 if (status != VA_STATUS_SUCCESS) {
666 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
667 "vaDestroyImage() failed: %d",
668 status);
669 }
670 }
671
672 if (!overlayEnabled) {
673 SDL_FreeSurface(newSurface);
674 return;
675 }
676
677 if (newSurface != nullptr) {
678 VAImage newImage;
679
680 SDL_assert(!SDL_MUSTLOCK(newSurface));
681
682 status = vaCreateImage(vaDeviceContext->display, &m_OverlayFormat, newSurface->w, newSurface->h, &newImage);
683 if (status != VA_STATUS_SUCCESS) {
684 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
685 "vaCreateImage() failed: %d",
686 status);
687 SDL_FreeSurface(newSurface);
688 return;
689 }
690
691 void* imagePixels;
692 status = vaMapBuffer(vaDeviceContext->display, newImage.buf, &imagePixels);
693 if (status != VA_STATUS_SUCCESS) {
694 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
695 "vaMapBuffer() failed: %d",
696 status);
697 SDL_FreeSurface(newSurface);
698 vaDestroyImage(vaDeviceContext->display, newImage.image_id);
699 return;
700 }
701
702 // Convert the surface to the proper format for the VAImage
703 SDL_ConvertPixels(newSurface->w, newSurface->h, newSurface->format->format,
704 newSurface->pixels, newSurface->pitch, m_OverlaySdlPixelFormat,
705 imagePixels, (int)newImage.pitches[0]);
706
707 status = vaUnmapBuffer(vaDeviceContext->display, newImage.buf);
708 if (status != VA_STATUS_SUCCESS) {
709 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
710 "vaUnmapBuffer() failed: %d",
711 status);
712 SDL_FreeSurface(newSurface);
713 vaDestroyImage(vaDeviceContext->display, newImage.image_id);
714 return;
715 }
716
717 SDL_Rect overlayRect;
718
719 if (type == Overlay::OverlayStatusUpdate) {
720 // Bottom Left
721 overlayRect.x = 0;
722 overlayRect.y = -newSurface->h;
723 }
724 else if (type == Overlay::OverlayDebug) {
725 // Top left
726 overlayRect.x = 0;
727 overlayRect.y = 0;
728 }
729
730 overlayRect.w = newSurface->w;
731 overlayRect.h = newSurface->h;
732
733 // Surface data is no longer needed
734 SDL_FreeSurface(newSurface);
735
736 VASubpictureID newSubpicture;
737 status = vaCreateSubpicture(vaDeviceContext->display, newImage.image_id, &newSubpicture);
738 if (status != VA_STATUS_SUCCESS) {
739 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
740 "vaCreateSubpicture() failed: %d",
741 status);
742 vaDestroyImage(vaDeviceContext->display, newImage.image_id);
743 return;
744 }
745
746 SDL_LockMutex(m_OverlayMutex);
747 m_OverlayImage[type] = newImage;
748 m_OverlaySubpicture[type] = newSubpicture;
749 m_OverlayRect[type] = overlayRect;
750 SDL_UnlockMutex(m_OverlayMutex);
751 }

760void
761VAAPIRenderer::renderFrame(AVFrame* frame)
762{
763 VASurfaceID surface = (VASurfaceID)(uintptr_t)frame->data[3];
764 AVHWDeviceContext* deviceContext = (AVHWDeviceContext*)m_HwContext->data;
765 AVVAAPIDeviceContext* vaDeviceContext = (AVVAAPIDeviceContext*)deviceContext->hwctx;
766
767 int windowWidth, windowHeight;
768 SDL_GetWindowSize(m_Window, &windowWidth, &windowHeight);
769
770 SDL_Rect src, dst;
771 src.x = src.y = 0;
772 src.w = frame->width;
773 src.h = frame->height;
774 dst.x = dst.y = 0;
775 dst.w = windowWidth;
776 dst.h = windowHeight;
777
778 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
779
780 if (m_WindowSystem == SDL_SYSWM_X11) {
781#ifdef HAVE_LIBVA_X11
782 unsigned int flags = 0;
783
784 // NB: Not all VAAPI drivers respect these flags. Many drivers
785 // just ignore them and do the color conversion as Rec 601.
786 switch (getFrameColorspace(frame)) {
787 case COLORSPACE_REC_709:
788 flags |= VA_SRC_BT709;
789 break;
790 case COLORSPACE_REC_601:
791 flags |= VA_SRC_BT601;
792 break;
793 default:
794 // Unsupported colorspace
795 SDL_assert(false);
796 break;
797 }
798
799 SDL_LockMutex(m_OverlayMutex);
800
801 VAImage associatedOverlayImages[Overlay::OverlayMax] = {};
802 VASubpictureID associatedOverlaySubpictures[Overlay::OverlayMax] = {};
803
804 // Associate our overlay subpictures to the current surface
805 for (int type = 0; type < Overlay::OverlayMax; type++) {
806 VAStatus status;
807
808 if (m_OverlaySubpicture[type] == 0) {
809 continue;
810 }
811
812 SDL_Rect overlayRect = m_OverlayRect[type];
813
814 // Negative values are relative to the other side of the window
815 if (overlayRect.x < 0) {
816 overlayRect.x += windowWidth;
817 }
818 if (overlayRect.y < 0) {
819 overlayRect.y += windowHeight;
820 }
821
822 status = vaAssociateSubpicture(vaDeviceContext->display,
823 m_OverlaySubpicture[type],
824 &surface,
825 1,
826 0,
827 0,
828 m_OverlayImage[type].width,
829 m_OverlayImage[type].height,
830 overlayRect.x,
831 overlayRect.y,
832 overlayRect.w,
833 overlayRect.h,
834 0);
835 if (status == VA_STATUS_SUCCESS) {
836 // Take temporary ownership of the overlay to prevent notifyOverlayUpdated()
837 // from freeing them frum underneath us. We need to release the lock while
838 // we render for performance reasons.
839 associatedOverlayImages[type] = m_OverlayImage[type];
840 associatedOverlaySubpictures[type] = m_OverlaySubpicture[type];
841
842 SDL_zero(m_OverlayImage[type]);
843 m_OverlaySubpicture[type] = 0;
844 }
845 else {
846 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
847 "vaAssociateSubpicture() failed: %d",
848 status);
849 }
850 }
851
852 SDL_UnlockMutex(m_OverlayMutex);
853
854 // This will draw the surface and any associated subpictures
855 // NB: This can take a full VBlank period to complete!
856 vaPutSurface(vaDeviceContext->display,
857 surface,
858 m_XWindow,
859 0, 0,
860 frame->width, frame->height,
861 dst.x, dst.y,
862 dst.w, dst.h,
863 NULL, 0, flags);
864
865 SDL_LockMutex(m_OverlayMutex);
866
867 // Now that we've reacquired the lock, we will need to reconcile the current
868 // state of the overlay with our saved state from before we unlocked.
869 for (int type = 0; type < Overlay::OverlayMax; type++) {
870 VAStatus status;
871
872 if (associatedOverlaySubpictures[type] == 0) {
873 continue;
874 }
875
876 // Deassociate the subpicture so it can be safely destroyed/replaced
877 status = vaDeassociateSubpicture(vaDeviceContext->display, associatedOverlaySubpictures[type], &surface, 1);
878 if (status != VA_STATUS_SUCCESS) {
879 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
880 "vaDeassociateSubpicture() failed: %d",
881 status);
882 }
883
884 // If a new subpicture was populated while we were unlocked, free the old one we took ownership of
885 if (m_OverlaySubpicture[type] != 0) {
886 status = vaDestroySubpicture(vaDeviceContext->display, associatedOverlaySubpictures[type]);
887 if (status != VA_STATUS_SUCCESS) {
888 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
889 "vaDestroySubpicture() failed: %d",
890 status);
891 }
892 }
893 else {
894 // If no new subpicture was populated, return ownership of this one
895 m_OverlaySubpicture[type] = associatedOverlaySubpictures[type];
896 }
897
898 // If a new image was populated while we were unlocked, free the one old we took ownership of
899 if (m_OverlayImage[type].image_id != 0) {
900 status = vaDestroyImage(vaDeviceContext->display, associatedOverlayImages[type].image_id);
901 if (status != VA_STATUS_SUCCESS) {
902 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
903 "vaDestroyImage() failed: %d",
904 status);
905 }
906 }
907 else {
908 // If no new image was populated, return ownership of this one
909 m_OverlayImage[type] = associatedOverlayImages[type];
910 }
911 }
912
913 SDL_UnlockMutex(m_OverlayMutex);
914#endif
915 }
916 else if (m_WindowSystem == SDL_SYSWM_WAYLAND) {
917 // We don't support direct rendering on Wayland, so we should
918 // never get called there. Many common Wayland compositors don't
919 // support YUV surfaces, so direct rendering would fail.
920 SDL_assert(false);
921 }
922 else {
923 // We don't accept anything else in initialize().
924 SDL_assert(false);
925 }
926}

930// Ensure that vaExportSurfaceHandle() is supported by the VA-API driver
931bool
932VAAPIRenderer::canExportSurfaceHandle(int layerTypeFlag, VADRMPRIMESurfaceDescriptor* descriptor) {
933 AVHWDeviceContext* deviceContext = (AVHWDeviceContext*)m_HwContext->data;
934 AVVAAPIDeviceContext* vaDeviceContext = (AVVAAPIDeviceContext*)deviceContext->hwctx;
935 VASurfaceID surfaceId;
936 VAStatus st;
937 VASurfaceAttrib attrs[2];
938 int attributeCount = 0;
939
940 if (qgetenv("VAAPI_FORCE_DIRECT") == "1") {
941 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
942 "Using direct rendering due to environment variable");
943 return false;
944 }
945
946 // FFmpeg handles setting these quirk flags for us
947 if (!(vaDeviceContext->driver_quirks & AV_VAAPI_DRIVER_QUIRK_ATTRIB_MEMTYPE)) {
948 attrs[attributeCount].type = VASurfaceAttribMemoryType;
949 attrs[attributeCount].flags = VA_SURFACE_ATTRIB_SETTABLE;
950 attrs[attributeCount].value.type = VAGenericValueTypeInteger;
951 attrs[attributeCount].value.value.i = VA_SURFACE_ATTRIB_MEM_TYPE_VA;
952 attributeCount++;
953 }
954
955 // These attributes are required for i965 to create a surface that can
956 // be successfully exported via vaExportSurfaceHandle(). YUV444 is not
957 // handled here but i965 supports no hardware with YUV444 decoding.
958 if (m_RequiresExplicitPixelFormat && !(m_VideoFormat & VIDEO_FORMAT_MASK_YUV444)) {
959 attrs[attributeCount].type = VASurfaceAttribPixelFormat;
960 attrs[attributeCount].flags = VA_SURFACE_ATTRIB_SETTABLE;
961 attrs[attributeCount].value.type = VAGenericValueTypeInteger;
962 attrs[attributeCount].value.value.i =
963 (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) ? VA_FOURCC_P010 : VA_FOURCC_NV12;
964 attributeCount++;
965 }
966
967 st = vaCreateSurfaces(vaDeviceContext->display,
968 (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) ?
969 ((m_VideoFormat & VIDEO_FORMAT_MASK_YUV444) ? VA_RT_FORMAT_YUV444_10 : VA_RT_FORMAT_YUV420_10) :
970 ((m_VideoFormat & VIDEO_FORMAT_MASK_YUV444) ? VA_RT_FORMAT_YUV444 : VA_RT_FORMAT_YUV420),
971 1280,
972 720,
973 &surfaceId,
974 1,
975 attrs,
976 attributeCount);
977 if (st != VA_STATUS_SUCCESS) {
978 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
979 "vaCreateSurfaces() failed: %d", st);
980 return false;
981 }
982
983 st = vaExportSurfaceHandle(vaDeviceContext->display,
984 surfaceId,
985 VA_SURFACE_ATTRIB_MEM_TYPE_DRM_PRIME_2,
986 VA_EXPORT_SURFACE_READ_ONLY | layerTypeFlag,
987 descriptor);
988
989 vaDestroySurfaces(vaDeviceContext->display, &surfaceId, 1);
990
991 if (st != VA_STATUS_SUCCESS) {
992 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
993 "vaExportSurfaceHandle() failed: %d", st);
994 return false;
995 }
996
997 for (size_t i = 0; i < descriptor->num_objects; ++i) {
998 close(descriptor->objects[i].fd);
999 descriptor->objects[i].fd = -1;
1000 }
1001
1002 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1003 "VAAPI driver supports exporting DRM PRIME surface handles with %s layers",
1004 layerTypeFlag == VA_EXPORT_SURFACE_COMPOSED_LAYERS ? "composed" : "separate");
1005 return true;
1006}

1041bool
1042VAAPIRenderer::initializeEGL(EGLDisplay dpy,
1043 const EGLExtensions &ext) {
1044 VADRMPRIMESurfaceDescriptor descriptor;
1045
1046 if (!m_EglImageFactory.initializeEGL(dpy, ext)) {
1047 return false;
1048 }
1049
1050 // Prefer exporting composed images absent a user override or lack of support for exporting or importing
1051 if (qgetenv("VAAPI_EGL_SEPARATE_LAYERS") == "1") {
1052 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1053 "Exporting separate layers due to environment variable override");
1054 m_EglExportType = EglExportType::Separate;
1055 }
1056 else if (!canExportSurfaceHandle(VA_EXPORT_SURFACE_COMPOSED_LAYERS, &descriptor)) {
1057 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1058 "Exporting separate layers due to lack of support for VA_EXPORT_SURFACE_COMPOSED_LAYERS");
1059 m_EglExportType = EglExportType::Separate;
1060 }
1061 else if (!m_EglImageFactory.supportsImportingFormat(dpy, descriptor.layers[0].drm_format)) {
1062 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1063 "Exporting separate layers due to lack of support for importing format: %08x", descriptor.layers[0].drm_format);
1064 m_EglExportType = EglExportType::Separate;
1065 }
1066 else if (!m_EglImageFactory.supportsImportingModifier(dpy, descriptor.layers[0].drm_format, descriptor.objects[0].drm_format_modifier)) {
1067 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1068 "Exporting separate layers due to lack of support for importing format and modifier: %08x %016" PRIx64,
1069 descriptor.layers[0].drm_format,
1070 descriptor.objects[0].drm_format_modifier);
1071 m_EglExportType = EglExportType::Separate;
1072 }
1073 else {
1074 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
1075 "Exporting composed layers with format and modifier: %08x %016" PRIx64,
1076 descriptor.layers[0].drm_format,
1077 descriptor.objects[0].drm_format_modifier);
1078 m_EglExportType = EglExportType::Composed;
1079 }
1080
1081 // Let's probe for EGL import support on separate layers too, but only warn if it's not supported
1082 if (m_EglExportType == EglExportType::Separate) {
1083 if (!canExportSurfaceHandle(VA_EXPORT_SURFACE_SEPARATE_LAYERS, &descriptor)) {
1084 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1085 "Exporting separate layers is not supported by the VAAPI driver");
1086 return false;
1087 }
1088
1089 for (uint32_t i = 0; i < descriptor.num_layers; i++) {
1090 if (!m_EglImageFactory.supportsImportingFormat(dpy, descriptor.layers[i].drm_format)) {
1091 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
1092 "EGL implementation lacks support for importing format: %08x", descriptor.layers[0].drm_format);
1093 }
1094 else if (!m_EglImageFactory.supportsImportingModifier(dpy, descriptor.layers[i].drm_format,
1095 descriptor.objects[descriptor.layers[i].object_index[0]].drm_format_modifier)) {
1096 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
1097 "EGL implementation lacks support for importing format and modifier: %08x %016" PRIx64,
1098 descriptor.layers[i].drm_format,
1099 descriptor.objects[descriptor.layers[i].object_index[0]].drm_format_modifier);
1100 }
1101 }
1102 }
1103
1104 return true;
