19QVariant ComputerModel::data(const QModelIndex& index, int role) const
20{
21 if (!index.isValid()) {
22 return QVariant();
23 }
24
25 Q_ASSERT(index.row() < m_Computers.count());
26
27 NvComputer* computer = m_Computers[index.row()];
28 QReadLocker lock(&computer->lock);
29
30 switch (role) {
31 case NameRole:
32 return computer->name;
33 case OnlineRole:
34 return computer->state == NvComputer::CS_ONLINE;
35 case PairedRole:
36 return computer->pairState == NvComputer::PS_PAIRED;
37 case BusyRole:
38 return computer->currentGameId != 0;
39 case WakeableRole:
40 return !computer->macAddress.isEmpty();
41 case StatusUnknownRole:
42 return computer->state == NvComputer::CS_UNKNOWN;
43 case ServerSupportedRole:
44 return computer->isSupportedServerVersion;
45 case DetailsRole: {
46 QString state, pairState;
47
48 switch (computer->state) {
49 case NvComputer::CS_ONLINE:
50 state = tr("Online");
51 break;
52 case NvComputer::CS_OFFLINE:
53 state = tr("Offline");
54 break;
55 default:
56 state = tr("Unknown");
57 break;
58 }
59
60 switch (computer->pairState) {
61 case NvComputer::PS_PAIRED:
62 pairState = tr("Paired");
63 break;
64 case NvComputer::PS_NOT_PAIRED:
65 pairState = tr("Unpaired");
66 break;
67 default:
68 pairState = tr("Unknown");
69 break;
70 }
71
72 return tr("Name: %1").arg(computer->name) + '\n' +
73 tr("Status: %1").arg(state) + '\n' +
74 tr("Active Address: %1").arg(computer->activeAddress.toString()) + '\n' +
75 tr("UUID: %1").arg(computer->uuid) + '\n' +
76 tr("Local Address: %1").arg(computer->localAddress.toString()) + '\n' +
77 tr("Remote Address: %1").arg(computer->remoteAddress.toString()) + '\n' +
78 tr("IPv6 Address: %1").arg(computer->ipv6Address.toString()) + '\n' +
79 tr("Manual Address: %1").arg(computer->manualAddress.toString()) + '\n' +
80 tr("MAC Address: %1").arg(computer->macAddress.isEmpty() ? tr("Unknown") : QString(computer->macAddress.toHex(':'))) + '\n' +
81 tr("Pair State: %1").arg(pairState) + '\n' +
82 tr("Running Game ID: %1").arg(computer->state == NvComputer::CS_ONLINE ? QString::number(computer->currentGameId) : tr("Unknown")) + '\n' +
83 tr("HTTPS Port: %1").arg(computer->state == NvComputer::CS_ONLINE ? QString::number(computer->activeHttpsPort) : tr("Unknown"));
84 }
85 default:
86 return QVariant();

117Session* ComputerModel::createSessionForCurrentGame(int computerIndex)
118{
119 Q_ASSERT(computerIndex < m_Computers.count());
120
121 NvComputer* computer = m_Computers[computerIndex];
122
123 // We must currently be streaming a game to use this function
124 Q_ASSERT(computer->currentGameId != 0);
125
126 for (NvApp& app : computer->appList) {
127 if (app.id == computer->currentGameId) {
128 return new Session(computer, app);
129 }
130 }
131
132 // We have a current running app but it's not in our app list
133 Q_ASSERT(false);
134 return nullptr;
135}
