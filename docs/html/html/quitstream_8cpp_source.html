48 void handleEvent(Event event)
49 {
50 Q_Q(Launcher);
51 NvApp app;
52
53 switch (event.type) {
54 // Occurs when CliQuitStreamSegue becomes visible and the UI calls launcher's execute()
55 case Event::Executed:
56 if (m_State == StateInit) {
57 m_State = StateSeekComputer;
58 m_ComputerManager = event.computerManager;
59 q->connect(m_ComputerManager, &ComputerManager::quitAppCompleted,
60 q, &Launcher::onQuitAppCompleted);
61
62 m_ComputerSeeker = new ComputerSeeker(m_ComputerManager, m_ComputerName, q);
63 q->connect(m_ComputerSeeker, &ComputerSeeker::computerFound,
64 q, &Launcher::onComputerFound);
65 q->connect(m_ComputerSeeker, &ComputerSeeker::errorTimeout,
66 q, &Launcher::onComputerSeekTimeout);
67 m_ComputerSeeker->start(COMPUTER_SEEK_TIMEOUT);
68
69 emit q->searchingComputer();
70 }
71 break;
72 // Occurs when computer search timed out
73 case Event::ComputerSeekTimedout:
74 if (m_State == StateSeekComputer) {
75 m_State = StateFailure;
76 emit q->failed(QObject::tr("Failed to connect to %1").arg(m_ComputerName));
77 }
78 break;
79 // Occurs when searched computer is found
80 case Event::ComputerFound:
81 if (m_State == StateSeekComputer) {
82 if (event.computer->pairState == NvComputer::PS_PAIRED) {
83 m_State = StateQuitApp;
84 emit q->quittingApp();
85 m_ComputerManager->quitRunningApp(event.computer);
86 } else {
87 m_State = StateFailure;
88 QString msg = QObject::tr("Computer %1 has not been paired. "
89 "Please open Moonlight to pair before streaming.")
90 .arg(event.computer->name);
91 emit q->failed(msg);
92 }
93 }
94 break;
95 // Occurs when app quit completed (error message is set if failed)
96 case Event::AppQuitCompleted:
97 if (m_State == StateQuitApp) {
98 if (event.errorMessage.isEmpty()) {
99 QCoreApplication::exit(0);
100 } else {
101 m_State = StateFailure;
102 emit q->failed(QObject::tr("Quitting app failed, reason: %1").arg(event.errorMessage));
103 }
104 }
105 break;
106 }
107 }
