67QString Path::getDataFilePath(QString fileName)
68{
69 QString candidatePath;
70
71 // Check the cache location first (used by Path::writeDataFile())
72 candidatePath = QDir(s_CacheDir).absoluteFilePath(fileName);
73 if (QFile::exists(candidatePath)) {
74 qInfo() << "Found" << fileName << "at" << candidatePath;
75 return candidatePath;
76 }
77
78 // Check the current directory
79 candidatePath = QDir(QDir::currentPath()).absoluteFilePath(fileName);
80 if (QFile::exists(candidatePath)) {
81 qInfo() << "Found" << fileName << "at" << candidatePath;
82 return candidatePath;
83 }
84
85 // Now check the data directories (for Linux, in particular)
86 candidatePath = QStandardPaths::locate(QStandardPaths::AppDataLocation, fileName);
87 if (!candidatePath.isEmpty() && QFile::exists(candidatePath)) {
88 qInfo() << "Found" << fileName << "at" << candidatePath;
89 return candidatePath;
90 }
91
92 // Now try the directory of our app installation (for Windows, if current dir doesn't find it)
93 candidatePath = QDir(QCoreApplication::applicationDirPath()).absoluteFilePath(fileName);
94 if (QFile::exists(candidatePath)) {
95 qInfo() << "Found" << fileName << "at" << candidatePath;
96 return candidatePath;
97 }
98
99 // Return the QRC embedded copy
100 candidatePath = ":/data/" + fileName;
101 qInfo() << "Found" << fileName << "at" << candidatePath;
102 return QString(candidatePath);
103}
