20{

22
23 SDL_zero(want);
24 want.freq = opusConfig->sampleRate;
25 want.format = AUDIO_F32SYS;
26 want.channels = opusConfig->channelCount;

28 // On PulseAudio systems, setting a value too small can cause underruns for other
29 // applications sharing this output device. We impose a floor of 480 samples (10 ms)
30 // to mitigate this issue. Otherwise, we will buffer up to 3 frames of audio which
31 // is 15 ms at regular 5 ms frames and 30 ms at 10 ms frames for slow connections.
32 // The buffering helps avoid audio underruns due to network jitter.
33 want.samples = SDL_max(480, opusConfig->samplesPerFrame * 3);
34

36 opusConfig->channelCount *
37 getAudioBufferSampleSize();
38

40 if (m_AudioDevice == 0) {
41 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
42 "Failed to open audio device: %s",
43 SDL_GetError());
44 return false;
45 }
46
47 m_AudioBuffer = SDL_malloc(m_FrameSize);
48 if (m_AudioBuffer == nullptr) {
49 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
50 "Failed to allocate audio buffer");
51 return false;
52 }
53
54 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
55 "Desired audio buffer: %u samples (%u bytes)",
56 want.samples,
57 want.samples * want.channels * getAudioBufferSampleSize());

95bool SdlAudioRenderer::submitAudio(int bytesWritten)
96{
97 if (bytesWritten == 0) {
98 // Nothing to do
99 return true;
100 }
101
102 // Don't queue if there's already more than 30 ms of audio data waiting
103 // in Moonlight's audio queue.
104 if (LiGetPendingAudioDuration() > 30) {
105 return true;
106 }
107
108 // Provide backpressure on the queue to ensure too many frames don't build up
109 // in SDL's audio queue, but don't wait forever to avoid a deadlock if the
110 // audio device fails.
111 for (int i = 0; i < 100; i++) {
112 // Our device may enter a permanent error status upon removal, so we need
113 // to recreate the audio device to pick up the new default audio device.
114 if (SDL_GetAudioDeviceStatus(m_AudioDevice) == SDL_AUDIO_STOPPED) {
115 return false;
116 }
117
118 // Only queue more samples where there are 10 frames or less in SDL's queue
119 if (SDL_GetQueuedAudioSize(m_AudioDevice) / m_FrameSize <= 10) {
120 break;
121 }
122
123 SDL_Delay(1);
124 }
125
126 if (SDL_QueueAudio(m_AudioDevice, m_AudioBuffer, bytesWritten) < 0) {
127 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
128 "Failed to queue audio sample: %s",
129 SDL_GetError());
130 }
131
132 return true;
133}
