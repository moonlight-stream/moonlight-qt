122QString
123NvHTTP::getServerInfo(NvLogLevel logLevel, bool fastFail)
124{
125 QString serverInfo;
126
127 // Check if we have a pinned cert and HTTPS port for this host yet
128 if (!m_ServerCert.isNull() && httpsPort() != 0)
129 {
130 try
131 {
132 // Always try HTTPS first, since it properly reports
133 // pairing status (and a few other attributes).
134 serverInfo = openConnectionToString(m_BaseUrlHttps,
135 "serverinfo",
136 nullptr,
137 fastFail ? FAST_FAIL_TIMEOUT_MS : REQUEST_TIMEOUT_MS,
138 logLevel);
139 // Throws if the request failed
140 verifyResponseStatus(serverInfo);
141 }
142 catch (const GfeHttpResponseException& e)
143 {
144 if (e.getStatusCode() == 401)
145 {
146 // Certificate validation error, fallback to HTTP
147 serverInfo = openConnectionToString(m_BaseUrlHttp,
148 "serverinfo",
149 nullptr,
150 fastFail ? FAST_FAIL_TIMEOUT_MS : REQUEST_TIMEOUT_MS,
151 logLevel);
152 verifyResponseStatus(serverInfo);
153 }
154 else
155 {
156 // Rethrow real errors
157 throw e;
158 }
159 }
160 }
161 else
162 {
163 // Only use HTTP prior to pairing or fetching HTTPS port
164 serverInfo = openConnectionToString(m_BaseUrlHttp,
165 "serverinfo",
166 nullptr,
167 fastFail ? FAST_FAIL_TIMEOUT_MS : REQUEST_TIMEOUT_MS,
168 logLevel);
169 verifyResponseStatus(serverInfo);
170
171 // Populate the HTTPS port
172 uint16_t httpsPort = getXmlString(serverInfo, "HttpsPort").toUShort();
173 if (httpsPort == 0) {
174 httpsPort = DEFAULT_HTTPS_PORT;
175 }
176 setHttpsPort(httpsPort);
177
178 // If we just needed to determine the HTTPS port, we'll try again over
179 // HTTPS now that we have the port number
180 if (!m_ServerCert.isNull()) {
181 return getServerInfo(logLevel, fastFail);
182 }
183 }
184
185 return serverInfo;
186}

328void
329NvHTTP::verifyResponseStatus(QString xml)
330{
331 QXmlStreamReader xmlReader(xml);
332
333 while (xmlReader.readNextStartElement())
334 {
335 if (xmlReader.name() == QString("root"))
336 {
337 // Status code can be 0xFFFFFFFF in some rare cases on GFE 3.20.3, and
338 // QString::toInt() will fail in that case, so use QString::toUInt()
339 // and cast the result to an int instead.
340 int statusCode = (int)xmlReader.attributes().value("status_code").toUInt();
341 if (statusCode == 200)
342 {
343 // Successful
344 return;
345 }
346 else
347 {
348 QString statusMessage = xmlReader.attributes().value("status_message").toString();
349 if (statusCode != 401) {
350 // 401 is expected for unpaired PCs when we fetch serverinfo over HTTPS
351 qWarning() << "Request failed:" << statusCode << statusMessage;
352 }
353 if (statusCode == -1 && statusMessage == "Invalid") {
354 // Special case handling an audio capture error which GFE doesn't
355 // provide any useful status message for.
356 statusCode = 418;
357 statusMessage = tr("Missing audio capture device. Reinstalling GeForce Experience should resolve this error.");
358 }
359 throw GfeHttpResponseException(statusCode, statusMessage);
360 }
361 }
362 }
363
364 throw GfeHttpResponseException(-1, "Malformed XML (missing root element)");
365}

417void NvHTTP::handleSslErrors(QNetworkReply* reply, const QList<QSslError>& errors)
418{
419 bool ignoreErrors = true;
420
421 if (m_ServerCert.isNull()) {
422 // We should never make an HTTPS request without a cert
423 Q_ASSERT(!m_ServerCert.isNull());
424 return;
425 }
426
427 for (const QSslError& error : errors) {
428 if (m_ServerCert != error.certificate()) {
429 ignoreErrors = false;
430 break;
431 }
432 }
433
434 if (ignoreErrors) {
435 reply->ignoreSslErrors(errors);

463QNetworkReply*
464NvHTTP::openConnection(QUrl baseUrl,
465 QString command,
466 QString arguments,
467 int timeoutMs,
468 NvLogLevel logLevel)
469{
470 // Port must be set
471 Q_ASSERT(baseUrl.port(0) != 0);
472
473 // Build a URL for the request
474 QUrl url(baseUrl);
475 url.setPath("/" + command);
476
477 // Use a common UID for Moonlight clients to allow them to quit
478 // games for each other (otherwise GFE gets screwed up and it requires
479 // manual intervention to solve).
480 url.setQuery("uniqueid=0123456789ABCDEF&uuid=" +
481 QUuid::createUuid().toRfc4122().toHex() +
482 ((arguments != nullptr) ? ("&" + arguments) : ""));
483
484 QNetworkRequest request(url);
485
486 // Add our client certificate
487 request.setSslConfiguration(IdentityManager::get()->getSslConfig());
488
489#if QT_VERSION >= QT_VERSION_CHECK(6, 0, 0)
490 // Disable HTTP/2 (GFE 3.22 doesn't like it) and Qt 6 enables it by default
491 request.setAttribute(QNetworkRequest::Http2AllowedAttribute, false);
492#endif
493
494#if QT_VERSION >= QT_VERSION_CHECK(6, 3, 0)
495 // Use fine-grained idle timeouts to avoid calling QNetworkAccessManager::clearAccessCache(),
496 // which tears down the NAM's global thread each time. We must not keep persistent connections
497 // or GFE will puke.
498 request.setAttribute(QNetworkRequest::ConnectionCacheExpiryTimeoutSecondsAttribute, 0);
499#endif
500
501 auto sslErrorsConnection = connect(m_Nam, &QNetworkAccessManager::sslErrors, this, &NvHTTP::handleSslErrors);
502 QNetworkReply* reply = m_Nam->get(request);
503
504 // Run the request with a timeout if requested
505 QEventLoop loop;
506 connect(reply, &QNetworkReply::finished, &loop, &QEventLoop::quit);
507 connect(QCoreApplication::instance(), &QCoreApplication::aboutToQuit, &loop, &QEventLoop::quit);
508 if (timeoutMs) {
509 QTimer::singleShot(timeoutMs, &loop, &QEventLoop::quit);
510 }
511 if (logLevel >= NvLogLevel::NVLL_VERBOSE) {
512 qInfo() << "Executing request:" << url.toString();
513 }
514 loop.exec(QEventLoop::ExcludeUserInputEvents);
515
516 // Abort the request if it timed out
517 if (!reply->isFinished())
518 {
519 if (logLevel >= NvLogLevel::NVLL_ERROR) {
520 qWarning() << "Aborting timed out request for" << url.toString();
521 }
522 reply->abort();
523 }
524
525#if QT_VERSION < QT_VERSION_CHECK(6, 3, 0)
526 // If we couldn't use fine-grained connection idle timeouts, kill them all now
527 m_Nam->clearAccessCache();
528#endif
529 disconnect(sslErrorsConnection);
530
531 // Handle error
532 if (reply->error() != QNetworkReply::NoError)
533 {
534 if (logLevel >= NvLogLevel::NVLL_ERROR) {
535 qWarning() << command << "request failed with error:" << reply->error();
536 }
537
538 if (reply->error() == QNetworkReply::SslHandshakeFailedError) {
539 // This will trigger falling back to HTTP for the serverinfo query
540 // then pairing again to get the updated certificate.
541 GfeHttpResponseException exception(401, "Server certificate mismatch");
542 delete reply;
543 throw exception;
544 }
545 else if (reply->error() == QNetworkReply::OperationCanceledError) {
546 QtNetworkReplyException exception(QNetworkReply::TimeoutError, "Request timed out");
547 delete reply;
548 throw exception;
549 }
550 else {
551 QtNetworkReplyException exception(reply->error(), reply->errorString());
552 delete reply;
553 throw exception;
554 }
555 }
556
557 return reply;
