135NvPairingManager::verifySignature(const QByteArray& data, const QByteArray& signature, const QByteArray& serverCertificate)
136{
137#if (OPENSSL_VERSION_NUMBER < 0x10100000L)
138 BIO* bio = BIO_new_mem_buf(const_cast<char*>(serverCertificate.data()), -1);
139#else
140 BIO* bio = BIO_new_mem_buf(serverCertificate.data(), -1);
141#endif
142 THROW_BAD_ALLOC_IF_NULL(bio);
143
144 X509* cert = PEM_read_bio_X509(bio, nullptr, nullptr, nullptr);
145 BIO_free_all(bio);
146
147 EVP_PKEY* pubKey = X509_get_pubkey(cert);
148 THROW_BAD_ALLOC_IF_NULL(pubKey);
149
150 EVP_MD_CTX* mdctx = EVP_MD_CTX_create();
151 THROW_BAD_ALLOC_IF_NULL(mdctx);
152
153 EVP_DigestVerifyInit(mdctx, nullptr, EVP_sha256(), nullptr, pubKey);
154 EVP_DigestVerifyUpdate(mdctx, data.data(), data.length());
155 int result = EVP_DigestVerifyFinal(mdctx, reinterpret_cast<unsigned char*>(const_cast<char*>(signature.data())), signature.length());
156
157 EVP_PKEY_free(pubKey);
158 EVP_MD_CTX_destroy(mdctx);
159 X509_free(cert);
160
161 return result > 0;
162}

172
