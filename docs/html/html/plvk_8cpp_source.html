157bool PlVkRenderer::chooseVulkanDevice(PDECODER_PARAMETERS params, bool hdrOutputRequired)
158{
159 uint32_t physicalDeviceCount = 0;
160 fn_vkEnumeratePhysicalDevices(m_PlVkInstance->instance, &physicalDeviceCount, nullptr);
161 std::vector<VkPhysicalDevice> physicalDevices(physicalDeviceCount);
162 fn_vkEnumeratePhysicalDevices(m_PlVkInstance->instance, &physicalDeviceCount, physicalDevices.data());
163
164 std::set<uint32_t> devicesTried;
165 VkPhysicalDeviceProperties deviceProps;
166
167 if (physicalDeviceCount == 0) {
168 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
169 "No Vulkan devices found!");
170 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
171 return false;
172 }
173
174 // First, try the first device in the list to support device selection layers
175 // that put the user's preferred GPU in the first slot.
176 fn_vkGetPhysicalDeviceProperties(physicalDevices[0], &deviceProps);
177 if (tryInitializeDevice(physicalDevices[0], &deviceProps, params, hdrOutputRequired)) {
178 return true;
179 }
180 devicesTried.emplace(0);
181
182 // Next, we'll try to match an integrated GPU, since we want to minimize
183 // power consumption and inter-GPU copies.
184 for (uint32_t i = 0; i < physicalDeviceCount; i++) {
185 // Skip devices we've already tried
186 if (devicesTried.find(i) != devicesTried.end()) {
187 continue;
188 }
189
190 VkPhysicalDeviceProperties deviceProps;
191 fn_vkGetPhysicalDeviceProperties(physicalDevices[i], &deviceProps);
192 if (deviceProps.deviceType == VK_PHYSICAL_DEVICE_TYPE_INTEGRATED_GPU) {
193 if (tryInitializeDevice(physicalDevices[i], &deviceProps, params, hdrOutputRequired)) {
194 return true;
195 }
196 devicesTried.emplace(i);
197 }
198 }
199
200 // Next, we'll try to match a discrete GPU.
201 for (uint32_t i = 0; i < physicalDeviceCount; i++) {
202 // Skip devices we've already tried
203 if (devicesTried.find(i) != devicesTried.end()) {
204 continue;
205 }
206
207 VkPhysicalDeviceProperties deviceProps;
208 fn_vkGetPhysicalDeviceProperties(physicalDevices[i], &deviceProps);
209 if (deviceProps.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) {
210 if (tryInitializeDevice(physicalDevices[i], &deviceProps, params, hdrOutputRequired)) {
211 return true;
212 }
213 devicesTried.emplace(i);
214 }
215 }
216
217 // Finally, we'll try matching any non-software device.
218 for (uint32_t i = 0; i < physicalDeviceCount; i++) {
219 // Skip devices we've already tried
220 if (devicesTried.find(i) != devicesTried.end()) {
221 continue;
222 }
223
224 VkPhysicalDeviceProperties deviceProps;
225 fn_vkGetPhysicalDeviceProperties(physicalDevices[i], &deviceProps);
226 if (tryInitializeDevice(physicalDevices[i], &deviceProps, params, hdrOutputRequired)) {
227 return true;
228 }
229 devicesTried.emplace(i);
230 }
231
232 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
233 "No suitable %sVulkan devices found!",
234 hdrOutputRequired ? "HDR-capable " : "");
235 return false;
236}

238bool PlVkRenderer::tryInitializeDevice(VkPhysicalDevice device, VkPhysicalDeviceProperties* deviceProps,
239 PDECODER_PARAMETERS decoderParams, bool hdrOutputRequired)
240{
241 // Check the Vulkan API version first to ensure it meets libplacebo's minimum
242 if (deviceProps->apiVersion < PL_VK_MIN_VERSION) {
243 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
244 "Vulkan device '%s' does not meet minimum Vulkan version",
245 deviceProps->deviceName);
246 return false;
247 }
248
249#ifdef Q_OS_WIN32
250 // Intel's Windows drivers seem to have interoperability issues as of FFmpeg 7.0.1
251 // when using Vulkan Video decoding. Since they also expose HEVC REXT profiles using
252 // D3D11VA, let's reject them here so we can select a different Vulkan device or
253 // just allow D3D11VA to take over.
254 if (m_HwAccelBackend && deviceProps->vendorID == 0x8086 && !qEnvironmentVariableIntValue("PLVK_ALLOW_INTEL")) {
255 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
256 "Skipping Intel GPU for Vulkan Video due to broken drivers");
257 return false;
258 }
259#endif
260
261 // If we're acting as the decoder backend, we need a physical device with Vulkan video support
262 if (m_HwAccelBackend) {
263 const char* videoDecodeExtension;
264
265 if (decoderParams->videoFormat & VIDEO_FORMAT_MASK_H264) {
266 videoDecodeExtension = VK_KHR_VIDEO_DECODE_H264_EXTENSION_NAME;
267 }
268 else if (decoderParams->videoFormat & VIDEO_FORMAT_MASK_H265) {
269 videoDecodeExtension = VK_KHR_VIDEO_DECODE_H265_EXTENSION_NAME;
270 }
271 else if (decoderParams->videoFormat & VIDEO_FORMAT_MASK_AV1) {
272 // FFmpeg 6.1 implemented an early Mesa extension for Vulkan AV1 decoding.
273 // FFmpeg 7.0 replaced that implementation with one based on the official extension.
274#if LIBAVCODEC_VERSION_MAJOR >= 61
275 videoDecodeExtension = VK_KHR_VIDEO_DECODE_AV1_EXTENSION_NAME;
276#else
277 videoDecodeExtension = "VK_MESA_video_decode_av1";
278#endif
279 }
280 else {
281 SDL_assert(false);
282 return false;
283 }
284
285 if (!isExtensionSupportedByPhysicalDevice(device, videoDecodeExtension)) {
286 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
287 "Vulkan device '%s' does not support %s",
288 deviceProps->deviceName,
289 videoDecodeExtension);
290 return false;
291 }
292 }
293
294 if (!isSurfacePresentationSupportedByPhysicalDevice(device)) {
295 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
296 "Vulkan device '%s' does not support presenting on window surface",
297 deviceProps->deviceName);
298 return false;
299 }
300
301 if (hdrOutputRequired && !isColorSpaceSupportedByPhysicalDevice(device, VK_COLOR_SPACE_HDR10_ST2084_EXT)) {
302 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
303 "Vulkan device '%s' does not support HDR10 (ST.2084 PQ)",
304 deviceProps->deviceName);
305 return false;
306 }
307
308 // Avoid software GPUs
309 if (deviceProps->deviceType == VK_PHYSICAL_DEVICE_TYPE_CPU && qgetenv("PLVK_ALLOW_SOFTWARE") != "1") {
310 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
311 "Vulkan device '%s' is a (probably slow) software renderer. Set PLVK_ALLOW_SOFTWARE=1 to allow using this device.",
312 deviceProps->deviceName);
313 return false;
314 }
315
316 pl_vulkan_params vkParams = pl_vulkan_default_params;
317 vkParams.instance = m_PlVkInstance->instance;
318 vkParams.get_proc_addr = m_PlVkInstance->get_proc_addr;
319 vkParams.surface = m_VkSurface;
320 vkParams.device = device;
321 vkParams.opt_extensions = k_OptionalDeviceExtensions;
322 vkParams.num_opt_extensions = SDL_arraysize(k_OptionalDeviceExtensions);
323 vkParams.extra_queues = m_HwAccelBackend ? VK_QUEUE_FLAG_BITS_MAX_ENUM : 0;
324 m_Vulkan = pl_vulkan_create(m_Log, &vkParams);
325 if (m_Vulkan == nullptr) {
326 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
327 "pl_vulkan_create() failed for '%s'",
328 deviceProps->deviceName);
329 return false;
330 }
331
332 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
333 "Vulkan rendering device chosen: %s",
334 deviceProps->deviceName);
335 return true;
336}

362bool PlVkRenderer::initialize(PDECODER_PARAMETERS params)
363{
364 m_Window = params->window;
365
366 unsigned int instanceExtensionCount = 0;
367 if (!SDL_Vulkan_GetInstanceExtensions(params->window, &instanceExtensionCount, nullptr)) {
368 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
369 "SDL_Vulkan_GetInstanceExtensions() #1 failed: %s",
370 SDL_GetError());
371 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
372 return false;
373 }
374
375 std::vector<const char*> instanceExtensions(instanceExtensionCount);
376 if (!SDL_Vulkan_GetInstanceExtensions(params->window, &instanceExtensionCount, instanceExtensions.data())) {
377 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
378 "SDL_Vulkan_GetInstanceExtensions() #2 failed: %s",
379 SDL_GetError());
380 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
381 return false;
382 }
383
384 pl_vk_inst_params vkInstParams = pl_vk_inst_default_params;
385 {
386 vkInstParams.debug_extra = !!qEnvironmentVariableIntValue("PLVK_DEBUG_EXTRA");
387 vkInstParams.debug = vkInstParams.debug_extra || !!qEnvironmentVariableIntValue("PLVK_DEBUG");
388 }
389 vkInstParams.get_proc_addr = (PFN_vkGetInstanceProcAddr)SDL_Vulkan_GetVkGetInstanceProcAddr();
390 vkInstParams.extensions = instanceExtensions.data();
391 vkInstParams.num_extensions = (int)instanceExtensions.size();
392 m_PlVkInstance = pl_vk_inst_create(m_Log, &vkInstParams);
393 if (m_PlVkInstance == nullptr) {
394 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
395 "pl_vk_inst_create() failed");
396 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
397 return false;
398 }
399
400 // Lookup all Vulkan functions we require
401 POPULATE_FUNCTION(vkDestroySurfaceKHR);
402 POPULATE_FUNCTION(vkGetPhysicalDeviceQueueFamilyProperties2);
403 POPULATE_FUNCTION(vkGetPhysicalDeviceSurfacePresentModesKHR);
404 POPULATE_FUNCTION(vkGetPhysicalDeviceSurfaceFormatsKHR);
405 POPULATE_FUNCTION(vkEnumeratePhysicalDevices);
406 POPULATE_FUNCTION(vkGetPhysicalDeviceProperties);
407 POPULATE_FUNCTION(vkGetPhysicalDeviceSurfaceSupportKHR);
408 POPULATE_FUNCTION(vkEnumerateDeviceExtensionProperties);
409
410 if (!SDL_Vulkan_CreateSurface(params->window, m_PlVkInstance->instance, &m_VkSurface)) {
411 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
412 "SDL_Vulkan_CreateSurface() failed: %s",
413 SDL_GetError());
414 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
415 return false;
416 }
417
418 // Enumerate physical devices and choose one that is suitable for our needs.
419 //
420 // For HDR streaming, we try to find an HDR-capable Vulkan device first then
421 // try another search without the HDR requirement if the first attempt fails.
422 if (!chooseVulkanDevice(params, params->videoFormat & VIDEO_FORMAT_MASK_10BIT) &&
423 (!(params->videoFormat & VIDEO_FORMAT_MASK_10BIT) || !chooseVulkanDevice(params, false))) {
424 return false;
425 }
426
427 VkPresentModeKHR presentMode;
428 if (params->enableVsync) {
429 // FIFO mode improves frame pacing compared with Mailbox, especially for
430 // platforms like X11 that lack a VSyncSource implementation for Pacer.
431 presentMode = VK_PRESENT_MODE_FIFO_KHR;
432 }
433 else {
434 // We want immediate mode for V-Sync disabled if possible
435 if (isPresentModeSupportedByPhysicalDevice(m_Vulkan->phys_device, VK_PRESENT_MODE_IMMEDIATE_KHR)) {
436 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
437 "Using Immediate present mode with V-Sync disabled");
438 presentMode = VK_PRESENT_MODE_IMMEDIATE_KHR;
439 }
440 else {
441 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
442 "Immediate present mode is not supported by the Vulkan driver. Latency may be higher than normal with V-Sync disabled.");
443
444 // FIFO Relaxed can tear if the frame is running late
445 if (isPresentModeSupportedByPhysicalDevice(m_Vulkan->phys_device, VK_PRESENT_MODE_FIFO_RELAXED_KHR)) {
446 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
447 "Using FIFO Relaxed present mode with V-Sync disabled");
448 presentMode = VK_PRESENT_MODE_FIFO_RELAXED_KHR;
449 }
450 // Mailbox at least provides non-blocking behavior
451 else if (isPresentModeSupportedByPhysicalDevice(m_Vulkan->phys_device, VK_PRESENT_MODE_MAILBOX_KHR)) {
452 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
453 "Using Mailbox present mode with V-Sync disabled");
454 presentMode = VK_PRESENT_MODE_MAILBOX_KHR;
455 }
456 // FIFO is always supported
457 else {
458 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
459 "Using FIFO present mode with V-Sync disabled");
460 presentMode = VK_PRESENT_MODE_FIFO_KHR;
461 }
462 }
463 }
464
465 pl_vulkan_swapchain_params vkSwapchainParams = {};
466 vkSwapchainParams.surface = m_VkSurface;
467 vkSwapchainParams.present_mode = presentMode;
468 vkSwapchainParams.swapchain_depth = 1; // No queued frames
469#if PL_API_VER >= 338
470 vkSwapchainParams.disable_10bit_sdr = true; // Some drivers don't dither 10-bit SDR output correctly
471#endif
472 m_Swapchain = pl_vulkan_create_swapchain(m_Vulkan, &vkSwapchainParams);
473 if (m_Swapchain == nullptr) {
474 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
475 "pl_vulkan_create_swapchain() failed");
476 return false;
477 }
478
479 m_Renderer = pl_renderer_create(m_Log, m_Vulkan->gpu);
480 if (m_Renderer == nullptr) {
481 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
482 "pl_renderer_create() failed");
483 return false;
484 }
485
486 // We only need an hwaccel device context if we're going to act as the backend renderer too
487 if (m_HwAccelBackend) {
488 m_HwDeviceCtx = av_hwdevice_ctx_alloc(AV_HWDEVICE_TYPE_VULKAN);
489 if (m_HwDeviceCtx == nullptr) {
490 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
491 "av_hwdevice_ctx_alloc(AV_HWDEVICE_TYPE_VULKAN) failed");
492 return false;
493 }
494
495 auto hwDeviceContext = ((AVHWDeviceContext *)m_HwDeviceCtx->data);
496 hwDeviceContext->user_opaque = this; // Used by lockQueue()/unlockQueue()
497
498 auto vkDeviceContext = (AVVulkanDeviceContext*)((AVHWDeviceContext *)m_HwDeviceCtx->data)->hwctx;
499 vkDeviceContext->get_proc_addr = m_PlVkInstance->get_proc_addr;
500 vkDeviceContext->inst = m_PlVkInstance->instance;
501 vkDeviceContext->phys_dev = m_Vulkan->phys_device;
502 vkDeviceContext->act_dev = m_Vulkan->device;
503 vkDeviceContext->device_features = *m_Vulkan->features;
504 vkDeviceContext->enabled_inst_extensions = m_PlVkInstance->extensions;
505 vkDeviceContext->nb_enabled_inst_extensions = m_PlVkInstance->num_extensions;
506 vkDeviceContext->enabled_dev_extensions = m_Vulkan->extensions;
507 vkDeviceContext->nb_enabled_dev_extensions = m_Vulkan->num_extensions;
508#if LIBAVUTIL_VERSION_INT > AV_VERSION_INT(58, 9, 100)
509 vkDeviceContext->lock_queue = lockQueue;
510 vkDeviceContext->unlock_queue = unlockQueue;
511#endif
512
513 // Populate the device queues for decoding this video format
514 populateQueues(params->videoFormat);
515
516 int err = av_hwdevice_ctx_init(m_HwDeviceCtx);
517 if (err < 0) {
518 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
519 "av_hwdevice_ctx_init() failed: %d",
520 err);
521 return false;
522 }
523 }
524
525 return true;
526}

544bool PlVkRenderer::mapAvFrameToPlacebo(const AVFrame *frame, pl_frame* mappedFrame)
545{
546 pl_avframe_params mapParams = {};
547 mapParams.frame = frame;
548 mapParams.tex = m_Textures;
549 if (!pl_map_avframe_ex(m_Vulkan->gpu, mappedFrame, &mapParams)) {
550 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
551 "pl_map_avframe_ex() failed");
552 return false;
553 }
554
555 // libplacebo assumes a minimum luminance value of 0 means the actual value was unknown.
556 // Since we assume the host values are correct, we use the PL_COLOR_HDR_BLACK constant to
557 // indicate infinite contrast.
558 //
559 // NB: We also have to check that the AVFrame actually had metadata in the first place,
560 // because libplacebo may infer metadata if the frame didn't have any.
561 if (av_frame_get_side_data(frame, AV_FRAME_DATA_MASTERING_DISPLAY_METADATA) && !mappedFrame->color.hdr.min_luma) {
562 mappedFrame->color.hdr.min_luma = PL_COLOR_HDR_BLACK;
563 }
564
565 // HACK: AMF AV1 encoding on the host PC does not set full color range properly in the
566 // bitstream data, so libplacebo incorrectly renders the content as limited range.
567 //
568 // As a workaround, set full range manually in the mapped frame ourselves.
569 mappedFrame->repr.levels = PL_COLOR_LEVELS_FULL;
570
571 return true;
572}

575{

579 fn_vkGetPhysicalDeviceQueueFamilyProperties2(m_Vulkan->phys_device, &queueFamilyCount, nullptr);
580
581 std::vector<VkQueueFamilyProperties2> queueFamilies(queueFamilyCount);
582 std::vector<VkQueueFamilyVideoPropertiesKHR> queueFamilyVideoProps(queueFamilyCount);
583 for (uint32_t i = 0; i < queueFamilyCount; i++) {
584 queueFamilyVideoProps[i].sType = VK_STRUCTURE_TYPE_QUEUE_FAMILY_VIDEO_PROPERTIES_KHR;
585 queueFamilies[i].sType = VK_STRUCTURE_TYPE_QUEUE_FAMILY_PROPERTIES_2;
586 queueFamilies[i].pNext = &queueFamilyVideoProps[i];
587 }
588
589 fn_vkGetPhysicalDeviceQueueFamilyProperties2(m_Vulkan->phys_device, &queueFamilyCount, queueFamilies.data());
590
591#if LIBAVUTIL_VERSION_INT >= AV_VERSION_INT(59, 34, 100)
592 Q_UNUSED(videoFormat);
593
594 for (uint32_t i = 0; i < queueFamilyCount; i++) {
595 vkDeviceContext->qf[i].idx = i;
596 vkDeviceContext->qf[i].num = queueFamilies[i].queueFamilyProperties.queueCount;
597 vkDeviceContext->qf[i].flags = (VkQueueFlagBits)queueFamilies[i].queueFamilyProperties.queueFlags;
598 vkDeviceContext->qf[i].video_caps = (VkVideoCodecOperationFlagBitsKHR)queueFamilyVideoProps[i].videoCodecOperations;
599 }
600 vkDeviceContext->nb_qf = queueFamilyCount;
601#else
602 vkDeviceContext->queue_family_index = m_Vulkan->queue_graphics.index;
603 vkDeviceContext->nb_graphics_queues = m_Vulkan->queue_graphics.count;
604 vkDeviceContext->queue_family_tx_index = m_Vulkan->queue_transfer.index;
605 vkDeviceContext->nb_tx_queues = m_Vulkan->queue_transfer.count;
606 vkDeviceContext->queue_family_comp_index = m_Vulkan->queue_compute.index;
607 vkDeviceContext->nb_comp_queues = m_Vulkan->queue_compute.count;
608
609 // Select a video decode queue that is capable of decoding our chosen format
610 for (uint32_t i = 0; i < queueFamilyCount; i++) {
611 if (queueFamilies[i].queueFamilyProperties.queueFlags & VK_QUEUE_VIDEO_DECODE_BIT_KHR) {
612 if (videoFormat & VIDEO_FORMAT_MASK_H264) {
613 if (queueFamilyVideoProps[i].videoCodecOperations & VK_VIDEO_CODEC_OPERATION_DECODE_H264_BIT_KHR) {
614 vkDeviceContext->queue_family_decode_index = i;
615 vkDeviceContext->nb_decode_queues = queueFamilies[i].queueFamilyProperties.queueCount;
616 break;
617 }
618 }
619 else if (videoFormat & VIDEO_FORMAT_MASK_H265) {
620 if (queueFamilyVideoProps[i].videoCodecOperations & VK_VIDEO_CODEC_OPERATION_DECODE_H265_BIT_KHR) {
621 vkDeviceContext->queue_family_decode_index = i;
622 vkDeviceContext->nb_decode_queues = queueFamilies[i].queueFamilyProperties.queueCount;
623 break;
624 }
625 }
626 else if (videoFormat & VIDEO_FORMAT_MASK_AV1) {
627#if LIBAVCODEC_VERSION_MAJOR >= 61
628 // VK_KHR_video_decode_av1 added VK_VIDEO_CODEC_OPERATION_DECODE_AV1_BIT_KHR to check for AV1
629 // decoding support on this queue. Since FFmpeg 6.1 used the older Mesa-specific AV1 extension,
630 // we'll just assume all video decode queues on this device support AV1 (since we checked that
631 // the physical device supports it earlier.
632 if (queueFamilyVideoProps[i].videoCodecOperations & VK_VIDEO_CODEC_OPERATION_DECODE_AV1_BIT_KHR)
633#endif
634 {
635 vkDeviceContext->queue_family_decode_index = i;
636 vkDeviceContext->nb_decode_queues = queueFamilies[i].queueFamilyProperties.queueCount;
637 break;
638 }
639 }
640 else {
641 SDL_assert(false);
642 }
643 }
644 }
645
646 if (vkDeviceContext->queue_family_decode_index < 0) {
647 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
648 "Unable to find compatible video decode queue!");
649 return false;
650 }
651#endif
652
653 return true;
654}

705void PlVkRenderer::waitToRender()
706{
707 // Check if the GPU has failed before doing anything else
708 if (pl_gpu_is_failed(m_Vulkan->gpu)) {
709 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
710 "GPU is in failed state. Recreating renderer.");
711 SDL_Event event;
712 event.type = SDL_RENDER_DEVICE_RESET;
713 SDL_PushEvent(&event);
714 return;
715 }
716
717#ifndef Q_OS_WIN32
718 // With libplacebo's Vulkan backend, all swap_buffers does is wait for queued
719 // presents to finish. This happens to be exactly what we want to do here, since
720 // it lets us wait to select a queued frame for rendering until we know that we
721 // can present without blocking in renderFrame().
722 //
723 // NB: This seems to cause performance problems with the Windows display stack
724 // (particularly on Nvidia) so we will only do this for non-Windows platforms.
725 pl_swapchain_swap_buffers(m_Swapchain);
726#endif
727
728 // Handle the swapchain being resized
729 int vkDrawableW, vkDrawableH;
730 SDL_Vulkan_GetDrawableSize(m_Window, &vkDrawableW, &vkDrawableH);
731 if (!pl_swapchain_resize(m_Swapchain, &vkDrawableW, &vkDrawableH)) {
732 // Swapchain (re)creation can fail if the window is occluded
733 return;
734 }
735
736 // Get the next swapchain buffer for rendering. If this fails, renderFrame()
737 // will try again.
738 //
739 // NB: After calling this successfully, we *MUST* call pl_swapchain_submit_frame(),
740 // hence the implementation of cleanupRenderContext() which does just this in case
741 // renderFrame() wasn't called after waitToRender().
742 if (pl_swapchain_start_frame(m_Swapchain, &m_SwapchainFrame)) {
743 m_HasPendingSwapchainFrame = true;
744 }
745}
746
747void PlVkRenderer::cleanupRenderContext()
748{
749 // We have to submit a pending swapchain frame before shutting down
750 // in order to release a mutex that pl_swapchain_start_frame() acquires.
751 if (m_HasPendingSwapchainFrame) {
752 pl_swapchain_submit_frame(m_Swapchain);
753 m_HasPendingSwapchainFrame = false;
754 }
755}
756
757void PlVkRenderer::renderFrame(AVFrame *frame)
758{
759 pl_frame mappedFrame, targetFrame;
760
761 // If waitToRender() failed to get the next swapchain frame, skip
762 // rendering this frame. It probably means the window is occluded.
763 if (!m_HasPendingSwapchainFrame) {
764 return;
765 }
766
767 if (!mapAvFrameToPlacebo(frame, &mappedFrame)) {
768 // This function logs internally
769 return;
770 }
771
772 // Adjust the swapchain if the colorspace of incoming frames has changed
773 if (!pl_color_space_equal(&mappedFrame.color, &m_LastColorspace)) {
774 m_LastColorspace = mappedFrame.color;
775 SDL_assert(pl_color_space_equal(&mappedFrame.color, &m_LastColorspace));
776 pl_swapchain_colorspace_hint(m_Swapchain, &mappedFrame.color);
777 }
778
779 // Reserve enough space to avoid allocating under the overlay lock
780 pl_overlay_part overlayParts[Overlay::OverlayMax] = {};
781 std::vector<pl_tex> texturesToDestroy;
782 std::vector<pl_overlay> overlays;
783 texturesToDestroy.reserve(Overlay::OverlayMax);
784 overlays.reserve(Overlay::OverlayMax);
785
786 pl_frame_from_swapchain(&targetFrame, &m_SwapchainFrame);
787
788 // We perform minimal processing under the overlay lock to avoid blocking threads updating the overlay
789 SDL_AtomicLock(&m_OverlayLock);
790 for (int i = 0; i < Overlay::OverlayMax; i++) {
791 // If we have a staging overlay, we need to transfer ownership to us
792 if (m_Overlays[i].hasStagingOverlay) {
793 if (m_Overlays[i].hasOverlay) {
794 texturesToDestroy.push_back(m_Overlays[i].overlay.tex);
795 }
796
797 // Copy the overlay fields from the staging area
798 m_Overlays[i].overlay = m_Overlays[i].stagingOverlay;
799
800 // We now own the staging overlay
801 m_Overlays[i].hasStagingOverlay = false;
802 SDL_zero(m_Overlays[i].stagingOverlay);
803 m_Overlays[i].hasOverlay = true;
804 }
805
806 // If we have an overlay but it's been disabled, free the overlay texture
807 if (m_Overlays[i].hasOverlay && !Session::get()->getOverlayManager().isOverlayEnabled((Overlay::OverlayType)i)) {
808 texturesToDestroy.push_back(m_Overlays[i].overlay.tex);
809 m_Overlays[i].hasOverlay = false;
810 }
811
812 // We have an overlay to draw
813 if (m_Overlays[i].hasOverlay) {
814 // Position the overlay
815 overlayParts[i].src = { 0, 0, (float)m_Overlays[i].overlay.tex->params.w, (float)m_Overlays[i].overlay.tex->params.h };
816 if (i == Overlay::OverlayStatusUpdate) {
817 // Bottom Left
818 overlayParts[i].dst.x0 = 0;
819 overlayParts[i].dst.y0 = SDL_max(0, targetFrame.crop.y1 - overlayParts[i].src.y1);
820 }
821 else if (i == Overlay::OverlayDebug) {
822 // Top left
823 overlayParts[i].dst.x0 = 0;
824 overlayParts[i].dst.y0 = 0;
825 }
826 overlayParts[i].dst.x1 = overlayParts[i].dst.x0 + overlayParts[i].src.x1;
827 overlayParts[i].dst.y1 = overlayParts[i].dst.y0 + overlayParts[i].src.y1;
828
829 m_Overlays[i].overlay.parts = &overlayParts[i];
830 m_Overlays[i].overlay.num_parts = 1;
831
832 overlays.push_back(m_Overlays[i].overlay);
833 }
834 }
835 SDL_AtomicUnlock(&m_OverlayLock);
836
837 SDL_Rect src;
838 src.x = mappedFrame.crop.x0;
839 src.y = mappedFrame.crop.y0;
840 src.w = mappedFrame.crop.x1 - mappedFrame.crop.x0;
841 src.h = mappedFrame.crop.y1 - mappedFrame.crop.y0;
842
843 SDL_Rect dst;
844 dst.x = targetFrame.crop.x0;
845 dst.y = targetFrame.crop.y0;
846 dst.w = targetFrame.crop.x1 - targetFrame.crop.x0;
847 dst.h = targetFrame.crop.y1 - targetFrame.crop.y0;
848
849 // Scale the video to the surface size while preserving the aspect ratio
850 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
851
852 targetFrame.crop.x0 = dst.x;
853 targetFrame.crop.y0 = dst.y;
854 targetFrame.crop.x1 = dst.x + dst.w;
855 targetFrame.crop.y1 = dst.y + dst.h;
856
857 // Render the video image and overlays into the swapchain buffer
858 targetFrame.num_overlays = (int)overlays.size();
859 targetFrame.overlays = overlays.data();
860 if (!pl_render_image(m_Renderer, &mappedFrame, &targetFrame, &pl_render_fast_params)) {
861 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
862 "pl_render_image() failed");
863 // NB: We must fallthrough to call pl_swapchain_submit_frame()
864 }
865
866 // Submit the frame for display and swap buffers
867 m_HasPendingSwapchainFrame = false;
868 if (!pl_swapchain_submit_frame(m_Swapchain)) {
869 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
870 "pl_swapchain_submit_frame() failed");
871
872 // Recreate the renderer
873 SDL_Event event;
874 event.type = SDL_RENDER_TARGETS_RESET;
875 SDL_PushEvent(&event);
876 goto UnmapExit;
877 }
878
879#ifdef Q_OS_WIN32
880 // On Windows, we swap buffers here instead of waitToRender()
881 // to avoid some performance problems on Nvidia GPUs.
882 pl_swapchain_swap_buffers(m_Swapchain);
883#endif
884
885UnmapExit:
886 // Delete any textures that need to be destroyed
887 for (pl_tex texture : texturesToDestroy) {
888 pl_tex_destroy(m_Vulkan->gpu, &texture);
889 }
890
891 pl_unmap_avframe(m_Vulkan->gpu, &mappedFrame);
892}

906void PlVkRenderer::notifyOverlayUpdated(Overlay::OverlayType type)
907{
908 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
909 if (newSurface == nullptr && Session::get()->getOverlayManager().isOverlayEnabled(type)) {
910 // The overlay is enabled and there is no new surface. Leave the old texture alone.
911 return;
912 }
913
914 SDL_AtomicLock(&m_OverlayLock);
915 // We want to clear the staging overlay flag even if a staging overlay is still present,
916 // since this ensures the render thread will not read from a partially initialized pl_tex
917 // as we modify or recreate the staging overlay texture outside the overlay lock.
918 m_Overlays[type].hasStagingOverlay = false;
919 SDL_AtomicUnlock(&m_OverlayLock);
920
921 // If there's no new staging overlay, free the old staging overlay texture.
922 // NB: This is safe to do outside the overlay lock because we're guaranteed
923 // to not have racing readers/writers if hasStagingOverlay is false.
924 if (newSurface == nullptr) {
925 pl_tex_destroy(m_Vulkan->gpu, &m_Overlays[type].stagingOverlay.tex);
926 SDL_zero(m_Overlays[type].stagingOverlay);
927 return;
928 }
929
930 // Find a compatible texture format
931 SDL_assert(newSurface->format->format == SDL_PIXELFORMAT_ARGB8888);
932 pl_fmt texFormat = pl_find_named_fmt(m_Vulkan->gpu, "bgra8");
933 if (!texFormat) {
934 SDL_FreeSurface(newSurface);
935 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
936 "pl_find_named_fmt(bgra8) failed");
937 return;
938 }
939
940 // Create a new texture for this overlay if necessary, otherwise reuse the existing texture.
941 // NB: We're guaranteed that the render thread won't be reading this concurrently because
942 // we set hasStagingOverlay to false above.
943 pl_tex_params texParams = {};
944 texParams.w = newSurface->w;
945 texParams.h = newSurface->h;
946 texParams.format = texFormat;
947 texParams.sampleable = true;
948 texParams.host_writable = true;
949 texParams.blit_src = !!(texFormat->caps & PL_FMT_CAP_BLITTABLE);
950 texParams.debug_tag = PL_DEBUG_TAG;
951 if (!pl_tex_recreate(m_Vulkan->gpu, &m_Overlays[type].stagingOverlay.tex, &texParams)) {
952 pl_tex_destroy(m_Vulkan->gpu, &m_Overlays[type].stagingOverlay.tex);
953 SDL_zero(m_Overlays[type].stagingOverlay);
954 SDL_FreeSurface(newSurface);
955 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
956 "pl_tex_recreate() failed");
957 return;
958 }
959
960 // Upload the surface data to the new texture
961 SDL_assert(!SDL_MUSTLOCK(newSurface));
962 pl_tex_transfer_params xferParams = {};
963 xferParams.tex = m_Overlays[type].stagingOverlay.tex;
964 xferParams.row_pitch = (size_t)newSurface->pitch;
965 xferParams.ptr = newSurface->pixels;
966 xferParams.callback = overlayUploadComplete;
967 xferParams.priv = newSurface;
968 if (!pl_tex_upload(m_Vulkan->gpu, &xferParams)) {
969 pl_tex_destroy(m_Vulkan->gpu, &m_Overlays[type].stagingOverlay.tex);
970 SDL_zero(m_Overlays[type].stagingOverlay);
971 SDL_FreeSurface(newSurface);
972 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
973 "pl_tex_upload() failed");
974 return;
975 }
976
977 // newSurface is now owned by the texture upload process. It will be freed in overlayUploadComplete()
978 newSurface = nullptr;
979
980 // Initialize the rest of the overlay params
981 m_Overlays[type].stagingOverlay.mode = PL_OVERLAY_NORMAL;
982 m_Overlays[type].stagingOverlay.coords = PL_OVERLAY_COORDS_DST_FRAME;
983 m_Overlays[type].stagingOverlay.repr = pl_color_repr_rgb;
984 m_Overlays[type].stagingOverlay.color = pl_color_space_srgb;
985
986 // Make this staging overlay visible to the render thread
987 SDL_AtomicLock(&m_OverlayLock);
988 SDL_assert(!m_Overlays[type].hasStagingOverlay);
989 m_Overlays[type].hasStagingOverlay = true;
990 SDL_AtomicUnlock(&m_OverlayLock);
991}
