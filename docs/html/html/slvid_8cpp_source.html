95 m_VideoContext = SLVideo_CreateContext();

97 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,

99 return false;
100 }
101
102 // Create a low latency H.264 stream
103 m_VideoStream = SLVideo_CreateStream(m_VideoContext, k_ESLVideoFormatH264, 1);
104 if (!m_VideoStream) {
105 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
106 "SLVideo_CreateStream() failed");
107 return false;
108 }
109
110 SLVideo_SetStreamVideoTransferMatrix(m_VideoStream, k_ESLVideoTransferMatrix_BT709);
111 SLVideo_SetStreamTargetFramerate(m_VideoStream, params->frameRate, 1);

113 SDL_GetWindowSize(params->window, &m_ViewportWidth, &m_ViewportHeight);
114
115 // Register ourselves for overlay callbacks

117 if (session != nullptr) {
118 session->getOverlayManager().setOverlayRenderer(this);
119 }

124int
125SLVideoDecoder::submitDecodeUnit(PDECODE_UNIT du)
126{
127 int err;
128
129 err = SLVideo_BeginFrame(m_VideoStream, du->fullLength);
130 if (err < 0) {
131 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
132 "SLVideo_BeginFrame() failed: %d (frame %d)",
133 err,
134 du->frameNumber);
135
136 // Need an IDR frame to resync
137 return DR_NEED_IDR;
138 }
139
140 PLENTRY entry = du->bufferList;
141 while (entry != nullptr) {
142 err = SLVideo_WriteFrameData(m_VideoStream,
143 entry->data,
144 entry->length);
145 if (err < 0) {
146 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
147 "SLVideo_WriteFrameData() failed: %d (frame %d)",
148 err,
149 du->frameNumber);
150
151 // Need an IDR frame to resync
152 return DR_NEED_IDR;
153 }
154
155 entry = entry->next;
156 }
157
158 err = SLVideo_SubmitFrame(m_VideoStream);
159 if (err < 0) {
160 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
161 "SLVideo_SubmitFrame() failed: %d (frame %d)",
162 err,
163 du->frameNumber);
164
165 // Need an IDR frame to resync
166 return DR_NEED_IDR;
167 }
168
169 return DR_OK;
170}

172void SLVideoDecoder::notifyOverlayUpdated(Overlay::OverlayType type)
173{
174 // SLVideo supports only one visible overlay at a time. Since we don't have
175 // stats like the FFmpeg-based decoders, we'll just support the status update
176 // overlay and nothing else.
177 if (type != Overlay::OverlayStatusUpdate) {
178 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
179 "Unsupported overlay type: %d", type);
180 return;
181 }
182
183 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
184 bool overlayEnabled = Session::get()->getOverlayManager().isOverlayEnabled(type);
185 if (newSurface == nullptr && overlayEnabled) {
186 // There's no updated surface and the overlay is enabled, so just leave the old surface alone.
187 return;
188 }
189
190 // Hide and free any existing overlay
191 if (m_Overlay != nullptr) {
192 SLVideo_HideOverlay(m_Overlay);
193 SLVideo_FreeOverlay(m_Overlay);
194 m_Overlay = nullptr;
195 }
196
197 if (!overlayEnabled) {
198 SDL_FreeSurface(newSurface);
199 return;
200 }
201
202 m_Overlay = SLVideo_CreateOverlay(m_VideoContext, newSurface->w, newSurface->h);
203 if (m_Overlay == nullptr) {
204 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
205 "SLVideo_CreateOverlay() failed");
206 SDL_FreeSurface(newSurface);
207 return;
208 }
209
210 uint32_t* pixels;
211 int pitch;
212 SLVideo_GetOverlayPixels(m_Overlay, &pixels, &pitch);
213
214 // Copy surface pixels into the new overlay
215 SDL_ConvertPixels(newSurface->w, newSurface->h, newSurface->format->format, newSurface->pixels, newSurface->pitch,
216 SDL_PIXELFORMAT_ARGB8888, pixels, pitch);
217
218 // Position the status overlay at the bottom left corner
219 float flWidth = (float)newSurface->w / m_ViewportWidth;
220 float flHeight = (float)newSurface->h / m_ViewportHeight;
221 SLVideo_SetOverlayDisplayArea(m_Overlay, 0.0f, 1.0f - flHeight, flWidth, flHeight);
222
223 // We're done with the surface now
224 SDL_FreeSurface(newSurface);
225
226 // Show the overlay
227 SLVideo_ShowOverlay(m_Overlay);
228}
