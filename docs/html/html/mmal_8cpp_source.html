46bool MmalRenderer::prepareDecoderContext(AVCodecContext* context, AVDictionary** options)
47{
48 // FFmpeg defaults this to 10 which is too large to fit in the default 64 MB VRAM split.
49 // Reducing to 2 seems to work fine for our bitstreams (max of 1 buffered frame needed).
50 av_dict_set_int(options, "extra_buffers", 2, 0);
51
52 // MMAL seems to dislike certain initial width and height values, but it seems okay
53 // with getting zero for the width and height. We'll zero them all the time to be safe.
54 context->width = 0;
55 context->height = 0;
56
57 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
58 "Using MMAL renderer");
59
60 return true;
61}

63void MmalRenderer::prepareToRender()
64{
65 // Create a renderer and draw a black background for the area not covered by the MMAL overlay.
66 // On the KMSDRM backend, this triggers the modeset that puts the CRTC into the mode we selected.
67 m_BackgroundRenderer = SDL_CreateRenderer(m_Window, -1, SDL_RENDERER_SOFTWARE);
68 if (m_BackgroundRenderer == nullptr) {
69 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
70 "SDL_CreateRenderer() failed: %s",
71 SDL_GetError());
72 return;
73 }
74
75 // SDL_CreateRenderer() can end up having to recreate our window (SDL_RecreateWindow())
76 // to ensure it's compatible with the renderer's OpenGL context. If that happens, we
77 // can get spurious SDL_WINDOWEVENT events that will cause us to (again) recreate our
78 // renderer. This can lead to an infinite to renderer recreation, so discard all
79 // SDL_WINDOWEVENT events after SDL_CreateRenderer().
80 Session* session = Session::get();
81 if (session != nullptr) {
82 // If we get here during a session, we need to synchronize with the event loop
83 // to ensure we don't drop any important events.
84 session->flushWindowEvents();
85 }
86 else {
87 // If we get here prior to the start of a session, just pump and flush ourselves.
88 SDL_PumpEvents();
89 SDL_FlushEvent(SDL_WINDOWEVENT);
90 }
91
92 SDL_SetRenderDrawColor(m_BackgroundRenderer, 0, 0, 0, SDL_ALPHA_OPAQUE);
93 SDL_RenderClear(m_BackgroundRenderer);
94 SDL_RenderPresent(m_BackgroundRenderer);
95}

97void MmalRenderer::updateDisplayRegion()
98{
99 MMAL_STATUS_T status;
100 int currentPosX, currentPosY;
101 MMAL_DISPLAYREGION_T dr;
102
103 dr.hdr.id = MMAL_PARAMETER_DISPLAYREGION;
104 dr.hdr.size = sizeof(MMAL_DISPLAYREGION_T);
105 dr.set = MMAL_DISPLAY_SET_DEST_RECT;
106
107 SDL_GetWindowPosition(m_Window, &currentPosX, &currentPosY);
108
109 if ((SDL_GetWindowFlags(m_Window) & SDL_WINDOW_INPUT_FOCUS) == 0) {
110 dr.dest_rect.x = 0;
111 dr.dest_rect.y = 0;
112 dr.dest_rect.width = 0;
113 dr.dest_rect.height = 0;
114
115 // Force a re-evaluation next time
116 m_LastWindowPosX = -1;
117 m_LastWindowPosY = -1;
118 }
119 else if (m_LastWindowPosX != currentPosX || m_LastWindowPosY != currentPosY) {
120 SDL_Rect src, dst;
121 src.x = src.y = 0;
122 src.w = m_VideoWidth;
123 src.h = m_VideoHeight;
124 dst.x = dst.y = 0;
125 SDL_GetWindowSize(m_Window, &dst.w, &dst.h);
126
127 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
128
129 dr.dest_rect.x = currentPosX + dst.x;
130 dr.dest_rect.y = currentPosY + dst.y;
131 dr.dest_rect.width = dst.w;
132 dr.dest_rect.height = dst.h;
133
134 m_LastWindowPosX = currentPosX;
135 m_LastWindowPosY = currentPosY;
136 }
137 else {
138 // Nothing to do
139 return;
140 }
141
142 status = mmal_port_parameter_set(m_InputPort, &dr.hdr);
143 if (status != MMAL_SUCCESS) {
144 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
145 "mmal_port_parameter_set() failed: %x (%s)",
146 status, mmal_status_to_string(status));
147 }

259// MMAL rendering will silently fail in Full KMS mode. We'll see if that's
260// enabled by reading sysfs. It's gross but it works.
261bool MmalRenderer::getDtDeviceStatus(QString name, bool ifUnknown)
262{
263 QDir dir("/sys/firmware/devicetree/base/soc");
264 QStringList matchingDir = dir.entryList(QStringList(name + "@*"), QDir::Dirs);
265 if (matchingDir.length() != 1) {
266 Q_ASSERT(matchingDir.isEmpty());
267 return ifUnknown;
268 }
269
270 if (!dir.cd(matchingDir.first())) {
271 return ifUnknown;
272 }
273
274 QFile statusFile(dir.filePath("status"));
275 if (!statusFile.open(QFile::ReadOnly)) {
276 // Per Device Tree docs, missing 'status' means enabled
277 return true;
278 }
279
280 QByteArray statusData = statusFile.readAll();
281 QString statusString(statusData);
282
283 // Per Device Tree docs, 'okay' and 'ok' are both acceptable
284 return statusString == "okay" || statusString == "ok";
285}

287bool MmalRenderer::isMmalOverlaySupported()
288{
289 if (qgetenv("MMAL_DISABLE_SUPPORT_CHECK") == "1") {
290 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
291 "MMAL overlay support check is disabled");
292 return true;
293 }
294
295 static bool mmalOverlayCheckCompleted = false;
296 static bool mmalOverlayCheckResult = true;
297
298 // This overlay support check is expensive, so only do it once. The stuff we're
299 // checking can't change without restarting Moonlight (or the whole system).
300 if (mmalOverlayCheckCompleted) {
301 SDL_MemoryBarrierAcquire();
302 return mmalOverlayCheckResult;
303 }
304
305 // vc4-fkms-v3d - firmwarekms is 'okay', hvs is 'disabled'
306 // vc4-kms-v3d - firmwarekms is 'disabled', hvs is 'okay' <- this is the bad one
307 // none - firmwarekms is 'disabled', hvs is 'disabled'
308 if (!getDtDeviceStatus("firmwarekms", true) && getDtDeviceStatus("hvs", true)) {
309 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
310 "Full KMS Mode is enabled! Hardware accelerated H.264 decoding will be unavailable!");
311 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
312 "Change 'dtoverlay=vc4-kms-v3d' to 'dtoverlay=vc4-fkms-v3d' in /boot/config.txt to fix this!");
313 mmalOverlayCheckResult = false;
314 }
315
316 // /dev/video19 is the rpivid stateless HEVC decoder
317 if (!QFile::exists("/dev/video19")) {
318 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
319 "Raspberry Pi HEVC decoder is not enabled! Add 'dtoverlay=rpivid-v4l2' to your /boot/config.txt to fix this!");
320 }
321 else if (strcmp(SDL_GetCurrentVideoDriver(), "KMSDRM") != 0) {
322 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
323 "Raspberry Pi HEVC decoder cannot be used from within a desktop environment. H.264 will be used instead.");
324 }
325
326 SDL_MemoryBarrierRelease();
327 mmalOverlayCheckCompleted = true;
328 return mmalOverlayCheckResult;
329}
