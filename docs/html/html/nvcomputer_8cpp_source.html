216bool NvComputer::wake() const
217{
218 QByteArray wolPayload;
219
220 {
221 QReadLocker readLocker(&lock);
222
223 if (state == NvComputer::CS_ONLINE) {
224 qWarning() << name << "is already online";
225 return true;
226 }
227
228 if (macAddress.isEmpty()) {
229 qWarning() << name << "has no MAC address stored";
230 return false;
231 }
232
233 // Create the WoL payload
234 wolPayload.append(QByteArray::fromHex("FFFFFFFFFFFF"));
235 for (int i = 0; i < 16; i++) {
236 wolPayload.append(macAddress);
237 }
238 Q_ASSERT(wolPayload.size() == 102);
239 }
240
241 // Ports used as-is
242 const quint16 STATIC_WOL_PORTS[] = {
243 9, // Standard WOL port (privileged port)
244 47009, // Port opened by Moonlight Internet Hosting Tool for WoL (non-privileged port)
245 };
246
247 // Ports offset by the HTTP base port for hosts using alternate ports
248 const quint16 DYNAMIC_WOL_PORTS[] = {
249 47998, 47999, 48000, 48002, 48010, // Ports opened by GFE
250 };
251
252 // Add the addresses that we know this host to be
253 // and broadcast addresses for this link just in
254 // case the host has timed out in ARP entries.
255 QMap<QString, quint16> addressMap;
256 QSet<quint16> basePortSet;
257 for (const NvAddress& addr : uniqueAddresses()) {
258 addressMap.insert(addr.address(), addr.port());
259 basePortSet.insert(addr.port());
260 }
261 addressMap.insert("255.255.255.255", 0);
262
263 // Try to broadcast on all available NICs
264 for (const QNetworkInterface& nic : QNetworkInterface::allInterfaces()) {
265 // Ensure the interface is up and skip the loopback adapter
266 if ((nic.flags() & QNetworkInterface::IsUp) == 0 ||
267 (nic.flags() & QNetworkInterface::IsLoopBack) != 0) {
268 continue;
269 }
270
271 QHostAddress allNodesMulticast("FF02::1");
272 for (const QNetworkAddressEntry& addr : nic.addressEntries()) {
273 // Store the scope ID for this NIC if IPv6 is enabled
274 if (!addr.ip().scopeId().isEmpty()) {
275 allNodesMulticast.setScopeId(addr.ip().scopeId());
276 }
277
278 // Skip IPv6 which doesn't support broadcast
279 if (!addr.broadcast().isNull()) {
280 addressMap.insert(addr.broadcast().toString(), 0);
281 }
282 }
283
284 if (!allNodesMulticast.scopeId().isEmpty()) {
285 addressMap.insert(allNodesMulticast.toString(), 0);
286 }
287 }
288
289 // Try all unique address strings or host names
290 bool success = false;
291 for (auto i = addressMap.constBegin(); i != addressMap.constEnd(); i++) {
292 QHostAddress literalAddress;
293 QList<QHostAddress> addressList;
294
295 // If this is an IPv4/IPv6 literal, don't use QHostInfo::fromName() because that will
296 // try to perform a reverse DNS lookup that leads to delays sending WoL packets.
297 if (literalAddress.setAddress(i.key())) {
298 addressList.append(literalAddress);
299 }
300 else {
301 QHostInfo hostInfo = QHostInfo::fromName(i.key());
302 if (hostInfo.error() != QHostInfo::NoError) {
303 qWarning() << "Error resolving" << i.key() << ":" << hostInfo.errorString();
304 continue;
305 }
306
307 addressList.append(hostInfo.addresses());
308 }
309
310 // Try all IP addresses that this string resolves to
311 for (QHostAddress& address : addressList) {
312 QUdpSocket sock;
313
314 // Send to all static ports
315 for (quint16 port : STATIC_WOL_PORTS) {
316 if (sock.writeDatagram(wolPayload, address, port)) {
317 qInfo().nospace().noquote() << "Sent WoL packet to " << name << " via " << address.toString() << ":" << port;
318 success = true;
319 }
320 else {
321 qWarning() << "Send failed:" << sock.error();
322 }
323 }
324
325 QList<quint16> basePorts;
326 if (i.value() != 0) {
327 // If we have a known base port for this address, use only that port
328 basePorts.append(i.value());
329 }
330 else {
331 // If this is a broadcast address without a known HTTP port, try all of them
332 basePorts.append(basePortSet.values());
333 }
334
335 // Send to all dynamic ports using the HTTP port offset(s) for this address
336 for (quint16 basePort : basePorts) {
337 for (quint16 port : DYNAMIC_WOL_PORTS) {
338 port = (port - 47989) + basePort;
339
340 if (sock.writeDatagram(wolPayload, address, port)) {
341 qInfo().nospace().noquote() << "Sent WoL packet to " << name << " via " << address.toString() << ":" << port;
342 success = true;
343 }
344 else {
345 qWarning() << "Send failed:" << sock.error();
346 }
347 }
348 }
349 }
350 }
351
352 return success;
353}

355NvComputer::ReachabilityType NvComputer::getActiveAddressReachability() const
356{
357 NvAddress copyOfActiveAddress;
358
359 {
360 QReadLocker readLocker(&lock);
361
362 if (activeAddress.isNull()) {
363 return ReachabilityType::RI_UNKNOWN;
364 }
365
366 // Grab a copy of the active address to avoid having to hold
367 // the computer lock while doing socket operations
368 copyOfActiveAddress = activeAddress;
369 }
370
371 QTcpSocket s;
372 s.setProxy(QNetworkProxy::NoProxy);
373 s.connectToHost(copyOfActiveAddress.address(), copyOfActiveAddress.port());
374 if (s.waitForConnected(3000)) {
375 Q_ASSERT(!s.localAddress().isNull());
376 Q_ASSERT(!s.peerAddress().isNull());
377
378 for (const QNetworkInterface& nic : QNetworkInterface::allInterfaces()) {
379 // Ensure the interface is up
380 if ((nic.flags() & QNetworkInterface::IsUp) == 0) {
381 continue;
382 }
383
384 for (const QNetworkAddressEntry& addr : nic.addressEntries()) {
385 if (addr.ip() == s.localAddress()) {
386 qInfo() << "Found matching interface:" << nic.humanReadableName() << nic.hardwareAddress() << nic.flags();
387
388#if QT_VERSION >= QT_VERSION_CHECK(5, 11, 0)
389 qInfo() << "Interface Type:" << nic.type();
390 qInfo() << "Interface MTU:" << nic.maximumTransmissionUnit();
391
392 if (nic.type() == QNetworkInterface::Virtual ||
393 nic.type() == QNetworkInterface::Ppp) {
394 // Treat PPP and virtual interfaces as likely VPNs
395 return ReachabilityType::RI_VPN;
396 }
397
398 if (nic.maximumTransmissionUnit() != 0 && nic.maximumTransmissionUnit() < 1500) {
399 // Treat MTUs under 1500 as likely VPNs
400 return ReachabilityType::RI_VPN;
401 }
402#endif
403
404 if (nic.flags() & QNetworkInterface::IsPointToPoint) {
405 // Treat point-to-point links as likely VPNs.
406 // This check detects OpenVPN on Unix-like OSes.
407 return ReachabilityType::RI_VPN;
408 }
409
410#ifdef Q_OS_WINDOWS
411 if (nic.name().startsWith("iftype53_") || nic.name().startsWith("iftype131_")) {
412 // Match by NDIS interface type. These values are Microsoft's recommended values for VPN connections:
413 // https://learn.microsoft.com/en-US/troubleshoot/windows-client/networking/windows-connection-manager-disconnects-wlan#more-information
414 //
415 // The following VPNs use IF_TYPE_PROP_VIRTUAL under Windows:
416 // - WireguardNT VPNs
417 // - All WinTun-based VPNs (such as Slack Nebula)
418 // - OpenVPN with tap-windows6
419 return ReachabilityType::RI_VPN;
420 }
421#endif
422
423 if (nic.hardwareAddress().startsWith("00:FF", Qt::CaseInsensitive)) {
424 // OpenVPN TAP interfaces have a MAC address starting with 00:FF on Windows
425 return ReachabilityType::RI_VPN;
426 }
427
428 if (nic.humanReadableName().startsWith("ZeroTier")) {
429 // ZeroTier interfaces always start with "ZeroTier"
430 return ReachabilityType::RI_VPN;
431 }
432
433 if (nic.humanReadableName().contains("VPN")) {
434 // This one is just a final VPN heuristic if all else fails
435 return ReachabilityType::RI_VPN;
436 }
437
438 // Didn't meet any of our VPN heuristics. Let's see if the peer address is on-link.
439 Q_ASSERT(addr.prefixLength() >= 0);
440 if (addr.prefixLength() >= 0 && s.localAddress().isInSubnet(s.peerAddress(), addr.prefixLength())) {
441 return ReachabilityType::RI_LAN;
442 }
443
444 // Default to unknown if nothing else matched
445 return ReachabilityType::RI_UNKNOWN;
446 }
447 }
448 }
449
450 qWarning() << "No match found for address:" << s.localAddress();
451 return ReachabilityType::RI_UNKNOWN;
452 }
453 else {
454 // If we fail to connect, just pretend that it's not a VPN
455 qWarning() << "Unable to check for reachability within 3 seconds";
456 return ReachabilityType::RI_UNKNOWN;
457 }
458}

480QVector<NvAddress> NvComputer::uniqueAddresses() const
481{
482 QReadLocker readLocker(&lock);
483 QVector<NvAddress> uniqueAddressList;
484
485 // Start with addresses correctly ordered
486 uniqueAddressList.append(activeAddress);
487 uniqueAddressList.append(localAddress);
488 uniqueAddressList.append(remoteAddress);
489 uniqueAddressList.append(ipv6Address);
490 uniqueAddressList.append(manualAddress);
491
492 // Prune duplicates (always giving precedence to the first)
493 for (int i = 0; i < uniqueAddressList.count(); i++) {
494 if (uniqueAddressList[i].isNull()) {
495 uniqueAddressList.remove(i);
496 i--;
497 continue;
498 }
499 for (int j = i + 1; j < uniqueAddressList.count(); j++) {
500 if (uniqueAddressList[i] == uniqueAddressList[j]) {
501 // Always remove the later occurrence
502 uniqueAddressList.remove(j);
503 j--;
504 }
505 }
506 }
507
508 // We must have at least 1 address
509 Q_ASSERT(!uniqueAddressList.isEmpty());
510
511 return uniqueAddressList;
512}
