15// This method returns key's value from QMap where the key is a QString.
16// Key matching is case insensitive.
17template <typename T>
18static T mapValue(QMap<QString, T> map, QString key)
19{
20 for(auto& item : map.toStdMap()) {
21 if (QString::compare(item.first, key, Qt::CaseInsensitive) == 0) {
22 return item.second;
23 }

151GlobalCommandLineParser::ParseResult GlobalCommandLineParser::parse(const QStringList &args)
152{
153 CommandLineParser parser;
154 parser.setupCommonOptions();
155 parser.setApplicationDescription(
156 "\n"
157 "Starts Moonlight normally if no arguments are given.\n"
158 "\n"
159 "Available actions:\n"
160 " list List the available apps on a host\n"
161 " quit Quit the currently running app\n"
162 " stream Start streaming an app\n"
163 " pair Pair a new host\n"
164 "\n"
165 "See 'moonlight <action> --help' for help of specific action."
166 );
167 parser.addPositionalArgument("action", "Action to execute", "<action>");
168 parser.parse(args);
169 auto posArgs = parser.positionalArguments();
170
171 if (posArgs.isEmpty()) {
172 // This method will not return and terminates the process if --version
173 // or --help is specified
174 parser.handleHelpAndVersionOptions();
175 parser.handleUnknownOptions();
176 return NormalStartRequested;
177 }
178 else {
179 // If users supply arguments that accept values prior to the "quit"
180 // or "stream" positional arguments, we will not be able to correctly
181 // parse the value out of the input because this QCommandLineParser
182 // doesn't know about all of the options that "quit" and "stream"
183 // commands can accept. To work around this issue, we just look
184 // for "quit" or "stream" positional arguments anywhere.
185 for (int i = 0; i < posArgs.size(); i++) {
186 QString action = posArgs.at(i).toLower();
187 if (action == "quit") {
188 return QuitRequested;
189 } else if (action == "stream") {
190 return StreamRequested;
191 } else if (action == "pair") {
192 return PairRequested;
193 } else if (action == "list") {
194 return ListRequested;
195 }
196 }
197
198 parser.showError(QString("Invalid action"));
199 }
