54bool Session::initializeAudioRenderer()
55{
56 int error;
57
58 SDL_assert(m_OriginalAudioConfig.channelCount > 0);
59 SDL_assert(m_AudioRenderer == nullptr);
60 SDL_assert(m_OpusDecoder == nullptr);
61
62 m_AudioRenderer = createAudioRenderer(&m_OriginalAudioConfig);
63
64 // We may be unable to create an audio renderer right now
65 if (m_AudioRenderer == nullptr) {
66 return false;
67 }
68
69 // Allow the chosen renderer to remap Opus channels as needed to ensure proper output
70 m_ActiveAudioConfig = m_OriginalAudioConfig;
71 m_AudioRenderer->remapChannels(&m_ActiveAudioConfig);
72
73 // Create the Opus decoder with the renderer's preferred channel mapping
74 m_OpusDecoder =
75 opus_multistream_decoder_create(m_ActiveAudioConfig.sampleRate,
76 m_ActiveAudioConfig.channelCount,
77 m_ActiveAudioConfig.streams,
78 m_ActiveAudioConfig.coupledStreams,
79 m_ActiveAudioConfig.mapping,
80 &error);
81 if (m_OpusDecoder == nullptr) {
82 delete m_AudioRenderer;
83 m_AudioRenderer = nullptr;
84 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
85 "Failed to create decoder: %d",
86 error);
87 return false;
88 }
89
90 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
91 "Audio stream has %d channels",
92 m_ActiveAudioConfig.channelCount);
93 return true;
94}

150void Session::arDecodeAndPlaySample(char* sampleData, int sampleLength)
151{
152 int samplesDecoded;
153
154#ifndef STEAM_LINK
155 // Set this thread to high priority to reduce the chance of missing
156 // our sample delivery time. On Steam Link, this causes starvation
157 // of other threads due to severely restricted CPU time available,
158 // so we will skip it on that platform.
159 if (s_ActiveSession->m_AudioSampleCount == 0) {
160 if (SDL_SetThreadPriority(SDL_THREAD_PRIORITY_HIGH) < 0) {
161 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
162 "Unable to set audio thread to high priority: %s",
163 SDL_GetError());
164 }
165 }
166#endif
167
168 // See if we need to drop this sample
169 if (s_ActiveSession->m_DropAudioEndTime != 0) {
170 if (SDL_TICKS_PASSED(SDL_GetTicks(), s_ActiveSession->m_DropAudioEndTime)) {
171 // Avoid calling SDL_GetTicks() now
172 s_ActiveSession->m_DropAudioEndTime = 0;
173
174 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
175 "Audio drop window has ended");
176 }
177 else {
178 // We're still in the drop window
179 return;
180 }
181 }
182
183 s_ActiveSession->m_AudioSampleCount++;
184
185 // If audio is muted, don't decode or play the audio
186 if (s_ActiveSession->m_AudioMuted) {
187 return;
188 }
189
190 if (s_ActiveSession->m_AudioRenderer != nullptr) {
191 int sampleSize = s_ActiveSession->m_AudioRenderer->getAudioBufferSampleSize();
192 int frameSize = sampleSize * s_ActiveSession->m_ActiveAudioConfig.channelCount;
193 int desiredBufferSize = frameSize * s_ActiveSession->m_ActiveAudioConfig.samplesPerFrame;
194 void* buffer = s_ActiveSession->m_AudioRenderer->getAudioBuffer(&desiredBufferSize);
195 if (buffer == nullptr) {
196 return;
197 }
198
199 if (s_ActiveSession->m_AudioRenderer->getAudioBufferFormat() == IAudioRenderer::AudioFormat::Float32NE) {
200 samplesDecoded = opus_multistream_decode_float(s_ActiveSession->m_OpusDecoder,
201 (unsigned char*)sampleData,
202 sampleLength,
203 (float*)buffer,
204 desiredBufferSize / frameSize,
205 0);
206 }
207 else {
208 samplesDecoded = opus_multistream_decode(s_ActiveSession->m_OpusDecoder,
209 (unsigned char*)sampleData,
210 sampleLength,
211 (short*)buffer,
212 desiredBufferSize / frameSize,
213 0);
214 }
215
216 // Update desiredSize with the number of bytes actually populated by the decoding operation
217 if (samplesDecoded > 0) {
218 SDL_assert(desiredBufferSize >= frameSize * samplesDecoded);
219 desiredBufferSize = frameSize * samplesDecoded;
220 }
221 else {
222 desiredBufferSize = 0;
223 }
224
225 if (!s_ActiveSession->m_AudioRenderer->submitAudio(desiredBufferSize)) {
226 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
227 "Reinitializing audio renderer after failure");
228
229 opus_multistream_decoder_destroy(s_ActiveSession->m_OpusDecoder);
230 s_ActiveSession->m_OpusDecoder = nullptr;
231
232 delete s_ActiveSession->m_AudioRenderer;
233 s_ActiveSession->m_AudioRenderer = nullptr;
234 }
235 }
236
237 // Only try to recreate the audio renderer every 200 samples (1 second)
238 // to avoid thrashing if the audio device is unavailable. It is
239 // safe to reinitialize here because we can't be torn down while
240 // the audio decoder/playback thread is still alive.
241 if (s_ActiveSession->m_AudioRenderer == nullptr && (s_ActiveSession->m_AudioSampleCount % 200) == 0) {
242 // Since we're doing this inline and audio initialization takes time, we need
243 // to drop samples to account for the time we've spent blocking audio rendering
244 // so we return to real-time playback and don't accumulate latency.
245 Uint32 audioReinitStartTime = SDL_GetTicks();
246 if (s_ActiveSession->initializeAudioRenderer()) {
247 Uint32 audioReinitStopTime = SDL_GetTicks();
248
249 s_ActiveSession->m_DropAudioEndTime = audioReinitStopTime + (audioReinitStopTime - audioReinitStartTime);
250 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
251 "Audio reinitialization took %d ms - starting drop window",
252 audioReinitStopTime - audioReinitStartTime);
253 }
254 }
