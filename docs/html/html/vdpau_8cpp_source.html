364void VDPAURenderer::notifyOverlayUpdated(Overlay::OverlayType type)
365{
366 VdpStatus status;
367
368 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
369 bool overlayEnabled = Session::get()->getOverlayManager().isOverlayEnabled(type);
370 if (newSurface == nullptr && overlayEnabled) {
371 // There's no updated surface and the overlay is enabled, so just leave the old surface alone.
372 return;
373 }
374
375 // Destroy the old surface
376 // NB: The mutex ensures the surface is not currently being read for rendering.
377 // NB 2: It is safe to unlock here because this thread is the only surface producer.
378 SDL_LockMutex(m_OverlayMutex);
379 VdpBitmapSurface oldBitmapSurface = m_OverlaySurface[type];
380 m_OverlaySurface[type] = 0;
381 SDL_UnlockMutex(m_OverlayMutex);
382
383 if (oldBitmapSurface != 0) {
384 status = m_VdpBitmapSurfaceDestroy(oldBitmapSurface);
385 if (status != VDP_STATUS_OK) {
386 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
387 "VdpBitmapSurfaceDestroy() failed: %s",
388 m_VdpGetErrorString(status));
389
390 // This should never happen.
391 SDL_assert(false);
392 }
393 }
394
395 if (!overlayEnabled) {
396 SDL_FreeSurface(newSurface);
397 return;
398 }
399
400 if (newSurface != nullptr) {
401 SDL_assert(!SDL_MUSTLOCK(newSurface));
402 SDL_assert(newSurface->format->format == SDL_PIXELFORMAT_ARGB8888);
403
404 VdpBitmapSurface newBitmapSurface = 0;
405 status = m_VdpBitmapSurfaceCreate(m_Device,
406 VDP_RGBA_FORMAT_B8G8R8A8,
407 newSurface->w,
408 newSurface->h,
409 VDP_TRUE, // Is this correct?
410 &newBitmapSurface);
411 if (status != VDP_STATUS_OK) {
412 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
413 "VdpBitmapSurfaceCreate() failed: %s",
414 m_VdpGetErrorString(status));
415 SDL_FreeSurface(newSurface);
416 return;
417 }
418
419 status = m_VdpBitmapSurfacePutBitsNative(newBitmapSurface,
420 &newSurface->pixels,
421 (const uint32_t*)&newSurface->pitch,
422 nullptr);
423 if (status != VDP_STATUS_OK) {
424 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
425 "VdpBitmapSurfacePutBitsNative() failed: %s",
426 m_VdpGetErrorString(status));
427 m_VdpBitmapSurfaceDestroy(newBitmapSurface);
428 SDL_FreeSurface(newSurface);
429 return;
430 }
431
432 VdpRect overlayRect;
433
434 if (type == Overlay::OverlayStatusUpdate) {
435 // Bottom Left
436 overlayRect.x0 = 0;
437 overlayRect.y0 = m_DisplayHeight - newSurface->h;
438 }
439 else if (type == Overlay::OverlayDebug) {
440 // Top left
441 overlayRect.x0 = 0;
442 overlayRect.y0 = 0;
443 }
444
445 overlayRect.x1 = overlayRect.x0 + newSurface->w;
446 overlayRect.y1 = overlayRect.y0 + newSurface->h;
447
448 // Surface data is no longer needed
449 SDL_FreeSurface(newSurface);
450
451 SDL_LockMutex(m_OverlayMutex);
452 m_OverlaySurface[type] = newBitmapSurface;
453 m_OverlayRect[type] = overlayRect;
454 SDL_UnlockMutex(m_OverlayMutex);
455 }

481{
482 VdpStatus status;
483
484 // Don't even bother locking the mutex if the overlay is disabled
485 if (!Session::get()->getOverlayManager().isOverlayEnabled(type)) {
486 return;
487 }
488
489 if (SDL_TryLockMutex(m_OverlayMutex) != 0) {
490 // If the overlay is currently being updated, skip rendering it this frame.
491 return;
492 }
493

497 return;

501 &m_OverlayRect[type],

503 nullptr,
504 nullptr,
505 &m_OverlayBlendState,
506 0);
507
508 SDL_UnlockMutex(m_OverlayMutex);
509
510 if (status != VDP_STATUS_OK) {
511 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
512 "VdpOutputSurfaceRenderBitmapSurface() failed: %s",
513 m_VdpGetErrorString(status));

527void VDPAURenderer::renderFrame(AVFrame* frame)
528{
529 VdpStatus status;
530 VdpVideoSurface videoSurface = (VdpVideoSurface)(uintptr_t)frame->data[3];
531
532 // This is safe without locking because this is always called on the main thread
533 VdpOutputSurface chosenSurface = m_OutputSurface[m_NextSurfaceIndex];
534 m_NextSurfaceIndex = (m_NextSurfaceIndex + 1) % OUTPUT_SURFACE_COUNT;
535
536 // We need to create the mixer on the fly, because we don't know the dimensions
537 // of our video surfaces in advance of decoding. We also need to recreate it when
538 // the frame format or size changes.
539 if (hasFrameFormatChanged(frame)) {
540 if (m_VideoMixer != 0) {
541 m_VdpVideoMixerDestroy(m_VideoMixer);
542 }
543
544 VdpChromaType videoSurfaceChroma;
545 uint32_t videoSurfaceWidth, videoSurfaceHeight;
546 status = m_VdpVideoSurfaceGetParameters(videoSurface, &videoSurfaceChroma,
547 &videoSurfaceWidth, &videoSurfaceHeight);
548 if (status != VDP_STATUS_OK) {
549 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
550 "VdpVideoSurfaceGetParameters() failed: %s",
551 m_VdpGetErrorString(status));
552 return;
553 }
554
555 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
556 "VDPAU surface size: %dx%d",
557 videoSurfaceWidth, videoSurfaceHeight);
558
559 #define PARAM_COUNT 3
560 const VdpVideoMixerParameter params[PARAM_COUNT] = {
561 VDP_VIDEO_MIXER_PARAMETER_VIDEO_SURFACE_WIDTH,
562 VDP_VIDEO_MIXER_PARAMETER_VIDEO_SURFACE_HEIGHT,
563 VDP_VIDEO_MIXER_PARAMETER_CHROMA_TYPE,
564 };
565 const void* const paramValues[PARAM_COUNT] = {
566 &videoSurfaceWidth,
567 &videoSurfaceHeight,
568 &videoSurfaceChroma,
569 };
570
571 status = m_VdpVideoMixerCreate(m_Device, 0, nullptr,
572 PARAM_COUNT, params, paramValues,
573 &m_VideoMixer);
574 if (status != VDP_STATUS_OK) {
575 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
576 "VdpVideoMixerCreate() failed: %s",
577 m_VdpGetErrorString(status));
578 return;
579 }
580 }
581
582 // Wait for this frame to be off the screen. This will usually be a no-op
583 // since it already happened in waitToRender(). However, that won't be the
584 // case is when frame pacing is enabled.
585 VdpTime pts;
586 m_VdpPresentationQueueBlockUntilSurfaceIdle(m_PresentationQueue, chosenSurface, &pts);
587
588 VdpRect sourceRect, outputRect;
589
590 SDL_Rect src, dst;
591 src.x = src.y = 0;
592 src.w = m_VideoWidth;
593 src.h = m_VideoHeight;
594 dst.x = dst.y = 0;
595 dst.w = m_DisplayWidth;
596 dst.h = m_DisplayHeight;
597
598 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
599
600 outputRect.x0 = dst.x;
601 outputRect.x1 = dst.x + dst.w;
602 outputRect.y0 = dst.y;
603 outputRect.y1 = dst.y + dst.h;
604
605 sourceRect.x0 = sourceRect.y0 = 0;
606 sourceRect.x1 = m_VideoWidth;
607 sourceRect.y1 = m_VideoHeight;
