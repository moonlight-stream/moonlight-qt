54void SdlInputHandler::sendGamepadState(GamepadState* state)
55{
56 SDL_assert(m_GamepadMask == 0x1 || m_MultiController);
57
58 // Handle Select+PS as the clickpad button on PS4/5 controllers without a clickpad mapping
59 int buttons = state->buttons;
60 if (state->clickpadButtonEmulationEnabled) {
61 if (state->buttons == (BACK_FLAG | SPECIAL_FLAG)) {
62 buttons = MISC_FLAG;
63 state->emulatedClickpadButtonDown = true;
64 }
65 else if (state->emulatedClickpadButtonDown) {
66 buttons &= ~MISC_FLAG;
67 state->emulatedClickpadButtonDown = false;
68 }
69 }
70
71 unsigned char lt = state->lt;
72 unsigned char rt = state->rt;
73 short lsX = state->lsX;
74 short lsY = state->lsY;
75 short rsX = state->rsX;
76 short rsY = state->rsY;
77
78 // When in single controller mode, merge all gamepad state together
79 if (!m_MultiController) {
80 for (int i = 0; i < MAX_GAMEPADS; i++) {
81 if (m_GamepadState[i].index == state->index) {
82 buttons |= m_GamepadState[i].buttons;
83 if (lt < m_GamepadState[i].lt) {
84 lt = m_GamepadState[i].lt;
85 }
86 if (rt < m_GamepadState[i].rt) {
87 rt = m_GamepadState[i].rt;
88 }
89
90 // We use abs() here instead of qAbs() for get proper integer promotion to
91 // correctly handle abs(-32768), which is not representable in a short.
92 if (abs(lsX) < abs(m_GamepadState[i].lsX) || abs(lsY) < abs(m_GamepadState[i].lsY)) {
93 lsX = m_GamepadState[i].lsX;
94 lsY = m_GamepadState[i].lsY;
95 }
96 if (abs(rsX) < abs(m_GamepadState[i].rsX) || abs(rsY) < abs(m_GamepadState[i].rsY)) {
97 rsX = m_GamepadState[i].rsX;
98 rsY = m_GamepadState[i].rsY;
99 }
100 }
101 }
102 }
103
104 LiSendMultiControllerEvent(state->index,
105 m_GamepadMask,
106 buttons,
107 lt,
108 rt,
109 lsX,
110 lsY,
111 rsX,
112 rsY);
113}

156Uint32 SdlInputHandler::mouseEmulationTimerCallback(Uint32 interval, void *param)
157{
158 auto gamepad = reinterpret_cast<GamepadState*>(param);
159
160 int rawX;
161 int rawY;
162
163 // Determine which analog stick is currently receiving the strongest input
164 if (abs(gamepad->lsX) + abs(gamepad->lsY) > abs(gamepad->rsX) + abs(gamepad->rsY)) {
165 rawX = gamepad->lsX;
166 rawY = -gamepad->lsY;
167 }
168 else {
169 rawX = gamepad->rsX;
170 rawY = -gamepad->rsY;
171 }
172
173 float deltaX;
174 float deltaY;
175
176 // Produce a base vector for mouse movement with increased speed as we deviate further from center
177 deltaX = qPow(rawX / 32766.0f * MOUSE_EMULATION_MOTION_MULTIPLIER, 3);
178 deltaY = qPow(rawY / 32766.0f * MOUSE_EMULATION_MOTION_MULTIPLIER, 3);
179
180 // Enforce deadzones
181 deltaX = qAbs(deltaX) > MOUSE_EMULATION_DEADZONE ? deltaX - MOUSE_EMULATION_DEADZONE : 0;
182 deltaY = qAbs(deltaY) > MOUSE_EMULATION_DEADZONE ? deltaY - MOUSE_EMULATION_DEADZONE : 0;
183
184 if (deltaX != 0 || deltaY != 0) {
185 LiSendMouseMoveEvent((short)deltaX, (short)deltaY);
186 }
187
188 return interval;
189}

191void SdlInputHandler::handleControllerAxisEvent(SDL_ControllerAxisEvent* event)
192{
193 SDL_JoystickID gameControllerId = event->which;
194 GamepadState* state = findStateForGamepad(gameControllerId);
195 if (state == NULL) {
196 return;
197 }
198
199 // Batch all pending axis motion events for this gamepad to save CPU time
200 SDL_Event nextEvent;
201 for (;;) {
202 switch (event->axis)
203 {
204 case SDL_CONTROLLER_AXIS_LEFTX:
205 state->lsX = event->value;
206 break;
207 case SDL_CONTROLLER_AXIS_LEFTY:
208 // Signed values have one more negative value than
209 // positive value, so inverting the sign on -32768
210 // could actually cause the value to overflow and
211 // wrap around to be negative again. Avoid that by
212 // capping the value at 32767.
213 state->lsY = -qMax(event->value, (short)-32767);
214 break;
215 case SDL_CONTROLLER_AXIS_RIGHTX:
216 state->rsX = event->value;
217 break;
218 case SDL_CONTROLLER_AXIS_RIGHTY:
219 state->rsY = -qMax(event->value, (short)-32767);
220 break;
221 case SDL_CONTROLLER_AXIS_TRIGGERLEFT:
222 state->lt = (unsigned char)(event->value * 255UL / 32767);
223 break;
224 case SDL_CONTROLLER_AXIS_TRIGGERRIGHT:
225 state->rt = (unsigned char)(event->value * 255UL / 32767);
226 break;
227 default:
228 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
229 "Unhandled controller axis: %d",
230 event->axis);
231 return;
232 }
233
234 // Check for another event to batch with
235 if (SDL_PeepEvents(&nextEvent, 1, SDL_PEEKEVENT, SDL_CONTROLLERAXISMOTION, SDL_CONTROLLERAXISMOTION) <= 0) {
236 break;
237 }
238
239 event = &nextEvent.caxis;
240 if (event->which != gameControllerId) {
241 // Stop batching if a different gamepad interrupts us
242 break;
243 }
244
245 // Remove the next event to batch
246 SDL_PeepEvents(&nextEvent, 1, SDL_GETEVENT, SDL_CONTROLLERAXISMOTION, SDL_CONTROLLERAXISMOTION);
247 }
248
249 // Only send the gamepad state to the host if it's not in mouse emulation mode
250 if (state->mouseEmulationTimer == 0) {
251 sendGamepadState(state);
252 }
253}

255void SdlInputHandler::handleControllerButtonEvent(SDL_ControllerButtonEvent* event)
256{
257 if (event->button >= SDL_arraysize(k_ButtonMap)) {
258 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
259 "No mapping for gamepad button: %u",
260 event->button);
261 return;
262 }
263
264 GamepadState* state = findStateForGamepad(event->which);
265 if (state == NULL) {
266 return;
267 }
268
269 if (m_SwapFaceButtons) {
270 switch (event->button) {
271 case SDL_CONTROLLER_BUTTON_A:
272 event->button = SDL_CONTROLLER_BUTTON_B;
273 break;
274 case SDL_CONTROLLER_BUTTON_B:
275 event->button = SDL_CONTROLLER_BUTTON_A;
276 break;
277 case SDL_CONTROLLER_BUTTON_X:
278 event->button = SDL_CONTROLLER_BUTTON_Y;
279 break;
280 case SDL_CONTROLLER_BUTTON_Y:
281 event->button = SDL_CONTROLLER_BUTTON_X;
282 break;
283 }
284 }
285
286 if (event->state == SDL_PRESSED) {
287 state->buttons |= k_ButtonMap[event->button];
288
289 if (event->button == SDL_CONTROLLER_BUTTON_START) {
290 state->lastStartDownTime = SDL_GetTicks();
291 }
292 else if (state->mouseEmulationTimer != 0) {
293 if (event->button == SDL_CONTROLLER_BUTTON_A) {
294 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_LEFT);
295 }
296 else if (event->button == SDL_CONTROLLER_BUTTON_B) {
297 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_RIGHT);
298 }
299 else if (event->button == SDL_CONTROLLER_BUTTON_X) {
300 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_MIDDLE);
301 }
302 else if (event->button == SDL_CONTROLLER_BUTTON_LEFTSHOULDER) {
303 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_X1);
304 }
305 else if (event->button == SDL_CONTROLLER_BUTTON_RIGHTSHOULDER) {
306 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_X2);
307 }
308 else if (event->button == SDL_CONTROLLER_BUTTON_DPAD_UP) {
309 LiSendScrollEvent(1);
310 }
311 else if (event->button == SDL_CONTROLLER_BUTTON_DPAD_DOWN) {
312 LiSendScrollEvent(-1);
313 }
314 else if (event->button == SDL_CONTROLLER_BUTTON_DPAD_RIGHT) {
315 LiSendHScrollEvent(1);
316 }
317 else if (event->button == SDL_CONTROLLER_BUTTON_DPAD_LEFT) {
318 LiSendHScrollEvent(-1);
319 }
320 }
321 }
322 else {
323 state->buttons &= ~k_ButtonMap[event->button];
324
325 if (event->button == SDL_CONTROLLER_BUTTON_START) {
326 if (SDL_GetTicks() - state->lastStartDownTime > MOUSE_EMULATION_LONG_PRESS_TIME) {
327 if (state->mouseEmulationTimer != 0) {
328 SDL_RemoveTimer(state->mouseEmulationTimer);
329 state->mouseEmulationTimer = 0;
330
331 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
332 "Mouse emulation deactivated");
333 Session::get()->notifyMouseEmulationMode(false);
334 }
335 else if (m_GamepadMouse) {
336 // Send the start button up event to the host, since we won't do it below
337 sendGamepadState(state);
338
339 state->mouseEmulationTimer = SDL_AddTimer(MOUSE_EMULATION_POLLING_INTERVAL, SdlInputHandler::mouseEmulationTimerCallback, state);
340
341 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
342 "Mouse emulation active");
343 Session::get()->notifyMouseEmulationMode(true);
344 }
345 }
346 }
347 else if (state->mouseEmulationTimer != 0) {
348 if (event->button == SDL_CONTROLLER_BUTTON_A) {
349 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_LEFT);
350 }
351 else if (event->button == SDL_CONTROLLER_BUTTON_B) {
352 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_RIGHT);
353 }
354 else if (event->button == SDL_CONTROLLER_BUTTON_X) {
355 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_MIDDLE);
356 }
357 else if (event->button == SDL_CONTROLLER_BUTTON_LEFTSHOULDER) {
358 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_X1);
359 }
360 else if (event->button == SDL_CONTROLLER_BUTTON_RIGHTSHOULDER) {
361 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, BUTTON_X2);
362 }
363 }
364 }
365
366 // Handle Start+Select+L1+R1 as a gamepad quit combo
367 if (state->buttons == (PLAY_FLAG | BACK_FLAG | LB_FLAG | RB_FLAG) && qgetenv("NO_GAMEPAD_QUIT") != "1") {
368 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
369 "Detected quit gamepad button combo");
370
371 // Push a quit event to the main loop
372 SDL_Event event;
373 event.type = SDL_QUIT;
374 event.quit.timestamp = SDL_GetTicks();
375 SDL_PushEvent(&event);
376
377 // Clear buttons down on this gamepad
378 LiSendMultiControllerEvent(state->index, m_GamepadMask,
379 0, 0, 0, 0, 0, 0, 0);
380 return;
381 }
382
383 // Handle Select+L1+R1+X as a gamepad overlay combo
384 if (state->buttons == (BACK_FLAG | LB_FLAG | RB_FLAG | X_FLAG)) {
385 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
386 "Detected stats toggle gamepad combo");
387
388 // Toggle the stats overlay
389 Session::get()->getOverlayManager().setOverlayState(Overlay::OverlayDebug,
390 !Session::get()->getOverlayManager().isOverlayEnabled(Overlay::OverlayDebug));
391
392 // Clear buttons down on this gamepad
393 LiSendMultiControllerEvent(state->index, m_GamepadMask,
394 0, 0, 0, 0, 0, 0, 0);
395 return;
396 }
397
398 // Only send the gamepad state to the host if it's not in mouse emulation mode
399 if (state->mouseEmulationTimer == 0) {
400 sendGamepadState(state);
401 }
402}

427 memcmp(event->data, state->lastGyroEventData, sizeof(event->data)) != 0) {

482void SdlInputHandler::handleControllerDeviceEvent(SDL_ControllerDeviceEvent* event)
483{
484 GamepadState* state;
485
486 if (event->type == SDL_CONTROLLERDEVICEADDED) {
487 int i;
488 const char* name;
489 SDL_GameController* controller;
490 const char* mapping;
491 char guidStr[33];
492 uint32_t hapticCaps;
493
494 controller = SDL_GameControllerOpen(event->which);
495 if (controller == NULL) {
496 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
497 "Failed to open gamepad: %s",
498 SDL_GetError());
499 return;
500 }
501
502 // SDL_CONTROLLERDEVICEADDED can be reported multiple times for the same
503 // gamepad in rare cases, because SDL doesn't fixup the device index in
504 // the SDL_CONTROLLERDEVICEADDED event if an unopened gamepad disappears
505 // before we've processed the add event.
506 for (int i = 0; i < MAX_GAMEPADS; i++) {
507 if (m_GamepadState[i].controller == controller) {
508 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
509 "Received duplicate add event for controller index: %d",
510 event->which);
511 SDL_GameControllerClose(controller);
512 return;
513 }
514 }
515
516 // We used to use SDL_GameControllerGetPlayerIndex() here but that
517 // can lead to strange issues due to bugs in Windows where an Xbox
518 // controller will join as player 2, even though no player 1 controller
519 // is connected at all. This pretty much screws any attempt to use
520 // the gamepad in single player games, so just assign them in order from 0.
521 i = 0;
522
523 for (; i < MAX_GAMEPADS; i++) {
524 SDL_assert(m_GamepadState[i].controller != controller);
525 if (m_GamepadState[i].controller == NULL) {
526 // Found an empty slot
527 break;
528 }
529 }
530
531 if (i == MAX_GAMEPADS) {
532 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
533 "No open gamepad slots found!");
534 SDL_GameControllerClose(controller);
535 return;
536 }
537
538 SDL_JoystickGetGUIDString(SDL_JoystickGetGUID(SDL_GameControllerGetJoystick(controller)),
539 guidStr, sizeof(guidStr));
540 if (m_IgnoreDeviceGuids.contains(guidStr, Qt::CaseInsensitive))
541 {
542 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
543 "Skipping ignored device with GUID: %s",
544 guidStr);
545 SDL_GameControllerClose(controller);
546 return;
547 }
548
549 state = &m_GamepadState[i];
550 if (m_MultiController) {
551 state->index = i;
552
553#if SDL_VERSION_ATLEAST(2, 0, 12)
554 // This will change indicators on the controller to show the assigned
555 // player index. For Xbox 360 controllers, that means updating the LED
556 // ring to light up the corresponding quadrant for this player.
557 SDL_GameControllerSetPlayerIndex(controller, state->index);
558#endif
559 }
560 else {
561 // Always player 1 in single controller mode
562 state->index = 0;
563 }
564
565 state->controller = controller;
566 state->jsId = SDL_JoystickInstanceID(SDL_GameControllerGetJoystick(state->controller));
567
568 hapticCaps = 0;
569#if SDL_VERSION_ATLEAST(2, 0, 18)
570 hapticCaps |= SDL_GameControllerHasRumble(controller) ? ML_HAPTIC_GC_RUMBLE : 0;
571 hapticCaps |= SDL_GameControllerHasRumbleTriggers(controller) ? ML_HAPTIC_GC_TRIGGER_RUMBLE : 0;
572#elif SDL_VERSION_ATLEAST(2, 0, 9)
573 // Perform a tiny rumbles to see if haptics are supported.
574 // NB: We cannot use zeros for rumble intensity or SDL will not actually call the JS driver
575 // and we'll get a (potentially false) success value returned.
576 hapticCaps |= SDL_GameControllerRumble(controller, 1, 1, 1) == 0 ? ML_HAPTIC_GC_RUMBLE : 0;
577#if SDL_VERSION_ATLEAST(2, 0, 14)
578 hapticCaps |= SDL_GameControllerRumbleTriggers(controller, 1, 1, 1) == 0 ? ML_HAPTIC_GC_TRIGGER_RUMBLE : 0;
579#endif
580#else
581 state->haptic = SDL_HapticOpenFromJoystick(SDL_GameControllerGetJoystick(state->controller));
582 state->hapticEffectId = -1;
583 state->hapticMethod = GAMEPAD_HAPTIC_METHOD_NONE;
584 if (state->haptic != nullptr) {
585 // Query for supported haptic effects
586 hapticCaps = SDL_HapticQuery(state->haptic);
587 hapticCaps |= SDL_HapticRumbleSupported(state->haptic) ?
588 ML_HAPTIC_SIMPLE_RUMBLE : 0;
589
590 if ((SDL_HapticQuery(state->haptic) & SDL_HAPTIC_LEFTRIGHT) == 0) {
591 if (SDL_HapticRumbleSupported(state->haptic)) {
592 if (SDL_HapticRumbleInit(state->haptic) == 0) {
593 state->hapticMethod = GAMEPAD_HAPTIC_METHOD_SIMPLERUMBLE;
594 }
595 }
596 if (state->hapticMethod == GAMEPAD_HAPTIC_METHOD_NONE) {
597 SDL_HapticClose(state->haptic);
598 state->haptic = nullptr;
599 }
600 } else {
601 state->hapticMethod = GAMEPAD_HAPTIC_METHOD_LEFTRIGHT;
602 }
603 }
604 else {
605 hapticCaps = 0;
606 }
607#endif
608
609 mapping = SDL_GameControllerMapping(state->controller);
610 name = SDL_GameControllerName(state->controller);
611
612 uint16_t vendorId = SDL_GameControllerGetVendor(state->controller);
613 uint16_t productId = SDL_GameControllerGetProduct(state->controller);
614 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
615 "Gamepad %d (player %d) is: %s (VID/PID: 0x%.4x/0x%.4x) (haptic capabilities: 0x%x) (mapping: %s -> %s)",
616 i,
617 state->index,
618 name != nullptr ? name : "<null>",
619 vendorId,
620 productId,
621 hapticCaps,
622 guidStr,
623 mapping != nullptr ? mapping : "<null>");
624 if (mapping != nullptr) {
625 SDL_free((void*)mapping);
626 }
627
628 // Add this gamepad to the gamepad mask
629 if (m_MultiController) {
630 // NB: Don't assert that it's unset here because we will already
631 // have the mask set for initially attached gamepads to avoid confusing
632 // apps running on the host.
633 m_GamepadMask |= (1 << state->index);
634 }
635 else {
636 SDL_assert(m_GamepadMask == 0x1);
637 }
638
639 SDL_JoystickPowerLevel powerLevel = SDL_JoystickCurrentPowerLevel(SDL_GameControllerGetJoystick(state->controller));
640
641#if SDL_VERSION_ATLEAST(2, 0, 14)
642 // On SDL 2.0.14 and later, we can provide enhanced controller information to the host PC
643 // for it to use as a hint for the type of controller to emulate.
644 uint32_t supportedButtonFlags = 0;
645 for (int i = 0; i < (int)SDL_arraysize(k_ButtonMap); i++) {
646 if (SDL_GameControllerHasButton(state->controller, (SDL_GameControllerButton)i)) {
647 supportedButtonFlags |= k_ButtonMap[i];
648 }
649 }
650
651 uint32_t capabilities = 0;
652 if (SDL_GameControllerGetBindForAxis(state->controller, SDL_CONTROLLER_AXIS_TRIGGERLEFT).bindType == SDL_CONTROLLER_BINDTYPE_AXIS ||
653 SDL_GameControllerGetBindForAxis(state->controller, SDL_CONTROLLER_AXIS_TRIGGERRIGHT).bindType == SDL_CONTROLLER_BINDTYPE_AXIS) {
654 // We assume these are analog triggers if the binding is to an axis rather than a button
655 capabilities |= LI_CCAP_ANALOG_TRIGGERS;
656 }
657 if (hapticCaps & ML_HAPTIC_GC_RUMBLE) {
658 capabilities |= LI_CCAP_RUMBLE;
659 }
660 if (hapticCaps & ML_HAPTIC_GC_TRIGGER_RUMBLE) {
661 capabilities |= LI_CCAP_TRIGGER_RUMBLE;
662 }
663 if (SDL_GameControllerGetNumTouchpads(state->controller) > 0) {
664 capabilities |= LI_CCAP_TOUCHPAD;
665 }
666 if (SDL_GameControllerHasSensor(state->controller, SDL_SENSOR_ACCEL)) {
667 capabilities |= LI_CCAP_ACCEL;
668 }
669 if (SDL_GameControllerHasSensor(state->controller, SDL_SENSOR_GYRO)) {
670 capabilities |= LI_CCAP_GYRO;
671 }
672 if (powerLevel != SDL_JOYSTICK_POWER_UNKNOWN || SDL_VERSION_ATLEAST(2, 24, 0)) {
673 capabilities |= LI_CCAP_BATTERY_STATE;
674 }
675 if (SDL_GameControllerHasLED(state->controller)) {
676 capabilities |= LI_CCAP_RGB_LED;
677 }
678
679 uint8_t type;
680 switch (SDL_GameControllerGetType(state->controller)) {
681 case SDL_CONTROLLER_TYPE_XBOX360:
682 case SDL_CONTROLLER_TYPE_XBOXONE:
683 type = LI_CTYPE_XBOX;
684 break;
685 case SDL_CONTROLLER_TYPE_PS3:
686 case SDL_CONTROLLER_TYPE_PS4:
687 case SDL_CONTROLLER_TYPE_PS5:
688 type = LI_CTYPE_PS;
689 break;
690 case SDL_CONTROLLER_TYPE_NINTENDO_SWITCH_PRO:
691#if SDL_VERSION_ATLEAST(2, 24, 0)
692 case SDL_CONTROLLER_TYPE_NINTENDO_SWITCH_JOYCON_LEFT:
693 case SDL_CONTROLLER_TYPE_NINTENDO_SWITCH_JOYCON_RIGHT:
694 case SDL_CONTROLLER_TYPE_NINTENDO_SWITCH_JOYCON_PAIR:
695#endif
696 type = LI_CTYPE_NINTENDO;
697 break;
698 default:
699 type = LI_CTYPE_UNKNOWN;
700 break;
701 }
702
703 // If this is a PlayStation controller that doesn't have a touchpad button mapped,
704 // we'll allow the Select+PS button combo to act as the touchpad.
705 state->clickpadButtonEmulationEnabled =
706#if SDL_VERSION_ATLEAST(2, 0, 14)
707 SDL_GameControllerGetBindForButton(state->controller, SDL_CONTROLLER_BUTTON_TOUCHPAD).bindType == SDL_CONTROLLER_BINDTYPE_NONE &&
708#endif
709 type == LI_CTYPE_PS;
710
711 LiSendControllerArrivalEvent(state->index, m_GamepadMask, type, supportedButtonFlags, capabilities);
712#else
713
714 // Send an empty event to tell the PC we've arrived
715 sendGamepadState(state);
716#endif
717
718 // Send a power level if it's known at this time
719 if (powerLevel != SDL_JOYSTICK_POWER_UNKNOWN) {
720 sendGamepadBatteryState(state, powerLevel);
721 }
722 }
723 else if (event->type == SDL_CONTROLLERDEVICEREMOVED) {
724 state = findStateForGamepad(event->which);
725 if (state != NULL) {
726 if (state->mouseEmulationTimer != 0) {
727 Session::get()->notifyMouseEmulationMode(false);
728 SDL_RemoveTimer(state->mouseEmulationTimer);
729 }
730
731 SDL_GameControllerClose(state->controller);
732
733#if !SDL_VERSION_ATLEAST(2, 0, 9)
734 if (state->haptic != nullptr) {
735 SDL_HapticClose(state->haptic);
736 }
737#endif
738
739 // Remove this from the gamepad mask in MC-mode
740 if (m_MultiController) {
741 SDL_assert(m_GamepadMask & (1 << state->index));
742 m_GamepadMask &= ~(1 << state->index);
743 }
744 else {
745 SDL_assert(m_GamepadMask == 0x1);
746 }
747
748 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
749 "Gamepad %d is gone",
750 state->index);
751
752 // Send a final event to let the PC know this gamepad is gone
753 LiSendMultiControllerEvent(state->index, m_GamepadMask,
754 0, 0, 0, 0, 0, 0, 0);
755
756 // Clear all remaining state from this slot
757 SDL_memset(state, 0, sizeof(*state));
758 }
759 }

791void SdlInputHandler::rumble(unsigned short controllerNumber, unsigned short lowFreqMotor, unsigned short highFreqMotor)
792{
793 // Make sure the controller number is within our supported count
794 if (controllerNumber >= MAX_GAMEPADS) {
795 return;
796 }
797
798#if SDL_VERSION_ATLEAST(2, 0, 9)
799 if (m_GamepadState[controllerNumber].controller != nullptr) {
800 SDL_GameControllerRumble(m_GamepadState[controllerNumber].controller, lowFreqMotor, highFreqMotor, 30000);
801 }
802#else
803 // Check if the controller supports haptics (and if the controller exists at all)
804 SDL_Haptic* haptic = m_GamepadState[controllerNumber].haptic;
805 if (haptic == nullptr) {
806 return;
807 }
808
809 // Stop the last effect we played
810 if (m_GamepadState[controllerNumber].hapticMethod == GAMEPAD_HAPTIC_METHOD_LEFTRIGHT) {
811 if (m_GamepadState[controllerNumber].hapticEffectId >= 0) {
812 SDL_HapticDestroyEffect(haptic, m_GamepadState[controllerNumber].hapticEffectId);
813 }
814 } else if (m_GamepadState[controllerNumber].hapticMethod == GAMEPAD_HAPTIC_METHOD_SIMPLERUMBLE) {
815 SDL_HapticRumbleStop(haptic);
816 }
817
818 // If this callback is telling us to stop both motors, don't bother queuing a new effect
819 if (lowFreqMotor == 0 && highFreqMotor == 0) {
820 return;
821 }
822
823 if (m_GamepadState[controllerNumber].hapticMethod == GAMEPAD_HAPTIC_METHOD_LEFTRIGHT) {
824 SDL_HapticEffect effect;
825 SDL_memset(&effect, 0, sizeof(effect));
826 effect.type = SDL_HAPTIC_LEFTRIGHT;
827
828 // The effect should last until we are instructed to stop or change it
829 effect.leftright.length = SDL_HAPTIC_INFINITY;
830
831 // SDL haptics range from 0-32767 but XInput uses 0-65535, so divide by 2 to correct for SDL's scaling
832 effect.leftright.large_magnitude = lowFreqMotor / 2;
833 effect.leftright.small_magnitude = highFreqMotor / 2;
834
835 // Play the new effect
836 m_GamepadState[controllerNumber].hapticEffectId = SDL_HapticNewEffect(haptic, &effect);
837 if (m_GamepadState[controllerNumber].hapticEffectId >= 0) {
838 SDL_HapticRunEffect(haptic, m_GamepadState[controllerNumber].hapticEffectId, 1);
839 }
840 } else if (m_GamepadState[controllerNumber].hapticMethod == GAMEPAD_HAPTIC_METHOD_SIMPLERUMBLE) {
841 SDL_HapticRumblePlay(haptic,
842 std::min(1.0, (GAMEPAD_HAPTIC_SIMPLE_HIFREQ_MOTOR_WEIGHT*highFreqMotor +
843 GAMEPAD_HAPTIC_SIMPLE_LOWFREQ_MOTOR_WEIGHT*lowFreqMotor) / 65535.0),
844 SDL_HAPTIC_INFINITY);
845 }
846#endif
847}
