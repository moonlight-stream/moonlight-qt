111void logToLoggerStream(QString& message)
112{
113#if defined(QT_DEBUG) && defined(Q_OS_WIN32)
114 // Output log messages to a debugger if attached
115 if (IsDebuggerPresent()) {
116 thread_local QString lineBuffer;
117 lineBuffer += message;
118 if (message.endsWith('\n')) {
119 OutputDebugStringW(lineBuffer.toStdWString().c_str());
120 lineBuffer.clear();
121 }
122 }
123#endif
124
125 // Strip session encryption keys and IVs from the logs
126 message.replace(k_RikeyRegex, "&rikey=REDACTED");
127 message.replace(k_RikeyIdRegex, "&rikeyid=REDACTED");
128
129#ifdef LOG_TO_FILE
130 auto oldLogSize = s_LogBytesWritten.fetchAndAddRelaxed(message.size());
131 if (oldLogSize >= k_MaxLogSizeBytes) {
132 return;
133 }
134 else if (oldLogSize >= k_MaxLogSizeBytes - message.size()) {
135 // Write one final message
136 message = "Log size limit reached!";
137 }
138#endif
139
140 if (g_AsyncLoggingEnabled) {
141 // Queue the log message to be written asynchronously
142 s_LoggerThread.start(new LoggerTask(message));
143 }
144 else {
145 // Log the message immediately
146 LoggerTask(message).run();
147 }
148}

150void sdlLogToDiskHandler(void*, int category, SDL_LogPriority priority, const char* message)

183 priorityTxt = "Critical";

190 QTime logTime = QTime::fromMSecsSinceStartOfDay(s_LoggerTime.elapsed());
191 QString txt = QString("%1 - SDL %2 (%3): %4\n").arg(logTime.toString()).arg(priorityTxt).arg(category).arg(message);
192
193 logToLoggerStream(txt);

274LONG WINAPI UnhandledExceptionHandler(struct _EXCEPTION_POINTERS *ExceptionInfo)

276 // Only write a dump for the first unhandled exception
277 if (InterlockedCompareExchange(&s_HitUnhandledException, 1, 0) != 0) {

280
281 WCHAR dmpFileName[MAX_PATH];
282 swprintf_s(dmpFileName, L"%ls\\Moonlight-%I64u.dmp",
283 (PWCHAR)QDir::toNativeSeparators(Path::getLogDir()).utf16(), QDateTime::currentSecsSinceEpoch());
284 QString qDmpFileName = QString::fromUtf16((const char16_t*)dmpFileName);
285 HANDLE dumpHandle = CreateFileW(dmpFileName, GENERIC_WRITE, 0, nullptr, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, nullptr);
286 if (dumpHandle != INVALID_HANDLE_VALUE) {
287 MINIDUMP_EXCEPTION_INFORMATION info;

290 info.ExceptionPointers = ExceptionInfo;
291 info.ClientPointers = FALSE;

293 DWORD typeFlags = MiniDumpWithIndirectlyReferencedMemory |
294 MiniDumpIgnoreInaccessibleMemory |
295 MiniDumpWithUnloadedModules |
296 MiniDumpWithThreadInfo;
297
298 if (MiniDumpWriteDump(GetCurrentProcess(),
299 GetCurrentProcessId(),
300 dumpHandle,
301 (MINIDUMP_TYPE)typeFlags,
302 &info,
303 nullptr,
304 nullptr)) {
305 qCritical() << "Unhandled exception! Minidump written to:" << qDmpFileName;
306 }
307 else {
308 qCritical() << "Unhandled exception! Failed to write dump:" << GetLastError();
309 }
310
311 CloseHandle(dumpHandle);
312 }
313 else {
314 qCritical() << "Unhandled exception! Failed to open dump file:" << qDmpFileName << "with error" << GetLastError();
315 }
316
317 // Sleep for a moment to allow the logging thread to finish up before crashing
318 if (g_AsyncLoggingEnabled) {
319 Sleep(500);
320 }
321
322 // Let the program crash and WER collect a dump
323 return EXCEPTION_CONTINUE_SEARCH;
324}
