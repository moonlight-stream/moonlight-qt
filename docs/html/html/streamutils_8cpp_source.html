87void StreamUtils::scaleSourceToDestinationSurface(SDL_Rect* src, SDL_Rect* dst)
88{
89 int dstH = SDL_ceilf((float)dst->w * src->h / src->w);
90 int dstW = SDL_ceilf((float)dst->h * src->w / src->h);
91
92 if (dstH > dst->h) {
93 dst->x += (dst->w - dstW) / 2;
94 dst->w = dstW;
95 }
96 else {
97 dst->y += (dst->h - dstH) / 2;
98 dst->h = dstH;

118int StreamUtils::getDisplayRefreshRate(SDL_Window* window)
119{
120 int displayIndex = SDL_GetWindowDisplayIndex(window);
121 if (displayIndex < 0) {
122 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
123 "Failed to get current display: %s",
124 SDL_GetError());
125
126 // Assume display 0 if it fails
127 displayIndex = 0;
128 }
129
130 SDL_DisplayMode mode;
131 if ((SDL_GetWindowFlags(window) & SDL_WINDOW_FULLSCREEN_DESKTOP) == SDL_WINDOW_FULLSCREEN) {
132 // Use the window display mode for full-screen exclusive mode
133 if (SDL_GetWindowDisplayMode(window, &mode) != 0) {
134 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
135 "SDL_GetWindowDisplayMode() failed: %s",
136 SDL_GetError());
137
138 // Assume 60 Hz
139 return 60;
140 }
141 }
142 else {
143 // Use the current display mode for windowed and borderless
144 if (SDL_GetCurrentDisplayMode(displayIndex, &mode) != 0) {
145 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
146 "SDL_GetCurrentDisplayMode() failed: %s",
147 SDL_GetError());
148
149 // Assume 60 Hz
150 return 60;
151 }
152 }
153
154 // May be zero if undefined
155 if (mode.refresh_rate == 0) {
156 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
157 "Refresh rate unknown; assuming 60 Hz");
158 mode.refresh_rate = 60;
159 }
160
161 return mode.refresh_rate;

164bool StreamUtils::hasFastAes()
165{
166#ifndef __has_builtin
167#define __has_builtin(x) 0
168#endif
169
170#if (__has_builtin(__builtin_cpu_supports) || (defined(__GNUC__) && __GNUC__ >= 6)) && defined(Q_PROCESSOR_X86)
171 return __builtin_cpu_supports("aes");
172#elif defined(__BUILTIN_CPU_SUPPORTS__) && defined(Q_PROCESSOR_POWER)
173 return __builtin_cpu_supports("vcrypto");
174#elif defined(Q_OS_WINDOWS) && defined(Q_PROCESSOR_ARM)
175 return IsProcessorFeaturePresent(PF_ARM_V8_CRYPTO_INSTRUCTIONS_AVAILABLE);
176#elif defined(_MSC_VER) && defined(Q_PROCESSOR_X86)
177 int regs[4];
178 __cpuid(regs, 1);
179 return regs[2] & (1 << 25); // AES-NI
180#elif defined(Q_OS_DARWIN)
181 // Everything that runs Catalina and later has AES-NI or ARMv8 crypto instructions
182 return true;
183#elif (defined(Q_OS_FREEBSD) || defined(Q_OS_OPENBSD)) && defined(Q_PROCESSOR_ARM) && QT_POINTER_SIZE == 4
184 unsigned long hwcap2 = 0;
185 elf_aux_info(AT_HWCAP2, &hwcap2, sizeof(hwcap2));
186 return (hwcap2 & HWCAP2_AES);
187#elif (defined(Q_OS_FREEBSD) || defined(Q_OS_OPENBSD)) && defined(Q_PROCESSOR_ARM) && QT_POINTER_SIZE == 8
188 unsigned long hwcap = 0;
189 elf_aux_info(AT_HWCAP, &hwcap, sizeof(hwcap));
190 return (hwcap & HWCAP_AES);
191#elif defined(Q_OS_LINUX) && defined(Q_PROCESSOR_ARM) && QT_POINTER_SIZE == 4
192 return getauxval(AT_HWCAP2) & HWCAP2_AES;
193#elif defined(Q_OS_LINUX) && defined(Q_PROCESSOR_ARM) && QT_POINTER_SIZE == 8
194 return getauxval(AT_HWCAP) & HWCAP_AES;
195#elif defined(Q_OS_LINUX) && defined(Q_PROCESSOR_RISCV)
196 riscv_hwprobe pairs[1] = {
197 { RISCV_HWPROBE_KEY_IMA_EXT_0, 0 },
198 };
199
200 // If this syscall is not implemented, we'll get -ENOSYS
201 // and the value field will remain zero.
202 __riscv_hwprobe(pairs, SDL_arraysize(pairs), 0, nullptr, 0);
203
204 return (pairs[0].value & (RISCV_HWPROBE_EXT_ZKNE | RISCV_HWPROBE_EXT_ZKND)) ==
205 (RISCV_HWPROBE_EXT_ZKNE | RISCV_HWPROBE_EXT_ZKND) ||
206 (pairs[0].value & RISCV_HWPROBE_EXT_ZVKNED);
207#elif QT_POINTER_SIZE == 4
208 #warning Unknown 32-bit platform. Assuming AES is slow on this CPU.
209 return false;
210#else
211 #warning Unknown 64-bit platform. Assuming AES is fast on this CPU.
212 return true;
213#endif
214}

216bool StreamUtils::getNativeDesktopMode(int displayIndex, SDL_DisplayMode* mode, SDL_Rect* safeArea)
217{
218#ifdef Q_OS_DARWIN
219#define MAX_DISPLAYS 16
220 CGDirectDisplayID displayIds[MAX_DISPLAYS];
221 uint32_t displayCount = 0;
222 CGGetActiveDisplayList(MAX_DISPLAYS, displayIds, &displayCount);
223 if (displayIndex >= (int)displayCount) {
224 return false;
225 }
226
227 SDL_zerop(mode);
228
229 // Retina displays have non-native resolutions both below and above (!) their
230 // native resolution, so it's impossible for us to figure out what's actually
231 // native on macOS using the SDL API alone. We'll talk to CoreGraphics to
232 // find the correct resolution and match it in our SDL list.
233 CFArrayRef modeList = CGDisplayCopyAllDisplayModes(displayIds[displayIndex], nullptr);
234 CFIndex count = CFArrayGetCount(modeList);
235 for (CFIndex i = 0; i < count; i++) {
236 auto cgMode = (CGDisplayModeRef)(CFArrayGetValueAtIndex(modeList, i));
237 if ((CGDisplayModeGetIOFlags(cgMode) & kDisplayModeNativeFlag) != 0) {
238 mode->w = static_cast<int>(CGDisplayModeGetWidth(cgMode));
239 mode->h = static_cast<int>(CGDisplayModeGetHeight(cgMode));
240 break;
241 }
242 }
243
244 safeArea->x = 0;
245 safeArea->y = 0;
246 safeArea->w = mode->w;
247 safeArea->h = mode->h;
248
249#if TARGET_CPU_ARM64
250 // Now that we found the native full-screen mode, let's look for one that matches along
251 // the width but not the height and we'll assume that's the safe area full-screen mode.
252 //
253 // There doesn't appear to be a CG API or flag that will tell us that a given mode
254 // is a "safe area" mode, so we have to use our own (brittle) heuristics. :(
255 //
256 // To avoid potential false positives, let's avoid checking for external displays, since
257 // we might have scenarios like a 1920x1200 display with an alternate 1920x1080 mode
258 // which would falsely trigger our notch detection here.
259 if (CGDisplayIsBuiltin(displayIds[displayIndex])) {
260 for (CFIndex i = 0; i < count; i++) {
261 auto cgMode = (CGDisplayModeRef)(CFArrayGetValueAtIndex(modeList, i));
262 auto cgModeWidth = static_cast<int>(CGDisplayModeGetWidth(cgMode));
263 auto cgModeHeight = static_cast<int>(CGDisplayModeGetHeight(cgMode));
264
265 // If the modes differ by more than 100, we'll assume it's not a notch mode
266 if (mode->w == cgModeWidth && mode->h != cgModeHeight && mode->h <= cgModeHeight + 100) {
267 safeArea->w = cgModeWidth;
268 safeArea->h = cgModeHeight;
269 }
270 }
271 }
272#endif
273
274 CFRelease(modeList);
275
276 // Special case for probing for notched displays prior to video subsystem initialization
277 // in Session::initialize() for Darwin only!
278 if (SDL_WasInit(SDL_INIT_VIDEO)) {
279 // Now find the SDL mode that matches the CG native mode
280 for (int i = 0; i < SDL_GetNumDisplayModes(displayIndex); i++) {
281 SDL_DisplayMode thisMode;
282 if (SDL_GetDisplayMode(displayIndex, i, &thisMode) == 0) {
283 if (thisMode.w == mode->w && thisMode.h == mode->h &&
284 thisMode.refresh_rate >= mode->refresh_rate) {
285 *mode = thisMode;
286 break;
287 }
288 }
289 }
290 }
291#else
292 SDL_assert(SDL_WasInit(SDL_INIT_VIDEO));
293
294 if (displayIndex >= SDL_GetNumVideoDisplays()) {
295 return false;
296 }
297
298 // We need to get the true display resolution without DPI scaling (since we use High DPI).
299 // Windows returns the real display resolution here, even if DPI scaling is enabled.
300 // macOS and Wayland report a resolution that includes the DPI scaling factor. Picking
301 // the first mode on Wayland will get the native resolution without the scaling factor
302 // (and macOS is handled in the #ifdef above).
303 if (!strcmp(SDL_GetCurrentVideoDriver(), "wayland")) {
304 if (SDL_GetDisplayMode(displayIndex, 0, mode) != 0) {
305 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
306 "SDL_GetDisplayMode() failed: %s",
307 SDL_GetError());
308 return false;
309 }
310 }
311 else {
312 if (SDL_GetDesktopDisplayMode(displayIndex, mode) != 0) {
313 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
314 "SDL_GetDesktopDisplayMode() failed: %s",
315 SDL_GetError());
316 return false;
317 }
318 }
319
320 safeArea->x = 0;
321 safeArea->y = 0;
322 safeArea->w = mode->w;
323 safeArea->h = mode->h;
324#endif
325
326 return true;
327}

329int StreamUtils::getDrmFdForWindow(SDL_Window* window, bool* mustClose)
330{
331 *mustClose = false;
332
333#if defined(SDL_VIDEO_DRIVER_KMSDRM) && SDL_VERSION_ATLEAST(2, 0, 15)
334 SDL_SysWMinfo info;
335 SDL_VERSION(&info.version);
336 if (!SDL_GetWindowWMInfo(window, &info)) {
337 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
338 "SDL_GetWindowWMInfo() failed: %s",
339 SDL_GetError());
340 return -1;
341 }
342
343 if (info.subsystem == SDL_SYSWM_KMSDRM) {
344 // If SDL has an FD, share that
345 if (info.info.kmsdrm.drm_fd >= 0) {
346 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
347 "Sharing DRM FD with SDL");
348 return info.info.kmsdrm.drm_fd;
349 }
350 else {
351 char path[128];
352 snprintf(path, sizeof(path), "/dev/dri/card%u", info.info.kmsdrm.dev_index);
353 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
354 "Opening DRM FD from SDL by path: %s",
355 path);
356 int fd = open(path, O_RDWR | O_CLOEXEC);
357 if (fd >= 0) {
358 *mustClose = true;
359 }
360 return fd;
361 }
362 }
363#else
364 Q_UNUSED(window);
365#endif
366
367 return -1;
368}

370int StreamUtils::getDrmFd(bool preferRenderNode)
371{
372#ifdef Q_OS_UNIX
373 const char* userDevice = SDL_getenv("DRM_DEV");
374 if (userDevice != nullptr) {
375 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
376 "Opening user-specified DRM device: %s",
377 userDevice);
378
379 return open(userDevice, O_RDWR | O_CLOEXEC);
380 }
381 else {
382 QDir driDir("/dev/dri");
383 int fd;
384
385 // We have to explicitly ask for devices to be returned
386 driDir.setFilter(QDir::Files | QDir::System);
387
388 if (preferRenderNode) {
389 // Try a render node first since we aren't using DRM for output in this codepath
390 for (QFileInfo& node : driDir.entryInfoList(QStringList("renderD*"))) {
391 QByteArray absolutePath = node.absoluteFilePath().toUtf8();
392 fd = open(absolutePath.constData(), O_RDWR | O_CLOEXEC);
393 if (fd >= 0) {
394 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
395 "Opened DRM render node: %s",
396 absolutePath.constData());
397 return fd;
398 }
399 }
400 }
401
402 // If that fails, try to use a primary node and hope for the best
403 for (QFileInfo& node : driDir.entryInfoList(QStringList("card*"))) {
404 QByteArray absolutePath = node.absoluteFilePath().toUtf8();
405 fd = open(absolutePath.constData(), O_RDWR | O_CLOEXEC);
406 if (fd >= 0) {
407 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
408 "Opened DRM primary node: %s",
409 absolutePath.constData());
410 return fd;
411 }
412 }
413 }
414#else
415 Q_UNUSED(preferRenderNode);
416#endif
417
418 return -1;
419}
