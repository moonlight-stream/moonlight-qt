271void SdlInputHandler::notifyMouseLeave()
272{
273 // SDL on Windows doesn't send the mouse button up until the mouse re-enters the window
274 // after leaving it. This breaks some of the Aero snap gestures, so we'll capture it to
275 // allow us to receive the mouse button up events later.
276 //
277 // On macOS and X11, capturing the mouse allows us to receive mouse motion outside the
278 // window (button up already worked without capture).
279 if (m_AbsoluteMouseMode && isCaptureActive()) {
280 // NB: Not using SDL_GetGlobalMouseState() because we want our state not the system's
281 Uint32 mouseState = SDL_GetMouseState(nullptr, nullptr);
282 for (Uint32 button = SDL_BUTTON_LEFT; button <= SDL_BUTTON_X2; button++) {
283 if (mouseState & SDL_BUTTON(button)) {
284 SDL_CaptureMouse(SDL_TRUE);
285 break;
286 }
287 }
288 }

291void SdlInputHandler::notifyFocusLost()
292{
293 // Release mouse cursor when another window is activated (e.g. by using ALT+TAB).
294 // This lets user to interact with our window's title bar and with the buttons in it.
295 // Doing this while the window is full-screen breaks the transition out of FS
296 // (desktop and exclusive), so we must check for that before releasing mouse capture.

298 setCaptureActive(false);

301 // Raise all keys that are currently pressed. If we don't do this, certain keys
302 // used in shortcuts that cause focus loss (such as Alt+Tab) may get stuck down.

331void SdlInputHandler::updateKeyboardGrabState()
332{
333 if (m_CaptureSystemKeysMode == StreamingPreferences::CSK_OFF) {
334 return;
335 }
336
337 bool shouldGrab = isCaptureActive();
338 Uint32 windowFlags = SDL_GetWindowFlags(m_Window);
339 if (m_CaptureSystemKeysMode == StreamingPreferences::CSK_FULLSCREEN &&
340 !(windowFlags & SDL_WINDOW_FULLSCREEN)) {
341 // Ungrab if it's fullscreen only and we left fullscreen
342 shouldGrab = false;
343 }
344
345 // Don't close the window on Alt+F4 when keyboard grab is enabled
346 SDL_SetHint(SDL_HINT_WINDOWS_NO_CLOSE_ON_ALT_F4, shouldGrab ? "1" : "0");
347
348#if SDL_VERSION_ATLEAST(2, 0, 15)
349 // On SDL 2.0.15+, we can get keyboard-only grab on Win32, X11, and Wayland.
350 // SDL 2.0.18 adds keyboard grab on macOS (if built with non-AppStore APIs).
351 SDL_SetWindowKeyboardGrab(m_Window, shouldGrab ? SDL_TRUE : SDL_FALSE);
352#endif

385void SdlInputHandler::setCaptureActive(bool active)
386{
387 if (active) {
388 // If we're in relative mode, try to activate SDL's relative mouse mode
389 if (m_AbsoluteMouseMode || SDL_SetRelativeMouseMode(SDL_TRUE) < 0) {
390 // Relative mouse mode didn't work or was disabled, so we'll just hide the cursor
391 SDL_ShowCursor(m_MouseCursorCapturedVisibilityState);
392 m_FakeCaptureActive = true;
393 }
394
395 // Synchronize the client and host cursor when activating absolute capture
396 if (m_AbsoluteMouseMode) {
397 int mouseX, mouseY;
398 int windowX, windowY;
399
400 // We have to use SDL_GetGlobalMouseState() because macOS may not reflect
401 // the new position of the mouse when outside the window.
402 SDL_GetGlobalMouseState(&mouseX, &mouseY);
403
404 // Convert global mouse state to window-relative
405 SDL_GetWindowPosition(m_Window, &windowX, &windowY);
406 mouseX -= windowX;
407 mouseY -= windowY;
408
409 if (isMouseInVideoRegion(mouseX, mouseY)) {
410 // Synthesize a mouse event to synchronize the cursor
411 SDL_MouseMotionEvent motionEvent = {};
412 motionEvent.type = SDL_MOUSEMOTION;
413 motionEvent.timestamp = SDL_GetTicks();
414 motionEvent.windowID = SDL_GetWindowID(m_Window);
415 motionEvent.x = mouseX;
416 motionEvent.y = mouseY;
417 handleMouseMotionEvent(&motionEvent);
418 }
419 }
420 }
421 else {
422 if (m_FakeCaptureActive) {
423 // Display the cursor again
424 SDL_ShowCursor(SDL_ENABLE);
425 m_FakeCaptureActive = false;
426 }
427 else {
428 SDL_SetRelativeMouseMode(SDL_FALSE);
429 }
430 }
431
432 // Update mouse pointer region constraints
433 updatePointerRegionLock();
434
435 // Now update the keyboard grab
436 updateKeyboardGrabState();
437}

440{

452 // to tell them apart.
