98EGLRenderer::~EGLRenderer()
99{
100 if (m_Context) {
101 // Reattach the GL context to the main thread for destruction
102 SDL_GL_MakeCurrent(m_Window, m_Context);
103 if (m_LastRenderSync != EGL_NO_SYNC) {
104 SDL_assert(m_eglDestroySync != nullptr);
105 m_eglDestroySync(m_EGLDisplay, m_LastRenderSync);
106 }
107 if (m_ShaderProgram) {
108 glDeleteProgram(m_ShaderProgram);
109 }
110 if (m_OverlayShaderProgram) {
111 glDeleteProgram(m_OverlayShaderProgram);
112 }
113 if (m_VAO) {
114 SDL_assert(m_glDeleteVertexArraysOES != nullptr);
115 m_glDeleteVertexArraysOES(1, &m_VAO);
116 }
117 for (int i = 0; i < EGL_MAX_PLANES; i++) {
118 if (m_Textures[i] != 0) {
119 glDeleteTextures(1, &m_Textures[i]);
120 }
121 }
122 for (int i = 0; i < Overlay::OverlayMax; i++) {
123 if (m_OverlayTextures[i] != 0) {
124 glDeleteTextures(1, &m_OverlayTextures[i]);
125 }
126 if (m_OverlayVbos[i] != 0) {
127 glDeleteBuffers(1, &m_OverlayVbos[i]);
128 }
129 }
130 SDL_GL_DeleteContext(m_Context);
131 }
132
133 if (m_DummyRenderer) {
134 SDL_DestroyRenderer(m_DummyRenderer);
135 }
136
137 av_frame_free(&m_LastFrame);
138
139 // Reset the global properties back to what they were before
140 SDL_SetHint(SDL_HINT_OPENGL_ES_DRIVER, "0");
141 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, m_OldContextProfileMask);
142 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, m_OldContextMajorVersion);
143 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, m_OldContextMinorVersion);
144}

186void EGLRenderer::renderOverlay(Overlay::OverlayType type, int viewportWidth, int viewportHeight)
187{
188 // Do nothing if this overlay is disabled
189 if (!Session::get()->getOverlayManager().isOverlayEnabled(type)) {
190 return;
191 }
192
193 // Upload a new overlay texture if needed
194 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
195 if (newSurface != nullptr) {
196 SDL_assert(!SDL_MUSTLOCK(newSurface));
197 SDL_assert(newSurface->format->format == SDL_PIXELFORMAT_ARGB8888);
198
199 glBindTexture(GL_TEXTURE_2D, m_OverlayTextures[type]);
200
201 void* packedPixelData = nullptr;
202 if (m_GlesMajorVersion >= 3 || m_HasExtUnpackSubimage) {
203 // If we are GLES 3.0+ or have GL_EXT_unpack_subimage, GL can handle any pitch
204 SDL_assert(newSurface->pitch % newSurface->format->BytesPerPixel == 0);
205 glPixelStorei(GL_UNPACK_ROW_LENGTH_EXT, newSurface->pitch / newSurface->format->BytesPerPixel);
206 }
207 else if (newSurface->pitch != newSurface->w * newSurface->format->BytesPerPixel) {
208 // If we can't use GL_UNPACK_ROW_LENGTH and the surface isn't tightly packed,
209 // we must allocate a tightly packed buffer and copy our pixels there.
210 packedPixelData = malloc(newSurface->w * newSurface->h * newSurface->format->BytesPerPixel);
211 if (!packedPixelData) {
212 SDL_FreeSurface(newSurface);
213 return;
214 }
215
216 SDL_ConvertPixels(newSurface->w, newSurface->h,
217 newSurface->format->format, newSurface->pixels, newSurface->pitch,
218 newSurface->format->format, packedPixelData, newSurface->w * newSurface->format->BytesPerPixel);
219 }
220
221 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, newSurface->w, newSurface->h, 0, GL_RGBA, GL_UNSIGNED_BYTE,
222 packedPixelData ? packedPixelData : newSurface->pixels);
223
224 if (packedPixelData) {
225 free(packedPixelData);
226 }
227
228 SDL_FRect overlayRect;
229
230 // These overlay positions differ from the other renderers because OpenGL
231 // places the origin in the lower-left corner instead of the upper-left.
232 if (type == Overlay::OverlayStatusUpdate) {
233 // Bottom Left
234 overlayRect.x = 0;
235 overlayRect.y = 0;
236 }
237 else if (type == Overlay::OverlayDebug) {
238 // Top left
239 overlayRect.x = 0;
240 overlayRect.y = viewportHeight - newSurface->h;
241 } else {
242 SDL_assert(false);
243 }
244
245 overlayRect.w = newSurface->w;
246 overlayRect.h = newSurface->h;
247
248 SDL_FreeSurface(newSurface);
249
250 // Convert screen space to normalized device coordinates
251 StreamUtils::screenSpaceToNormalizedDeviceCoords(&overlayRect, viewportWidth, viewportHeight);
252
253 OVERLAY_VERTEX verts[] =
254 {
255 {overlayRect.x + overlayRect.w, overlayRect.y + overlayRect.h, 1.0f, 0.0f},
256 {overlayRect.x, overlayRect.y + overlayRect.h, 0.0f, 0.0f},
257 {overlayRect.x, overlayRect.y, 0.0f, 1.0f},
258 {overlayRect.x, overlayRect.y, 0.0f, 1.0f},
259 {overlayRect.x + overlayRect.w, overlayRect.y, 1.0f, 1.0f},
260 {overlayRect.x + overlayRect.w, overlayRect.y + overlayRect.h, 1.0f, 0.0f}
261 };
262
263 glBindBuffer(GL_ARRAY_BUFFER, m_OverlayVbos[type]);
264 glBufferData(GL_ARRAY_BUFFER, sizeof(verts), verts, GL_STATIC_DRAW);
265
266 SDL_AtomicSet(&m_OverlayHasValidData[type], 1);
267 }
268
269 if (!SDL_AtomicGet(&m_OverlayHasValidData[type])) {
270 // If the overlay is not populated yet or is stale, don't render it.
271 return;
272 }
273
274 // Adjust the viewport to the whole window before rendering the overlays
275 glViewport(0, 0, viewportWidth, viewportHeight);
276
277 glUseProgram(m_OverlayShaderProgram);
278
279 glBindBuffer(GL_ARRAY_BUFFER, m_OverlayVbos[type]);
280 glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(OVERLAY_VERTEX), (void*)offsetof(OVERLAY_VERTEX, x));
281 glEnableVertexAttribArray(0);
282 glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, sizeof(OVERLAY_VERTEX), (void*)offsetof(OVERLAY_VERTEX, u));
283 glEnableVertexAttribArray(1);
284
285 glActiveTexture(GL_TEXTURE0);
286 glBindTexture(GL_TEXTURE_2D, m_OverlayTextures[type]);
287
288 glDrawArrays(GL_TRIANGLES, 0, 6);
289}

291int EGLRenderer::loadAndBuildShader(int shaderType,
292 const char *file) {
293 GLuint shader = glCreateShader(shaderType);
294 if (!shader || shader == GL_INVALID_ENUM) {
295 EGL_LOG(Error, "Can't create shader: %d", glGetError());
296 return 0;
297 }
298
299 auto sourceData = Path::readDataFile(file);
300 GLint len = sourceData.size();
301 const char *buf = sourceData.data();
302
303 glShaderSource(shader, 1, &buf, &len);
304 glCompileShader(shader);
305 GLint status;
306 glGetShaderiv(shader, GL_COMPILE_STATUS, &status);
307 if (!status) {
308 char shaderLog[512];
309 glGetShaderInfoLog(shader, sizeof (shaderLog), nullptr, shaderLog);
310 EGL_LOG(Error, "Cannot load shader \"%s\": %s", file, shaderLog);
311 return 0;
312 }
313
314 return shader;
315}

317unsigned EGLRenderer::compileShader(const char* vertexShaderSrc, const char* fragmentShaderSrc) {
318 unsigned shader = 0;
319
320 GLuint vertexShader = loadAndBuildShader(GL_VERTEX_SHADER, vertexShaderSrc);
321 if (!vertexShader)
322 return false;
323
324 GLuint fragmentShader = loadAndBuildShader(GL_FRAGMENT_SHADER, fragmentShaderSrc);
325 if (!fragmentShader)
326 goto fragError;
327
328 shader = glCreateProgram();
329 if (!shader) {
330 EGL_LOG(Error, "Cannot create shader program");
331 goto progFailCreate;
332 }
333
334 glAttachShader(shader, vertexShader);
335 glAttachShader(shader, fragmentShader);
336 glLinkProgram(shader);
337 int status;
338 glGetProgramiv(shader, GL_LINK_STATUS, &status);
339 if (!status) {
340 char shader_log[512];
341 glGetProgramInfoLog(shader, sizeof (shader_log), nullptr, shader_log);
342 EGL_LOG(Error, "Cannot link shader program: %s", shader_log);
343 glDeleteProgram(shader);
344 shader = 0;
345 }
346
347progFailCreate:
348 glDeleteShader(fragmentShader);
349fragError:
350 glDeleteShader(vertexShader);
351 return shader;
352}

354bool EGLRenderer::compileShaders() {
355 SDL_assert(!m_ShaderProgram);
356 SDL_assert(!m_OverlayShaderProgram);
357
358 SDL_assert(m_EGLImagePixelFormat != AV_PIX_FMT_NONE);
359
360 // XXX: TODO: other formats
361 if (m_EGLImagePixelFormat == AV_PIX_FMT_NV12 || m_EGLImagePixelFormat == AV_PIX_FMT_P010) {
362 m_ShaderProgram = compileShader("egl_nv12.vert", "egl_nv12.frag");
363 if (!m_ShaderProgram) {
364 return false;
365 }
366
367 m_ShaderProgramParams[NV12_PARAM_YUVMAT] = glGetUniformLocation(m_ShaderProgram, "yuvmat");
368 m_ShaderProgramParams[NV12_PARAM_OFFSET] = glGetUniformLocation(m_ShaderProgram, "offset");
369 m_ShaderProgramParams[NV12_PARAM_CHROMA_OFFSET] = glGetUniformLocation(m_ShaderProgram, "chromaOffset");
370 m_ShaderProgramParams[NV12_PARAM_PLANE1] = glGetUniformLocation(m_ShaderProgram, "plane1");
371 m_ShaderProgramParams[NV12_PARAM_PLANE2] = glGetUniformLocation(m_ShaderProgram, "plane2");
372
373 // Set up constant uniforms
374 glUseProgram(m_ShaderProgram);
375 glUniform1i(m_ShaderProgramParams[NV12_PARAM_PLANE1], 0);
376 glUniform1i(m_ShaderProgramParams[NV12_PARAM_PLANE2], 1);
377 glUseProgram(0);
378 }
379 else if (m_EGLImagePixelFormat == AV_PIX_FMT_DRM_PRIME) {
380 m_ShaderProgram = compileShader("egl_opaque.vert", "egl_opaque.frag");
381 if (!m_ShaderProgram) {
382 return false;
383 }
384
385 m_ShaderProgramParams[OPAQUE_PARAM_TEXTURE] = glGetUniformLocation(m_ShaderProgram, "uTexture");
386
387 // Set up constant uniforms
388 glUseProgram(m_ShaderProgram);
389 glUniform1i(m_ShaderProgramParams[OPAQUE_PARAM_TEXTURE], 0);
390 glUseProgram(0);
391 }
392 else {
393 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
394 "Unsupported EGL pixel format: %d",
395 m_EGLImagePixelFormat);
396 SDL_assert(false);
397 return false;
398 }
399
400 m_OverlayShaderProgram = compileShader("egl_overlay.vert", "egl_overlay.frag");
401 if (!m_OverlayShaderProgram) {
402 return false;
403 }
404
405 m_OverlayShaderProgramParams[OVERLAY_PARAM_TEXTURE] = glGetUniformLocation(m_OverlayShaderProgram, "uTexture");
406
407 glUseProgram(m_OverlayShaderProgram);
408 glUniform1i(m_OverlayShaderProgramParams[OVERLAY_PARAM_TEXTURE], 0);
409 glUseProgram(0);
410
411 return true;
412}

414bool EGLRenderer::initialize(PDECODER_PARAMETERS params)
415{
416 m_Window = params->window;
417
418 // It's not safe to attempt to opportunistically create a GLES2
419 // renderer prior to 2.0.10. If GLES2 isn't available, SDL will
420 // attempt to dereference a null pointer and crash Moonlight.
421 // https://bugzilla.libsdl.org/show_bug.cgi?id=4350
422 // https://hg.libsdl.org/SDL/rev/84618d571795
423 if (!SDL_VERSION_ATLEAST(2, 0, 10)) {
424 EGL_LOG(Error, "Not supported until SDL 2.0.10");
425 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
426 return false;
427 }
428
429 // This renderer doesn't support HDR, so pick a different one.
430 // HACK: This avoids a deadlock in SDL_CreateRenderer() if
431 // Vulkan was used before and SDL is trying to load EGL.
432 if (params->videoFormat & VIDEO_FORMAT_MASK_10BIT) {
433 EGL_LOG(Info, "EGL doesn't support HDR rendering");
434 return false;
435 }
436
437 // This hint will ensure we use EGL to retrieve our GL context,
438 // even on X11 where that is not the default. EGL is required
439 // to avoid a crash in Mesa.
440 // https://gitlab.freedesktop.org/mesa/mesa/issues/1011
441 SDL_SetHint(SDL_HINT_OPENGL_ES_DRIVER, "1");
442 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_ES);
443 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
444 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 0);
445
446 int renderIndex;
447 int maxRenderers = SDL_GetNumRenderDrivers();
448 SDL_assert(maxRenderers >= 0);
449
450 SDL_RendererInfo renderInfo;
451 for (renderIndex = 0; renderIndex < maxRenderers; ++renderIndex) {
452 if (SDL_GetRenderDriverInfo(renderIndex, &renderInfo))
453 continue;
454 if (!strcmp(renderInfo.name, "opengles2")) {
455 SDL_assert(renderInfo.flags & SDL_RENDERER_ACCELERATED);
456 break;
457 }
458 }
459 if (renderIndex == maxRenderers) {
460 EGL_LOG(Error, "Could not find a suitable SDL_Renderer");
461 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
462 return false;
463 }
464
465 m_DummyRenderer = SDL_CreateRenderer(m_Window, renderIndex, SDL_RENDERER_ACCELERATED);
466 if (!m_DummyRenderer) {
467 // Print the error here (before it gets clobbered), but ensure that we flush window
468 // events just in case SDL re-created the window before eventually failing.
469 EGL_LOG(Error, "SDL_CreateRenderer() failed: %s", SDL_GetError());
470 }
471
472 // SDL_CreateRenderer() can end up having to recreate our window (SDL_RecreateWindow())
473 // to ensure it's compatible with the renderer's OpenGL context. If that happens, we
474 // can get spurious SDL_WINDOWEVENT events that will cause us to (again) recreate our
475 // renderer. This can lead to an infinite to renderer recreation, so discard all
476 // SDL_WINDOWEVENT events after SDL_CreateRenderer().
477 Session* session = Session::get();
478 if (session != nullptr) {
479 // If we get here during a session, we need to synchronize with the event loop
480 // to ensure we don't drop any important events.
481 session->flushWindowEvents();
482 }
483 else {
484 // If we get here prior to the start of a session, just pump and flush ourselves.
485 SDL_PumpEvents();
486 SDL_FlushEvent(SDL_WINDOWEVENT);
487 }
488
489 // Now we finally bail if we failed during SDL_CreateRenderer() above.
490 if (!m_DummyRenderer) {
491 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
492 return false;
493 }
494
495 SDL_SysWMinfo info;
496 SDL_VERSION(&info.version);
497 if (!SDL_GetWindowWMInfo(params->window, &info)) {
498 EGL_LOG(Error, "SDL_GetWindowWMInfo() failed: %s", SDL_GetError());
499 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
500 return false;
501 }
502
503 if (!(m_Context = SDL_GL_CreateContext(params->window))) {
504 EGL_LOG(Error, "Cannot create OpenGL context: %s", SDL_GetError());
505 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
506 return false;
507 }
508 if (SDL_GL_MakeCurrent(params->window, m_Context)) {
509 EGL_LOG(Error, "Cannot use created EGL context: %s", SDL_GetError());
510 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
511 return false;
512 }
513
514 {
515 int r, g, b, a;
516 SDL_GL_GetAttribute(SDL_GL_RED_SIZE, &r);
517 SDL_GL_GetAttribute(SDL_GL_GREEN_SIZE, &g);
518 SDL_GL_GetAttribute(SDL_GL_BLUE_SIZE, &b);
519 SDL_GL_GetAttribute(SDL_GL_ALPHA_SIZE, &a);
520 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
521 "Color buffer is: R%dG%dB%dA%d",
522 r, g, b, a);
523 }
524
525 SDL_GL_GetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, &m_GlesMajorVersion);
526 SDL_GL_GetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, &m_GlesMinorVersion);
527
528 // We can use GL_UNPACK_ROW_LENGTH for a more optimized upload of non-tightly-packed textures
529 m_HasExtUnpackSubimage = SDL_GL_ExtensionSupported("GL_EXT_unpack_subimage");
530
531 m_EGLDisplay = eglGetCurrentDisplay();
532 if (m_EGLDisplay == EGL_NO_DISPLAY) {
533 EGL_LOG(Error, "Cannot get EGL display: %d", eglGetError());
534 return false;
535 }
536
537 const EGLExtensions eglExtensions(m_EGLDisplay);
538 if (!eglExtensions.isSupported("EGL_KHR_image_base") &&
539 !eglExtensions.isSupported("EGL_KHR_image")) {
540 EGL_LOG(Error, "EGL_KHR_image unsupported");
541 return false;
542 }
543 else if (!SDL_GL_ExtensionSupported("GL_OES_EGL_image")) {
544 EGL_LOG(Error, "GL_OES_EGL_image unsupported");
545 return false;
546 }
547
548 if (!m_Backend->initializeEGL(m_EGLDisplay, eglExtensions))
549 return false;
550
551 if (!(m_glEGLImageTargetTexture2DOES = (typeof(m_glEGLImageTargetTexture2DOES))eglGetProcAddress("glEGLImageTargetTexture2DOES"))) {
552 EGL_LOG(Error,
553 "EGL: cannot retrieve `glEGLImageTargetTexture2DOES` address");
554 return false;
555 }
556
557 // Vertex arrays are an extension on OpenGL ES 2.0
558 if (SDL_GL_ExtensionSupported("GL_OES_vertex_array_object")) {
559 m_glGenVertexArraysOES = (typeof(m_glGenVertexArraysOES))eglGetProcAddress("glGenVertexArraysOES");
560 m_glBindVertexArrayOES = (typeof(m_glBindVertexArrayOES))eglGetProcAddress("glBindVertexArrayOES");
561 m_glDeleteVertexArraysOES = (typeof(m_glDeleteVertexArraysOES))eglGetProcAddress("glDeleteVertexArraysOES");
562 }
563 else {
564 // They are included in OpenGL ES 3.0 as part of the standard
565 m_glGenVertexArraysOES = (typeof(m_glGenVertexArraysOES))eglGetProcAddress("glGenVertexArrays");
566 m_glBindVertexArrayOES = (typeof(m_glBindVertexArrayOES))eglGetProcAddress("glBindVertexArray");
567 m_glDeleteVertexArraysOES = (typeof(m_glDeleteVertexArraysOES))eglGetProcAddress("glDeleteVertexArrays");
568 }
569
570 if (!m_glGenVertexArraysOES || !m_glBindVertexArrayOES || !m_glDeleteVertexArraysOES) {
571 EGL_LOG(Error, "Failed to find VAO functions");
572 return false;
573 }
574
575 // EGL_KHR_fence_sync is an extension for EGL 1.1+
576 if (eglExtensions.isSupported("EGL_KHR_fence_sync")) {
577 // eglCreateSyncKHR() has a slightly different prototype to eglCreateSync()
578 m_eglCreateSyncKHR = (typeof(m_eglCreateSyncKHR))eglGetProcAddress("eglCreateSyncKHR");
579 m_eglDestroySync = (typeof(m_eglDestroySync))eglGetProcAddress("eglDestroySyncKHR");
580 m_eglClientWaitSync = (typeof(m_eglClientWaitSync))eglGetProcAddress("eglClientWaitSyncKHR");
581 }
582 else {
583 // EGL 1.5 introduced sync support to the core specification
584 m_eglCreateSync = (typeof(m_eglCreateSync))eglGetProcAddress("eglCreateSync");
585 m_eglDestroySync = (typeof(m_eglDestroySync))eglGetProcAddress("eglDestroySync");
586 m_eglClientWaitSync = (typeof(m_eglClientWaitSync))eglGetProcAddress("eglClientWaitSync");
587 }
588
589 if (!(m_eglCreateSync || m_eglCreateSyncKHR) || !m_eglDestroySync || !m_eglClientWaitSync) {
590 EGL_LOG(Warn, "Failed to find sync functions");
591
592 // Sub-optimal, but not fatal
593 m_eglCreateSync = nullptr;
594 m_eglCreateSyncKHR = nullptr;
595 m_eglDestroySync = nullptr;
596 m_eglClientWaitSync = nullptr;
597 }
598
599 // SDL always uses swap interval 0 under the hood on Wayland systems,
600 // because the compositor guarantees tear-free rendering. In this
601 // situation, swap interval > 0 behaves as a frame pacing option
602 // rather than a way to eliminate tearing as SDL will block in
603 // SwapBuffers until the compositor consumes the frame. This will
604 // needlessly increases latency, so we should avoid it.
605 //
606 // HACK: In SDL 2.0.22+ on GNOME systems with fractional DPI scaling,
607 // the Wayland viewport can be stale when using Super+Left/Right/Up
608 // to resize the window. This seems to happen significantly more often
609 // with vsync enabled, so this also mitigates that problem too.
610 if (params->enableVsync
611#ifdef SDL_VIDEO_DRIVER_WAYLAND
612 && info.subsystem != SDL_SYSWM_WAYLAND
613#endif
614 ) {
615 SDL_GL_SetSwapInterval(1);
616
617#if SDL_VERSION_ATLEAST(2, 0, 15) && defined(SDL_VIDEO_DRIVER_KMSDRM)
618 // The SDL KMSDRM backend already enforces double buffering (due to
619 // SDL_HINT_VIDEO_DOUBLE_BUFFER=1), so calling glFinish() after
620 // SDL_GL_SwapWindow() will block an extra frame and lock rendering
621 // at 1/2 the display refresh rate.
622 if (info.subsystem != SDL_SYSWM_KMSDRM)
623#endif
624 {
625 m_BlockingSwapBuffers = true;
626 }
627 } else {
628 SDL_GL_SetSwapInterval(0);
629 }
630
631 glGenTextures(EGL_MAX_PLANES, m_Textures);
632 for (size_t i = 0; i < EGL_MAX_PLANES; ++i) {
633 glBindTexture(GL_TEXTURE_EXTERNAL_OES, m_Textures[i]);
634 glTexParameteri(GL_TEXTURE_EXTERNAL_OES, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
635 glTexParameteri(GL_TEXTURE_EXTERNAL_OES, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
636 glTexParameteri(GL_TEXTURE_EXTERNAL_OES, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
637 glTexParameteri(GL_TEXTURE_EXTERNAL_OES, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
638 }
639
640 glGenBuffers(Overlay::OverlayMax, m_OverlayVbos);
641 glGenTextures(Overlay::OverlayMax, m_OverlayTextures);
642 for (size_t i = 0; i < Overlay::OverlayMax; ++i) {
643 glBindTexture(GL_TEXTURE_2D, m_OverlayTextures[i]);
644 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
645 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
646 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
647 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
648 }
649
650 glEnable(GL_BLEND);
651 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
652
653 GLenum err = glGetError();
654 if (err != GL_NO_ERROR)
655 EGL_LOG(Error, "OpenGL error: %d", err);
656
657 // Detach the context from this thread, so the render thread can attach it
658 SDL_GL_MakeCurrent(m_Window, nullptr);
659
660 if (err == GL_NO_ERROR) {
661 // If we got a working GL implementation via EGL, avoid using GLX from now on.
662 // GLX will cause problems if we later want to use EGL again on this window.
663 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION, "EGL passed preflight checks. Using EGL for GL context creation.");
664 SDL_SetHint(SDL_HINT_VIDEO_X11_FORCE_EGL, "1");
665 }
666
667 return err == GL_NO_ERROR;
668}

670bool EGLRenderer::specialize() {
671 SDL_assert(!m_VAO);
672
673 if (!compileShaders())
674 return false;
675
676 // The viewport should have the aspect ratio of the video stream
677 static const float vertices[] = {
678 // pos .... // tex coords
679 1.0f, 1.0f, 1.0f, 0.0f,
680 1.0f, -1.0f, 1.0f, 1.0f,
681 -1.0f, -1.0f, 0.0f, 1.0f,
682 -1.0f, 1.0f, 0.0f, 0.0f,
683
684 };
685 static const unsigned int indices[] = {
686 0, 1, 3,
687 1, 2, 3,
688 };
689
690 glUseProgram(m_ShaderProgram);
691
692 unsigned int VBO, EBO;
693 m_glGenVertexArraysOES(1, &m_VAO);
694 glGenBuffers(1, &VBO);
695 glGenBuffers(1, &EBO);
696
697 m_glBindVertexArrayOES(m_VAO);
698
699 glBindBuffer(GL_ARRAY_BUFFER, VBO);
700 glBufferData(GL_ARRAY_BUFFER, sizeof (vertices), vertices, GL_STATIC_DRAW);
701
702 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
703 glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof (indices), indices, GL_STATIC_DRAW);
704
705 glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, 4 * sizeof(float), (void*)0);
706 glEnableVertexAttribArray(0);
707 glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 4 * sizeof(float), (void*)(2 * sizeof (float)));
708 glEnableVertexAttribArray(1);
709
710 glBindBuffer(GL_ARRAY_BUFFER, 0);
711 m_glBindVertexArrayOES(0);
712
713 glDeleteBuffers(1, &VBO);
714 glDeleteBuffers(1, &EBO);
715
716 GLenum err = glGetError();
717 if (err != GL_NO_ERROR) {
718 EGL_LOG(Error, "OpenGL error: %d", err);
719 }
720
721 return err == GL_NO_ERROR;
722}

730void EGLRenderer::waitToRender()
731{
732 // Ensure our GL context is active on this thread
733 // See comment in renderFrame() for more details.
734 SDL_GL_MakeCurrent(m_Window, m_Context);
735
736 // Wait for the previous buffer swap to finish before picking the next frame to render.
737 // This way we'll get the latest available frame and render it without blocking.
738 if (m_BlockingSwapBuffers) {
739 // Try to use eglClientWaitSync() if the driver supports it
740 if (m_LastRenderSync != EGL_NO_SYNC) {
741 SDL_assert(m_eglClientWaitSync != nullptr);
742 m_eglClientWaitSync(m_EGLDisplay, m_LastRenderSync, EGL_SYNC_FLUSH_COMMANDS_BIT, EGL_FOREVER);
743 }
744 else {
745 // Use glFinish() if fences aren't available
746 glFinish();
747 }
748 }
749}

763void EGLRenderer::renderFrame(AVFrame* frame)
764{
765 EGLImage imgs[EGL_MAX_PLANES];
766
767 // Attach our GL context to the render thread
768 // NB: It should already be current, unless the SDL render event watcher
769 // performs a rendering operation (like a viewport update on resize) on
770 // our fake SDL_Renderer. If it's already current, this is a no-op.
771 SDL_GL_MakeCurrent(m_Window, m_Context);
772
773 // Find the native read-back format and load the shaders
774 if (m_EGLImagePixelFormat == AV_PIX_FMT_NONE) {
775 m_EGLImagePixelFormat = m_Backend->getEGLImagePixelFormat();
776 EGL_LOG(Info, "EGLImage pixel format: %d", m_EGLImagePixelFormat);
777
778 SDL_assert(m_EGLImagePixelFormat != AV_PIX_FMT_NONE);
779
780 if (!specialize()) {
781 m_EGLImagePixelFormat = AV_PIX_FMT_NONE;
782
783 // Failure to specialize is fatal. We must reset the renderer
784 // to recover successfully.
785 //
786 // Note: This seems to be easy to trigger when transitioning from
787 // maximized mode by dragging the window down on GNOME 42 using
788 // XWayland. Other strategies like calling glGetError() don't seem
789 // to be able to detect this situation for some reason.
790 SDL_Event event;
791 event.type = SDL_RENDER_TARGETS_RESET;
792 SDL_PushEvent(&event);
793
794 return;
795 }
796 }
797
798 ssize_t plane_count = m_Backend->exportEGLImages(frame, m_EGLDisplay, imgs);
799 if (plane_count < 0)
800 return;
801 for (ssize_t i = 0; i < plane_count; ++i) {
802 glActiveTexture(GL_TEXTURE0 + i);
803 glBindTexture(GL_TEXTURE_EXTERNAL_OES, m_Textures[i]);
804 m_glEGLImageTargetTexture2DOES(GL_TEXTURE_EXTERNAL_OES, imgs[i]);
805 }
806
807 glClear(GL_COLOR_BUFFER_BIT);
808
809 int drawableWidth, drawableHeight;
810 SDL_GL_GetDrawableSize(m_Window, &drawableWidth, &drawableHeight);
811
812 // Set the viewport to the size of the aspect-ratio-scaled video
813 SDL_Rect src, dst;
814 src.x = src.y = dst.x = dst.y = 0;
815 src.w = frame->width;
816 src.h = frame->height;
817 dst.w = drawableWidth;
818 dst.h = drawableHeight;
819 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
820 glViewport(dst.x, dst.y, dst.w, dst.h);
821
822 glUseProgram(m_ShaderProgram);
823 m_glBindVertexArrayOES(m_VAO);
824
825 // If the frame format has changed, we'll need to recompute the constants
826 if (hasFrameFormatChanged(frame) && (m_EGLImagePixelFormat == AV_PIX_FMT_NV12 || m_EGLImagePixelFormat == AV_PIX_FMT_P010)) {
827 std::array<float, 9> colorMatrix;
828 std::array<float, 3> yuvOffsets;
829 std::array<float, 2> chromaOffset;
830
831 getFramePremultipliedCscConstants(frame, colorMatrix, yuvOffsets);
832 getFrameChromaCositingOffsets(frame, chromaOffset);
833 chromaOffset[0] /= frame->width;
834 chromaOffset[1] /= frame->height;
835
836 glUniformMatrix3fv(m_ShaderProgramParams[NV12_PARAM_YUVMAT], 1, GL_FALSE, colorMatrix.data());
837 glUniform3fv(m_ShaderProgramParams[NV12_PARAM_OFFSET], 1, yuvOffsets.data());
838 glUniform2fv(m_ShaderProgramParams[NV12_PARAM_CHROMA_OFFSET], 1, chromaOffset.data());
839 }
840
841 glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_INT, 0);
842
843 m_glBindVertexArrayOES(0);
844
845 for (int i = 0; i < Overlay::OverlayMax; i++) {
846 renderOverlay((Overlay::OverlayType)i, drawableWidth, drawableHeight);
847 }
848
849 SDL_GL_SwapWindow(m_Window);
850
851 if (m_BlockingSwapBuffers) {
852 // This glClear() requires the new back buffer to complete. This ensures
853 // our eglClientWaitSync() or glFinish() call in waitToRender() will not
854 // return before the new buffer is actually ready for rendering.
855 glClear(GL_COLOR_BUFFER_BIT);
856
857 // If we this EGL implementation supports fences, use those to delay
858 // rendering the next frame until this one is completed. If not, we'll
859 // have to just use glFinish().
860 if (m_eglClientWaitSync != nullptr) {
861 // Delete the sync object from last render
862 if (m_LastRenderSync != EGL_NO_SYNC) {
863 m_eglDestroySync(m_EGLDisplay, m_LastRenderSync);
864 }
865
866 // Create a new sync object that will be signalled when the buffer swap is completed
867 if (m_eglCreateSync != nullptr) {
868 m_LastRenderSync = m_eglCreateSync(m_EGLDisplay, EGL_SYNC_FENCE, nullptr);
869 }
870 else {
871 SDL_assert(m_eglCreateSyncKHR != nullptr);
872 m_LastRenderSync = m_eglCreateSyncKHR(m_EGLDisplay, EGL_SYNC_FENCE, nullptr);
873 }
874 }
875 }
876
877 m_Backend->freeEGLImages(m_EGLDisplay, imgs);
878
879 // Free the DMA-BUF backing the last frame now that it is definitely
880 // no longer being used anymore. While the PRIME FD stays around until
881 // EGL is done with it, the memory backing it may be reused by FFmpeg
882 // before the GPU has read it. This is particularly noticeable on the
883 // RK3288-based TinkerBoard when V-Sync is disabled.
884 av_frame_unref(m_LastFrame);
885 av_frame_move_ref(m_LastFrame, frame);
886}

903 return true;
