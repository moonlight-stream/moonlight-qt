148void AppModel::updateAppList(QVector<NvApp> newList)
149{
150 m_AllApps = newList;
151
152 QVector<NvApp> newVisibleList = getVisibleApps(newList);
153
154 // Process removals and updates first
155 for (int i = 0; i < m_VisibleApps.count(); i++) {
156 const NvApp& existingApp = m_VisibleApps.at(i);
157
158 bool found = false;
159 for (const NvApp& newApp : newVisibleList) {
160 if (existingApp.id == newApp.id) {
161 // If the data changed, update it in our list
162 if (existingApp != newApp) {
163 m_VisibleApps.replace(i, newApp);
164 emit dataChanged(createIndex(i, 0), createIndex(i, 0));
165 }
166
167 found = true;
168 break;
169 }
170 }
171
172 if (!found) {
173 beginRemoveRows(QModelIndex(), i, i);
174 m_VisibleApps.removeAt(i);
175 endRemoveRows();
176 i--;
177 }
178 }
179
180 // Process additions now
181 for (const NvApp& newApp : newVisibleList) {
182 int insertionIndex = m_VisibleApps.size();
183 bool found = false;
184
185 for (int i = 0; i < m_VisibleApps.count(); i++) {
186 const NvApp& existingApp = m_VisibleApps.at(i);
187
188 if (existingApp.id == newApp.id) {
189 found = true;
190 break;
191 }
192 else if (existingApp.name.toLower() > newApp.name.toLower()) {
193 insertionIndex = i;
194 break;
195 }
196 }
197
198 if (!found) {
199 beginInsertRows(QModelIndex(), insertionIndex, insertionIndex);
200 m_VisibleApps.insert(insertionIndex, newApp);
201 endInsertRows();
202 }
203 }
204
205 Q_ASSERT(newVisibleList == m_VisibleApps);
206}

253void AppModel::handleComputerStateChanged(NvComputer* computer)
254{
255 // Ignore updates for computers that aren't ours
256 if (computer != m_Computer) {
257 return;
258 }
259
260 // If the computer has gone offline or we've been unpaired,
261 // signal the UI so we can go back to the PC view.
262 if (m_Computer->state == NvComputer::CS_OFFLINE ||
263 m_Computer->pairState == NvComputer::PS_NOT_PAIRED) {
264 emit computerLost();
265 return;
266 }
267
268 // First, process additions/removals from the app list. This
269 // is required because the new game may now be running, so
270 // we can't check that first.
271 if (computer->appList != m_AllApps) {
272 updateAppList(computer->appList);
273 }
274
275 // Finally, process changes to the active app
276 if (computer->currentGameId != m_CurrentGameId) {
277 // First, invalidate the running state of newly running game
278 for (int i = 0; i < m_VisibleApps.count(); i++) {
279 if (m_VisibleApps[i].id == computer->currentGameId) {
280 emit dataChanged(createIndex(i, 0),
281 createIndex(i, 0),
282 QVector<int>() << RunningRole);
283 break;
284 }
285 }
286
287 // Next, invalidate the running state of the old game (if it exists)
288 if (m_CurrentGameId != 0) {
289 for (int i = 0; i < m_VisibleApps.count(); i++) {
290 if (m_VisibleApps[i].id == m_CurrentGameId) {
291 emit dataChanged(createIndex(i, 0),
292 createIndex(i, 0),
293 QVector<int>() << RunningRole);
294 break;
295 }
296 }
297 }
298
299 // Now update our internal state
300 m_CurrentGameId = m_Computer->currentGameId;
301 }
302}
