75void DxVsyncSource::waitForVsync()
76{
77 NTSTATUS status;
78
79 // If the monitor has changed from last time, open the new adapter
80 HMONITOR currentMonitor = MonitorFromWindow(m_Window, MONITOR_DEFAULTTONEAREST);
81 if (currentMonitor != m_LastMonitor) {
82 MONITORINFOEXA monitorInfo = {};
83 monitorInfo.cbSize = sizeof(monitorInfo);
84 if (!GetMonitorInfoA(currentMonitor, &monitorInfo)) {
85 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
86 "GetMonitorInfo() failed: %d",
87 GetLastError());
88 return;
89 }
90
91 DEVMODEA monitorMode;
92 monitorMode.dmSize = sizeof(monitorMode);
93 if (!EnumDisplaySettingsA(monitorInfo.szDevice, ENUM_CURRENT_SETTINGS, &monitorMode)) {
94 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
95 "EnumDisplaySettings() failed: %d",
96 GetLastError());
97 return;
98 }
99
100 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
101 "Monitor changed: %s %d Hz",
102 monitorInfo.szDevice,
103 monitorMode.dmDisplayFrequency);
104
105 // Close the old adapter
106 if (m_WaitForVblankEventParams.hAdapter != 0) {
107 D3DKMT_CLOSEADAPTER closeAdapterParams = {};
108 closeAdapterParams.hAdapter = m_WaitForVblankEventParams.hAdapter;
109 m_D3DKMTCloseAdapter(&closeAdapterParams);
110 }
111
112 D3DKMT_OPENADAPTERFROMHDC openAdapterParams = {};
113 openAdapterParams.hDc = CreateDCA(nullptr, monitorInfo.szDevice, nullptr, nullptr);
114 if (!openAdapterParams.hDc) {
115 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
116 "CreateDC() failed: %d",
117 GetLastError());
118 return;
119 }
120
121 // Open the new adapter
122 status = m_D3DKMTOpenAdapterFromHdc(&openAdapterParams);
123 DeleteDC(openAdapterParams.hDc);
124
125 if (status != STATUS_SUCCESS) {
126 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
127 "D3DKMTOpenAdapterFromHdc() failed: %x",
128 status);
129 return;
130 }
131
132 m_WaitForVblankEventParams.hAdapter = openAdapterParams.hAdapter;
133 m_WaitForVblankEventParams.hDevice = 0;
134 m_WaitForVblankEventParams.VidPnSourceId = openAdapterParams.VidPnSourceId;
135
136 m_LastMonitor = currentMonitor;
137 }
138
139 status = m_D3DKMTWaitForVerticalBlankEvent(&m_WaitForVblankEventParams);
140 if (status != STATUS_SUCCESS) {
141 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
142 "D3DKMTWaitForVerticalBlankEvent() failed: %x",
143 status);
144 return;
145 }
146}
