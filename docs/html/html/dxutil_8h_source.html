16 static bool isFormatHybridDecodedByHardware(int videoFormat, unsigned int vendorId, unsigned int deviceId) {
17 if (vendorId == 0x8086) {
18 // Intel seems to encode the series in the high byte of
19 // the device ID. We want to avoid the "Partial" acceleration
20 // support explicitly. Those will claim to have HW acceleration
21 // but perform badly.
22 // https://en.wikipedia.org/wiki/Intel_Graphics_Technology#Capabilities_(GPU_video_acceleration)
23 // https://raw.githubusercontent.com/GameTechDev/gpudetect/master/IntelGfx.cfg
24 switch (deviceId & 0xFF00) {
25 case 0x0400: // Haswell
26 case 0x0A00: // Haswell
27 case 0x0D00: // Haswell
28 case 0x1600: // Broadwell
29 case 0x2200: // Cherry Trail and Braswell
30 // Block these for HEVC to avoid hybrid decode
31 return (videoFormat & VIDEO_FORMAT_MASK_H265) != 0;
32 case 0x1900: // Skylake
33 // Blacklist these for HEVC Main10 to avoid hybrid decode.
34 // Regular HEVC Main is fine though.
35 if (videoFormat == VIDEO_FORMAT_H265_MAIN10) {
36 return true;
37 }
38 default:
39 break;
40 }
41 }
42 else if (vendorId == 0x10DE) {
43 // For NVIDIA, we wait to avoid those GPUs with Feature Set E
44 // for HEVC decoding, since that's hybrid. It appears that Kepler GPUs
45 // also had some hybrid decode support (per DXVA2 Checker) so we'll
46 // blacklist those too.
47 // https://en.wikipedia.org/wiki/Nvidia_PureVideo
48 // https://bluesky23.yukishigure.com/en/dxvac/deviceInfo/decoder.html
49 // http://envytools.readthedocs.io/en/latest/hw/pciid.html (missing GM200)
50 if ((deviceId >= 0x1180 && deviceId <= 0x11BF) || // GK104
51 (deviceId >= 0x11C0 && deviceId <= 0x11FF) || // GK106
52 (deviceId >= 0x0FC0 && deviceId <= 0x0FFF) || // GK107
53 (deviceId >= 0x1000 && deviceId <= 0x103F) || // GK110/GK110B
54 (deviceId >= 0x1280 && deviceId <= 0x12BF) || // GK208
55 (deviceId >= 0x1340 && deviceId <= 0x137F) || // GM108
56 (deviceId >= 0x1380 && deviceId <= 0x13BF) || // GM107
57 (deviceId >= 0x13C0 && deviceId <= 0x13FF) || // GM204
58 (deviceId >= 0x1617 && deviceId <= 0x161A) || // GM204
59 (deviceId == 0x1667) || // GM204
60 (deviceId >= 0x17C0 && deviceId <= 0x17FF)) { // GM200
61 // Avoid HEVC on Feature Set E GPUs
62 return (videoFormat & VIDEO_FORMAT_MASK_H265) != 0;
63 }
