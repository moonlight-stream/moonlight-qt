51 void handleEvent(Event event)
52 {
53 Q_Q(Launcher);
54 Session* session;
55 NvApp app;
56
57 switch (event.type) {
58 // Occurs when CliStartStreamSegue becomes visible and the UI calls launcher's execute()
59 case Event::Executed:
60 if (m_State == StateInit) {
61 m_State = StateSeekComputer;
62 m_ComputerManager = event.computerManager;
63
64 m_ComputerSeeker = new ComputerSeeker(m_ComputerManager, m_ComputerName, q);
65 q->connect(m_ComputerSeeker, &ComputerSeeker::computerFound,
66 q, &Launcher::onComputerFound);
67 q->connect(m_ComputerSeeker, &ComputerSeeker::errorTimeout,
68 q, &Launcher::onTimeout);
69 m_ComputerSeeker->start(COMPUTER_SEEK_TIMEOUT);
70
71 q->connect(m_ComputerManager, &ComputerManager::computerStateChanged,
72 q, &Launcher::onComputerUpdated);
73 q->connect(m_ComputerManager, &ComputerManager::quitAppCompleted,
74 q, &Launcher::onQuitAppCompleted);
75
76 emit q->searchingComputer();
77 }
78 break;
79 // Occurs when searched computer is found
80 case Event::ComputerFound:
81 if (m_State == StateSeekComputer) {
82 if (event.computer->pairState == NvComputer::PS_PAIRED) {
83 m_State = StateSeekApp;
84 m_Computer = event.computer;
85 m_TimeoutTimer->start(APP_SEEK_TIMEOUT);
86 emit q->searchingApp();
87 } else {
88 m_State = StateFailure;
89 QString msg = QObject::tr("Computer %1 has not been paired. "
90 "Please open Moonlight to pair before streaming.")
91 .arg(event.computer->name);
92 emit q->failed(msg);
93 }
94 }
95 break;
96 // Occurs when a computer is updated
97 case Event::ComputerUpdated:
98 if (m_State == StateSeekApp) {
99 int index = getAppIndex();
100 if (-1 != index) {
101 app = m_Computer->appList[index];
102 m_TimeoutTimer->stop();
103 if (isNotStreaming() || isStreamingApp(app)) {
104 m_State = StateStartSession;
105 session = new Session(m_Computer, app, m_Preferences);
106 emit q->sessionCreated(app.name, session);
107 } else {
108 emit q->appQuitRequired(getCurrentAppName());
109 }
110 }
111 }
112 break;
113 // Occurs when there was another app running on computer and user accepted quit
114 // confirmation dialog
115 case Event::AppQuitRequested:
116 if (m_State == StateSeekApp) {
117 m_ComputerManager->quitRunningApp(m_Computer);
118 }
119 break;
120 // Occurs when the previous app quit has been completed, handles quitting errors if any
121 // happened. ComputerUpdated event's handler handles session start when previous app has
122 // quit.
123 case Event::AppQuitCompleted:
124 if (m_State == StateSeekApp && !event.errorMessage.isEmpty()) {
125 m_State = StateFailure;
126 emit q->failed(QObject::tr("Quitting app failed, reason: %1").arg(event.errorMessage));
127 }
128 break;
129 // Occurs when computer or app search timed out
130 case Event::Timedout:
131 if (m_State == StateSeekComputer) {
132 m_State = StateFailure;
133 emit q->failed(QObject::tr("Failed to connect to %1").arg(m_ComputerName));
134 }
135 if (m_State == StateSeekApp) {
136 m_State = StateFailure;
137 emit q->failed(QObject::tr("Failed to find application %1").arg(m_AppName));
138 }
139 break;
140 }
