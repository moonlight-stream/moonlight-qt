64
65 m_ComputerSeeker = new ComputerSeeker(m_ComputerManager, m_ComputerName, q);

72 emit q->searchingComputer();
73 }
74 break;
75 // Occurs when searched computer is found
76 case Event::ComputerFound:
77 if (m_State == StateSeekComputer) {
78 if (event.computer->pairState == NvComputer::PS_PAIRED) {
79 m_State = StateFailure;

81 emit q->failed(msg);
82 }
83 else {
84 Q_ASSERT(!m_PredefinedPin.isEmpty());

86 m_State = StatePairing;
87 m_ComputerManager->pairHost(event.computer, m_PredefinedPin);
88 emit q->pairing(event.computer->name, m_PredefinedPin);
89 }
90 }
91 break;
92 // Occurs when pairing operation completes
93 case Event::PairingCompleted:
94 if (m_State == StatePairing) {
95 if (event.errorMessage.isEmpty()) {

104 break;

107 if (m_State == StateSeekComputer) {
108 m_State = StateFailure;
