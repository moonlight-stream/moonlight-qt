13bool SLAudioRenderer::prepareForPlayback(const OPUS_MULTISTREAM_CONFIGURATION* opusConfig)
14{

22 // This number is pretty conservative (especially for surround), but
23 // it's hard to avoid since we get crushed by CPU limitations.

41 "Using SLAudio renderer with %d samples per frame",

47void SLAudioRenderer::remapChannels(POPUS_MULTISTREAM_CONFIGURATION opusConfig) {
48 OPUS_MULTISTREAM_CONFIGURATION originalConfig = *opusConfig;
49
50 // The Moonlight's default channel order is FL,FR,C,LFE,RL,RR,SL,SR
51 // SLAudio expects FL,C,FR,RL,RR,(SL,SR),LFE for 5.1/7.1 so we swap the channels around to match
52
53 if (opusConfig->channelCount == 3 || opusConfig->channelCount >= 6) {
54 // Swap FR and C
55 opusConfig->mapping[1] = originalConfig.mapping[2];
56 opusConfig->mapping[2] = originalConfig.mapping[1];
57 }
58
59 if (opusConfig->channelCount >= 6) {
60 // SLAudio expects the LFE channel at the end
61 opusConfig->mapping[opusConfig->channelCount - 1] = originalConfig.mapping[3];
62
63 // Slide the other channels down
64 memcpy(&opusConfig->mapping[3],
65 &originalConfig.mapping[4],
66 opusConfig->channelCount - 4);
67 }
68}

97bool SLAudioRenderer::submitAudio(int bytesWritten)
98{
