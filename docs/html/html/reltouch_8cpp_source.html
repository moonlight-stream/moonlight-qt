47void SdlInputHandler::handleRelativeFingerEvent(SDL_TouchFingerEvent* event)
48{
49 int fingerIndex = -1;
50
51 // Observations on Windows 10: x and y appear to be relative to 0,0 of the window client area.
52 // Although SDL documentation states they are 0.0 - 1.0 float values, they can actually be higher
53 // or lower than those values as touch events continue for touches started within the client area that
54 // leave the client area during a drag motion.
55 // dx and dy are deltas from the last touch event, not the first touch down.
56
57 // Determine the index of this finger using our list
58 // of fingers that are currently active on screen.
59 // This is also required to handle finger up which
60 // where the finger will not be in SDL_GetTouchFinger()
61 // anymore.
62 if (event->type != SDL_FINGERDOWN) {
63 for (int i = 0; i < MAX_FINGERS; i++) {
64 if (event->fingerId == m_TouchDownEvent[i].fingerId) {
65 fingerIndex = i;
66 break;
67 }
68 }
69 }
70 else {
71 // Resolve the new finger by determining the ID of each
72 // finger on the display.
73 int numTouchFingers = SDL_GetNumTouchFingers(event->touchId);
74 for (int i = 0; i < numTouchFingers; i++) {
75 SDL_Finger* finger = SDL_GetTouchFinger(event->touchId, i);
76 SDL_assert(finger != nullptr);
77 if (finger != nullptr) {
78 if (finger->id == event->fingerId) {
79 fingerIndex = i;
80 break;
81 }
82 }
83 }
84 }
85
86 if (fingerIndex < 0 || fingerIndex >= MAX_FINGERS) {
87 // Too many fingers
88 return;
89 }
90
91 // Handle cursor motion based on the position of the
92 // primary finger on screen
93 if (fingerIndex == 0) {
94 // The event x and y values are relative to our window width
95 // and height. However, we want to scale them to be relative
96 // to the host resolution. Fortunately this is easy since we
97 // already have normalized values. We'll just multiply them
98 // by the stream dimensions to get real X and Y values rather
99 // than the client window dimensions.
100 short deltaX = static_cast<short>(event->dx * m_StreamWidth);
101 short deltaY = static_cast<short>(event->dy * m_StreamHeight);
102 if (deltaX != 0 || deltaY != 0) {
103 LiSendMouseMoveEvent(deltaX, deltaY);
104 }
105 }
106
107 // Start a drag timer when primary or secondary
108 // fingers go down
109 if (event->type == SDL_FINGERDOWN &&
110 (fingerIndex == 0 || fingerIndex == 1)) {
111 SDL_RemoveTimer(m_DragTimer);
112 m_DragTimer = SDL_AddTimer(DRAG_ACTIVATION_DELAY,
113 dragTimerCallback,
114 this);
115 }
116
117 if (event->type == SDL_FINGERMOTION) {
118 // If it's outside the deadzone delta, cancel drags and taps
119 if (qSqrt(qPow(event->x - m_TouchDownEvent[fingerIndex].x, 2) +
120 qPow(event->y - m_TouchDownEvent[fingerIndex].y, 2)) > DEAD_ZONE_DELTA) {
121 SDL_RemoveTimer(m_DragTimer);
122 m_DragTimer = 0;
123
124 // This effectively cancels the tap logic below
125 m_TouchDownEvent[fingerIndex].timestamp = 0;
126 }
127 }
128
129 if (event->type == SDL_FINGERUP) {
130 // Cancel the drag timer on finger up
131 SDL_RemoveTimer(m_DragTimer);
132 m_DragTimer = 0;
133
134 // Release any drag
135 if (m_DragButton != 0) {
136 LiSendMouseButtonEvent(BUTTON_ACTION_RELEASE, m_DragButton);
137 m_DragButton = 0;
138 }
139 // 2 finger tap
140 else if (event->timestamp - m_TouchDownEvent[1].timestamp < 250) {
141 // Zero timestamp of the primary finger to ensure we won't
142 // generate a left click if the primary finger comes up soon.
143 m_TouchDownEvent[0].timestamp = 0;
144
145 // Press down the right mouse button
146 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_RIGHT);
147
148 // Queue a timer to release it in 100 ms
149 SDL_RemoveTimer(m_RightButtonReleaseTimer);
150 m_RightButtonReleaseTimer = SDL_AddTimer(TAP_BUTTON_RELEASE_DELAY,
151 releaseRightButtonTimerCallback,
152 nullptr);
153 }
154 // 1 finger tap
155 else if (event->timestamp - m_TouchDownEvent[0].timestamp < 250) {
156 // Press down the left mouse button
157 LiSendMouseButtonEvent(BUTTON_ACTION_PRESS, BUTTON_LEFT);
158
159 // Queue a timer to release it in 100 ms
160 SDL_RemoveTimer(m_LeftButtonReleaseTimer);
161 m_LeftButtonReleaseTimer = SDL_AddTimer(TAP_BUTTON_RELEASE_DELAY,
162 releaseLeftButtonTimerCallback,
163 nullptr);
164 }
165 }
166
167 m_NumFingersDown = SDL_GetNumTouchFingers(event->touchId);
168
169 if (event->type == SDL_FINGERDOWN) {
170 m_TouchDownEvent[fingerIndex] = *event;
171 }
172 else if (event->type == SDL_FINGERUP) {
173 m_TouchDownEvent[fingerIndex] = {};
174 }
175}
