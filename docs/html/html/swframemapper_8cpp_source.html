34 m_SwPixelFormat = (AVPixelFormat)outputFrame->format;
35 m_MapFrame = true;
36 goto Exit;
37 }
38 else {
39 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,

41 outputFrame->format);

43 }

46 "av_hwframe_map() is unsupported (error: %d)",

62 // and non-preferred rendering format rather than the other way around. This is
63 // why we loop through the readback format list in order, rather than searching
64 // for the format from getPreferredPixelFormat() in the list.

67 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,

69 formats[i]);
70 continue;
71 }
72
73 m_SwPixelFormat = formats[i];
74 break;
75 }
76
77 av_freep(&formats);
78
79Exit:

81

83 if (m_SwPixelFormat == AV_PIX_FMT_NONE) {
84 if (m_Renderer->isPixelFormatSupported(m_VideoFormat, hwFrameCtx->sw_format)) {
85 m_SwPixelFormat = hwFrameCtx->sw_format;
86 }
87 else {
88 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
89 "Unable to find compatible hwframe transfer format (sw_format = %d)",

92 }
93 }

97 m_SwPixelFormat,
98 m_MapFrame ? "yes" : "no");
99 return true;

103{
104 int err;
105
106 // setVideoFormat() must have been called before our first frame
107 SDL_assert(m_VideoFormat != 0);
108
109 if (m_SwPixelFormat == AV_PIX_FMT_NONE) {

125 // performance penalties on Intel hardware with VAAPI due to mappings

131 err);

135 }
136 else {
137 err = av_hwframe_transfer_data(swFrame, hwFrame, 0);
138 if (err < 0) {

140 "av_hwframe_transfer_data() failed: %d",
141 err);
142 av_frame_free(&swFrame);
143 return nullptr;

145
146 // av_hwframe_transfer_data() doesn't transfer metadata
147 // (and can even nuke existing metadata in dst), so we
