358 void getFramePremultipliedCscConstants(const AVFrame* frame, std::array<float, 9> &cscMatrix, std::array<float, 3> &offsets) {
359 static const std::array<float, 9> k_CscMatrix_Bt601 = {
360 1.0f, 1.0f, 1.0f,
361 0.0f, -0.3441f, 1.7720f,
362 1.4020f, -0.7141f, 0.0f,
363 };
364 static const std::array<float, 9> k_CscMatrix_Bt709 = {
365 1.0f, 1.0f, 1.0f,
366 0.0f, -0.1873f, 1.8556f,
367 1.5748f, -0.4681f, 0.0f,
368 };
369 static const std::array<float, 9> k_CscMatrix_Bt2020 = {
370 1.0f, 1.0f, 1.0f,
371 0.0f, -0.1646f, 1.8814f,
372 1.4746f, -0.5714f, 0.0f,
373 };
374
375 bool fullRange = isFrameFullRange(frame);
376 int bitsPerChannel = getFrameBitsPerChannel(frame);
377 int channelRange = (1 << bitsPerChannel);
378 double yMin = (fullRange ? 0 : (16 << (bitsPerChannel - 8)));
379 double yMax = (fullRange ? (channelRange - 1) : (235 << (bitsPerChannel - 8)));
380 double yScale = (channelRange - 1) / (yMax - yMin);
381 double uvMin = (fullRange ? 0 : (16 << (bitsPerChannel - 8)));
382 double uvMax = (fullRange ? (channelRange - 1) : (240 << (bitsPerChannel - 8)));
383 double uvScale = (channelRange - 1) / (uvMax - uvMin);
384
385 // Calculate YUV offsets
386 offsets[0] = yMin / (double)(channelRange - 1);
387 offsets[1] = (channelRange / 2) / (double)(channelRange - 1);
388 offsets[2] = (channelRange / 2) / (double)(channelRange - 1);
389
390 // Start with the standard full range color matrix
391 switch (getFrameColorspace(frame)) {
392 default:
393 case COLORSPACE_REC_601:
394 cscMatrix = k_CscMatrix_Bt601;
395 break;
396 case COLORSPACE_REC_709:
397 cscMatrix = k_CscMatrix_Bt709;
398 break;
399 case COLORSPACE_REC_2020:
400 cscMatrix = k_CscMatrix_Bt2020;
401 break;
402 }
403
404 // Scale the color matrix according to the color range
405 for (int i = 0; i < 3; i++) {
406 cscMatrix[i] *= yScale;
407 }
408 for (int i = 3; i < 9; i++) {
409 cscMatrix[i] *= uvScale;
410 }
411 }

413 void getFrameChromaCositingOffsets(const AVFrame* frame, std::array<float, 2> &chromaOffsets) {
414 const AVPixFmtDescriptor* formatDesc = av_pix_fmt_desc_get(getFrameSwPixelFormat(frame));
415 if (!formatDesc) {
416 SDL_assert(formatDesc);
417 chromaOffsets.fill(0);
418 return;
419 }
420
421 SDL_assert(formatDesc->log2_chroma_w <= 1);
422 SDL_assert(formatDesc->log2_chroma_h <= 1);
423
424 switch (frame->chroma_location) {
425 default:
426 case AVCHROMA_LOC_LEFT:
427 chromaOffsets[0] = 0.5;
428 chromaOffsets[1] = 0;
429 break;
430 case AVCHROMA_LOC_CENTER:
431 chromaOffsets[0] = 0;
432 chromaOffsets[1] = 0;
433 break;
434 case AVCHROMA_LOC_TOPLEFT:
435 chromaOffsets[0] = 0.5;
436 chromaOffsets[1] = 0.5;
437 break;
438 case AVCHROMA_LOC_TOP:
439 chromaOffsets[0] = 0;
440 chromaOffsets[1] = 0.5;
441 break;
442 case AVCHROMA_LOC_BOTTOMLEFT:
443 chromaOffsets[0] = 0.5;
444 chromaOffsets[1] = -0.5;
445 break;
446 case AVCHROMA_LOC_BOTTOM:
447 chromaOffsets[0] = 0;
448 chromaOffsets[1] = -0.5;
449 break;
450 }
451
452 // Force the offsets to 0 if chroma is not subsampled in that dimension
453 if (formatDesc->log2_chroma_w == 0) {
454 chromaOffsets[0] = 0;
455 }
456 if (formatDesc->log2_chroma_h == 0) {
457 chromaOffsets[1] = 0;
458 }
