28void SdlGamepadKeyNavigation::enable()
29{
30 if (m_Enabled) {
31 return;
32 }
33
34 // We have to initialize and uninitialize this in enable()/disable()
35 // because we need to get out of the way of the Session class. If it
36 // doesn't get to reinitialize the GC subsystem, it won't get initial
37 // arrival events. Additionally, there's a race condition between
38 // our QML objects being destroyed and SDL being deinitialized that
39 // this solves too.
40 if (SDL_InitSubSystem(SDL_INIT_GAMECONTROLLER) != 0) {
41 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
42 "SDL_InitSubSystem(SDL_INIT_GAMECONTROLLER) failed: %s",
43 SDL_GetError());
44 return;
45 }
46
47 MappingManager mappingManager;
48 mappingManager.applyMappings();
49
50 // Drop all pending gamepad add events. SDL will generate these for us
51 // on first init of the GC subsystem. We can't depend on them due to
52 // overlapping lifetimes of SdlGamepadKeyNavigation instances, so we
53 // will attach ourselves.
54 SDL_PumpEvents();
55 SDL_FlushEvent(SDL_CONTROLLERDEVICEADDED);
56
57 // Open all currently attached game controllers
58 int numJoysticks = SDL_NumJoysticks();
59 for (int i = 0; i < numJoysticks; i++) {
60 if (SDL_IsGameController(i)) {
61 SDL_GameController* gc = SDL_GameControllerOpen(i);
62 if (gc != nullptr) {
63 m_Gamepads.append(gc);
64 }
65 }
66 }
67
68 m_Enabled = true;
69
70 // Start the polling timer if the window is focused
71 updateTimerState();
72}

98void SdlGamepadKeyNavigation::onPollingTimerFired()
99{
100 SDL_Event event;
101
102 // Discard any pending button events on the first poll to avoid picking up
103 // stale input data from the stream session (like the quit combo).
104 if (m_FirstPoll) {
105 SDL_PumpEvents();
106 SDL_FlushEvent(SDL_CONTROLLERBUTTONDOWN);
107 SDL_FlushEvent(SDL_CONTROLLERBUTTONUP);
108
109 m_FirstPoll = false;
110 }
111
112 while (SDL_PollEvent(&event)) {
113 switch (event.type) {
114 case SDL_QUIT:
115 // SDL may send us a quit event since we initialize
116 // the video subsystem on startup. If we get one,
117 // forward it on for Qt to take care of.
118 QCoreApplication::instance()->quit();
119 break;
120 case SDL_CONTROLLERBUTTONDOWN:
121 case SDL_CONTROLLERBUTTONUP:
122 {
123 QEvent::Type type =
124 event.type == SDL_CONTROLLERBUTTONDOWN ?
125 QEvent::Type::KeyPress : QEvent::Type::KeyRelease;
126
127 // Swap face buttons if needed
128 if (m_Prefs->swapFaceButtons) {
129 switch (event.cbutton.button) {
130 case SDL_CONTROLLER_BUTTON_A:
131 event.cbutton.button = SDL_CONTROLLER_BUTTON_B;
132 break;
133 case SDL_CONTROLLER_BUTTON_B:
134 event.cbutton.button = SDL_CONTROLLER_BUTTON_A;
135 break;
136 case SDL_CONTROLLER_BUTTON_X:
137 event.cbutton.button = SDL_CONTROLLER_BUTTON_Y;
138 break;
139 case SDL_CONTROLLER_BUTTON_Y:
140 event.cbutton.button = SDL_CONTROLLER_BUTTON_X;
141 break;
142 }
143 }
144
145 switch (event.cbutton.button) {
146 case SDL_CONTROLLER_BUTTON_DPAD_UP:
147 if (m_UiNavMode) {
148 // Back-tab
149 sendKey(type, Qt::Key_Tab, Qt::ShiftModifier);
150 }
151 else {
152 sendKey(type, Qt::Key_Up);
153 }
154 break;
155 case SDL_CONTROLLER_BUTTON_DPAD_DOWN:
156 if (m_UiNavMode) {
157 sendKey(type, Qt::Key_Tab);
158 }
159 else {
160 sendKey(type, Qt::Key_Down);
161 }
162 break;
163 case SDL_CONTROLLER_BUTTON_DPAD_LEFT:
164 sendKey(type, Qt::Key_Left);
165 break;
166 case SDL_CONTROLLER_BUTTON_DPAD_RIGHT:
167 sendKey(type, Qt::Key_Right);
168 break;
169 case SDL_CONTROLLER_BUTTON_A:
170 if (m_UiNavMode) {
171 sendKey(type, Qt::Key_Space);
172 }
173 else {
174 sendKey(type, Qt::Key_Return);
175 }
176 break;
177 case SDL_CONTROLLER_BUTTON_B:
178 sendKey(type, Qt::Key_Escape);
179 break;
180 case SDL_CONTROLLER_BUTTON_X:
181 sendKey(type, Qt::Key_Menu);
182 break;
183 case SDL_CONTROLLER_BUTTON_Y:
184 case SDL_CONTROLLER_BUTTON_START:
185 // HACK: We use this keycode to inform main.qml
186 // to show the settings when Key_Menu is handled
187 // by the control in focus.
188 sendKey(type, Qt::Key_Hangup);
189 break;
190 default:
191 break;
192 }
193 break;
194 }
195 case SDL_CONTROLLERDEVICEADDED:
196 SDL_GameController* gc = SDL_GameControllerOpen(event.cdevice.which);
197 if (gc != nullptr) {
198 // SDL_CONTROLLERDEVICEADDED can be reported multiple times for the same
199 // gamepad in rare cases, because SDL doesn't fixup the device index in
200 // the SDL_CONTROLLERDEVICEADDED event if an unopened gamepad disappears
201 // before we've processed the add event.
202 if (!m_Gamepads.contains(gc)) {
203 m_Gamepads.append(gc);
204 }
205 else {
206 // We already have this game controller open
207 SDL_GameControllerClose(gc);
208 }
209 }
210 break;
211 }
212 }
213
214 // Handle analog sticks by polling
215 for (auto gc : m_Gamepads) {
216 short leftX = SDL_GameControllerGetAxis(gc, SDL_CONTROLLER_AXIS_LEFTX);
217 short leftY = SDL_GameControllerGetAxis(gc, SDL_CONTROLLER_AXIS_LEFTY);
218 if (SDL_GetTicks() - m_LastAxisNavigationEventTime < AXIS_NAVIGATION_REPEAT_DELAY) {
219 // Do nothing
220 }
221 else if (leftY < -30000) {
222 if (m_UiNavMode) {
223 // Back-tab
224 sendKey(QEvent::Type::KeyPress, Qt::Key_Tab, Qt::ShiftModifier);
225 sendKey(QEvent::Type::KeyRelease, Qt::Key_Tab, Qt::ShiftModifier);
226 }
227 else {
228 sendKey(QEvent::Type::KeyPress, Qt::Key_Up);
229 sendKey(QEvent::Type::KeyRelease, Qt::Key_Up);
230 }
231
232 m_LastAxisNavigationEventTime = SDL_GetTicks();
233 }
234 else if (leftY > 30000) {
235 if (m_UiNavMode) {
236 sendKey(QEvent::Type::KeyPress, Qt::Key_Tab);
237 sendKey(QEvent::Type::KeyRelease, Qt::Key_Tab);
238 }
239 else {
240 sendKey(QEvent::Type::KeyPress, Qt::Key_Down);
241 sendKey(QEvent::Type::KeyRelease, Qt::Key_Down);
242 }
243
244 m_LastAxisNavigationEventTime = SDL_GetTicks();
245 }
246 else if (leftX < -30000) {
247 sendKey(QEvent::Type::KeyPress, Qt::Key_Left);
248 sendKey(QEvent::Type::KeyRelease, Qt::Key_Left);
249 m_LastAxisNavigationEventTime = SDL_GetTicks();
250 }
251 else if (leftX > 30000) {
252 sendKey(QEvent::Type::KeyPress, Qt::Key_Right);
253 sendKey(QEvent::Type::KeyRelease, Qt::Key_Right);
254 m_LastAxisNavigationEventTime = SDL_GetTicks();
255 }
256 }

270{
