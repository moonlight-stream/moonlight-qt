25double BandwidthTracker::GetAverageMbps() {
26 std::lock_guard<std::mutex> lock(mtx);
27 auto now = steady_clock::now();
28 auto ms = duration_cast<milliseconds>(now.time_since_epoch());
29 int currentIndex = (ms.count() / bucketIntervalMs) % bucketCount;
30 int maxBuckets = bucketCount / 4;
31 size_t totalBytes = 0;
32 steady_clock::time_point oldestBucket = now;
33
34 // Sum bytes from 25% most recent buckets as long as they are completed
35 for (int i = 0; i < maxBuckets; i++) {
36 int idx = (currentIndex - i + bucketCount) % bucketCount;
37 const Bucket &bucket = buckets[idx];
38 if (isValid(bucket, now) && (now - bucket.start >= milliseconds(bucketIntervalMs))) {
39 totalBytes += bucket.bytes;
40 if (bucket.start < oldestBucket) {
41 oldestBucket = bucket.start;
42 }
43 }
44 }
45
46 double elapsed = duration<double>(now - oldestBucket).count();
47 if (elapsed <= 0.0) {
48 return 0.0;
49 }
50
51 return totalBytes * 8.0 / 1000000.0 / elapsed;
52}
