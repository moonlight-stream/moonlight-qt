70StreamingPreferences* StreamingPreferences::get(QQmlEngine *qmlEngine)
71{
72 {
73 QReadLocker readGuard(&s_GlobalPrefsLock);
74
75 // If we have a preference object and it's associated with a QML engine or
76 // if the caller didn't specify a QML engine, return the existing object.
77 if (s_GlobalPrefs && (s_GlobalPrefs->m_QmlEngine || !qmlEngine)) {
78 // The lifetime logic here relies on the QML engine also being a singleton.
79 Q_ASSERT(!qmlEngine || s_GlobalPrefs->m_QmlEngine == qmlEngine);
80 return s_GlobalPrefs;
81 }
82 }
83
84 {
85 QWriteLocker writeGuard(&s_GlobalPrefsLock);
86
87 // If we already have an preference object but the QML engine is now available,
88 // associate the QML engine with the preferences.
89 if (s_GlobalPrefs) {
90 if (!s_GlobalPrefs->m_QmlEngine) {
91 s_GlobalPrefs->m_QmlEngine = qmlEngine;
92 }
93 else {
94 // We could reach this codepath if another thread raced with us
95 // and created the object while we were outside the pref lock.
96 Q_ASSERT(!qmlEngine || s_GlobalPrefs->m_QmlEngine == qmlEngine);
97 }
98 }
99 else {
100 s_GlobalPrefs = new StreamingPreferences(qmlEngine);
101 }
102
103 return s_GlobalPrefs;

209bool StreamingPreferences::retranslate()
210{
211 static QTranslator* translator = nullptr;
212
213#if QT_VERSION < QT_VERSION_CHECK(5, 10, 0)
214 if (m_QmlEngine != nullptr) {
215 // Dynamic retranslation is not supported until Qt 5.10
216 return false;
217 }
218#endif
219
220 QTranslator* newTranslator = new QTranslator();
221 QString languageSuffix = getSuffixFromLanguage(language);
222
223 // Remove the old translator, even if we can't load a new one.
224 // Otherwise we'll be stuck with the old translated values instead
225 // of defaulting to English.
226 if (translator != nullptr) {
227 QCoreApplication::removeTranslator(translator);
228 delete translator;
229 translator = nullptr;
230 }
231
232 if (newTranslator->load(QString(":/languages/qml_") + languageSuffix)) {
233 qInfo() << "Successfully loaded translation for" << languageSuffix;
234
235 translator = newTranslator;
236 QCoreApplication::installTranslator(translator);
237 }
238 else {
239 qInfo() << "No translation available for" << languageSuffix;
240 delete newTranslator;
241 }
242
243 if (m_QmlEngine != nullptr) {
244#if QT_VERSION >= QT_VERSION_CHECK(5, 10, 0)
245 // This is a dynamic retranslation from the settings page.
246 // We have to kick the QML engine into reloading our text.
247 m_QmlEngine->retranslate();
248#else
249 // Unreachable below Qt 5.10 due to the check above
250 Q_ASSERT(false);
251#endif
252 }
253 else {
254 // This is a translation from a non-QML context, which means
255 // it is probably app startup. There's nothing to refresh.
256 }
257
258 return true;
259}

381int StreamingPreferences::getDefaultBitrate(int width, int height, int fps, bool yuv444)
382{
383 // Don't scale bitrate linearly beyond 60 FPS. It's definitely not a linear
384 // bitrate increase for frame rate once we get to values that high.
385 float frameRateFactor = (fps <= 60 ? fps : (qSqrt(fps / 60.f) * 60.f)) / 30.f;
386
387 // TODO: Collect some empirical data to see if these defaults make sense.
388 // We're just using the values that the Shield used, as we have for years.
389 static const struct resTable {
390 int pixels;
391 int factor;
392 } resTable[] {
393 { 640 * 360, 1 },
394 { 854 * 480, 2 },
395 { 1280 * 720, 5 },
396 { 1920 * 1080, 10 },
397 { 2560 * 1440, 20 },
398 { 3840 * 2160, 40 },
399 { -1, -1 },
400 };
401
402 // Calculate the resolution factor by linear interpolation of the resolution table
403 float resolutionFactor;
404 int pixels = width * height;
405 for (int i = 0;; i++) {
406 if (pixels == resTable[i].pixels) {
407 // We can bail immediately for exact matches
408 resolutionFactor = resTable[i].factor;
409 break;
410 }
411 else if (pixels < resTable[i].pixels) {
412 if (i == 0) {
413 // Never go below the lowest resolution entry
414 resolutionFactor = resTable[i].factor;
415 }
416 else {
417 // Interpolate between the entry greater than the chosen resolution (i) and the entry less than the chosen resolution (i-1)
418 resolutionFactor = ((float)(pixels - resTable[i-1].pixels) / (resTable[i].pixels - resTable[i-1].pixels)) * (resTable[i].factor - resTable[i-1].factor) + resTable[i-1].factor;
419 }
420 break;
421 }
422 else if (resTable[i].pixels == -1) {
423 // Never go above the highest resolution entry
424 resolutionFactor = resTable[i-1].factor;
425 break;
426 }
427 }
428
429 if (yuv444) {
430 // This is rough estimation based on the fact that 4:4:4 doubles the amount of raw YUV data compared to 4:2:0
431 resolutionFactor *= 2;
432 }
433
434 return qRound(resolutionFactor * frameRateFactor) * 1000;
435}
