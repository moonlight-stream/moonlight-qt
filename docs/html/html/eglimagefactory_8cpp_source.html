431bool EglImageFactory::supportsImportingFormat(EGLDisplay dpy, EGLint format)
432{
433 if (!m_eglQueryDmaBufFormatsEXT) {
434 // These are the standard formats used for importing separate layers of NV12.
435 // We will assume all EGL implementations can handle these.
436 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
437 "Assuming R8 and GR88 format support because eglQueryDmaBufFormatsEXT() is not supported");
438 return format == DRM_FORMAT_R8 || format == DRM_FORMAT_GR88;
439 }
440
441 // Get the number of formats
442 EGLint numFormats;
443 if (!m_eglQueryDmaBufFormatsEXT(dpy, 0, nullptr, &numFormats)) {
444 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
445 "eglQueryDmaBufFormatsEXT() #1 failed: %d", eglGetError());
446 return false;
447 }
448 else if (numFormats == 0) {
449 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
450 "eglQueryDmaBufFormatsEXT() returned no supported formats!");
451 return false;
452 }
453
454 std::vector<EGLint> formats(numFormats);
455 if (!m_eglQueryDmaBufFormatsEXT(dpy, numFormats, formats.data(), &numFormats)) {
456 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
457 "eglQueryDmaBufFormatsEXT() #2 failed: %d", eglGetError());
458 return false;
459 }
460
461 for (EGLint i = 0; i < numFormats; i++) {
462 if (format == formats[i]) {
463 return true;
464 }
465 }
466
467 return false;
468}
