53bool CompatFetcher::isGfeVersionSupported(QString gfeVersion)
54{
55 QSettings settings;
56
57 if (gfeVersion.trimmed().isEmpty()) {
58 // If we don't have a GFE version, just allow it
59 return true;
60 }
61
62 QString latestSupportedVersion = settings.value(COMPAT_KEY COMPAT_VERSION).toString();
63 if (latestSupportedVersion.isEmpty()) {
64 // We don't have compat data yet, so just assume it's supported
65 return true;
66 }
67
68 QStringList latestSupportedVersionQuad = latestSupportedVersion.split('.');
69 QStringList gfeVersionQuad = gfeVersion.split('.');
70
71 if (gfeVersionQuad.count() <= 1) {
72 qWarning() << "Failed to parse GFE version:" << gfeVersion;
73 return true;
74 }
75
76 if (latestSupportedVersionQuad.count() <= 1) {
77 qWarning() << "Failed to parse latest supported version:" << latestSupportedVersion;
78 return true;
79 }
80
81 for (int i = 0;; i++) {
82 int actualVerVal = 0;
83 int latestSupportedVal = 0;
84
85 // Treat missing decimal places as 0
86 if (i < gfeVersionQuad.count()) {
87 bool ok;
88
89 actualVerVal = gfeVersionQuad[i].toInt(&ok);
90 if (!ok || actualVerVal < 0) {
91 // Return true to be safe
92 qWarning() << "Failed to parse GFE version:" << gfeVersion;
93 return true;
94 }
95 }
96 if (i < latestSupportedVersionQuad.count()) {
97 bool ok;
98
99 latestSupportedVal = latestSupportedVersionQuad[i].toInt(&ok);
100 if (!ok || latestSupportedVal < 0) {
101 // Return true to be safe
102 qWarning() << "Failed to parse latest supported version:" << latestSupportedVersion;
103 return true;
104 }
105 }
106
107 if (i >= gfeVersionQuad.count() && i >= latestSupportedVersionQuad.count()) {
108 // Equal versions - this is fine
109 return true;
110 }
111
112 if (actualVerVal < latestSupportedVal) {
113 // Actual version is lower than latest supported - this is fine
114 return true;
115 }
116 else if (actualVerVal > latestSupportedVal) {
117 // Actual version is greater than latest supported - this is bad
118 qWarning() << "GFE version" << gfeVersion << "is not supported by this version of Moonlight";
119 return false;
120 }
121 }
