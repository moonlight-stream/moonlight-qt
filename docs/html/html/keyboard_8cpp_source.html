16void SdlInputHandler::performSpecialKeyCombo(KeyCombo combo)
17{
18 switch (combo) {
19 case KeyComboQuit:
20 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
21 "Detected quit key combo");
22
23 // Push a quit event to the main loop
24 SDL_Event event;
25 event.type = SDL_QUIT;
26 event.quit.timestamp = SDL_GetTicks();
27 SDL_PushEvent(&event);
28 break;
29
30 case KeyComboUngrabInput:
31 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
32 "Detected mouse capture toggle combo");
33
34 // Stop handling future input
35 setCaptureActive(!isCaptureActive());
36
37 // Force raise all keys to ensure they aren't stuck,
38 // since we won't get their key up events.
39 raiseAllKeys();
40 break;
41
42 case KeyComboToggleFullScreen:
43 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
44 "Detected full-screen toggle combo");
45 Session::s_ActiveSession->toggleFullscreen();
46
47 // Force raise all keys just be safe across this full-screen/windowed
48 // transition just in case key events get lost.
49 raiseAllKeys();
50 break;
51
52 case KeyComboToggleStatsOverlay:
53 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
54 "Detected stats toggle combo");
55
56 // Toggle the stats overlay
57 Session::get()->getOverlayManager().setOverlayState(Overlay::OverlayDebug,
58 !Session::get()->getOverlayManager().isOverlayEnabled(Overlay::OverlayDebug));
59 break;
60
61 case KeyComboToggleMouseMode:
62 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
63 "Detected mouse mode toggle combo");
64
65 // Uncapture input
66 setCaptureActive(false);
67
68 // Toggle mouse mode
69 m_AbsoluteMouseMode = !m_AbsoluteMouseMode;
70
71 // Recapture input
72 setCaptureActive(true);
73 break;
74
75 case KeyComboToggleCursorHide:
76 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
77 "Detected show mouse combo");
78
79 if (!SDL_GetRelativeMouseMode()) {
80 m_MouseCursorCapturedVisibilityState = !m_MouseCursorCapturedVisibilityState;
81 SDL_ShowCursor(m_MouseCursorCapturedVisibilityState);
82 }
83 else {
84 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
85 "Cursor can only be shown in remote desktop mouse mode");
86 }
87 break;
88
89 case KeyComboToggleMinimize:
90 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
91 "Detected minimize combo");
92 SDL_MinimizeWindow(m_Window);
93 break;
94
95 case KeyComboPasteText:
96 {
97 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
98 "Detected type clipboard text combo");
99
100 // Force raise all keys to ensure that none of them interfere
101 // with the text we're going to type.
102 raiseAllKeys();
103
104 char* text;
105 if (SDL_HasClipboardText() && (text = SDL_GetClipboardText()) != nullptr) {
106 // Sending both CR and LF will lead to two newlines in the destination for
107 // each newline in the source, so we fix up any CRLFs into just a single LF.
108 for (char* c = text; *c != 0; c++) {
109 if (*c == '\r' && *(c + 1) == '\n') {
110 // We're using strlen() rather than strlen() - 1 since we need to add 1
111 // to copy the null terminator which is not included in strlen()'s count.
112 memmove(c, c + 1, strlen(c));
113 }
114 }
115
116 // Send this text to the PC
117 LiSendUtf8TextEvent(text, (unsigned int)strlen(text));
118
119 // SDL_GetClipboardText() allocates, so we must free
120 SDL_free((void*)text);
121 }
122 else {
123 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
124 "No text in clipboard to paste!");
125 }
126 break;
127 }
128
129 case KeyComboTogglePointerRegionLock:
130 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
131 "Detected pointer region lock toggle combo");
132 m_PointerRegionLockActive = !m_PointerRegionLockActive;
133
134 // Remember that the user changed this manually, so we don't mess with it anymore
135 // during windowed <-> full-screen transitions.
136 m_PointerRegionLockToggledByUser = true;
137
138 // Apply the new region lock
139 updatePointerRegionLock();
140 break;
141
142 case KeyComboQuitAndExit:
143 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
144 "Detected quitAndExit key combo");
145
146 // Indicate that we want to exit afterwards
147 Session::get()->setShouldExitAfterQuit();
148
149 // Push a quit event to the main loop
150 SDL_Event quitExitEvent;
151 quitExitEvent.type = SDL_QUIT;
152 quitExitEvent.quit.timestamp = SDL_GetTicks();
153 SDL_PushEvent(&quitExitEvent);
154 break;
155
156 default:
157 Q_UNREACHABLE();
158 }

167 if (event->repeat) {
168 // Ignore repeat key down events
169 SDL_assert(event->state == SDL_PRESSED);
170 return;
171 }
172
173 // Check for our special key combos
174 if ((event->state == SDL_PRESSED) &&
175 (event->keysym.mod & KMOD_CTRL) &&
176 (event->keysym.mod & KMOD_ALT) &&
177 (event->keysym.mod & KMOD_SHIFT)) {
178 // First we test the SDLK combos for matches,
179 // that way we ensure that latin keyboard users
180 // can match to the key they see on their keyboards.
181 // If nothing matches that, we'll then go on to
182 // checking scancodes so non-latin keyboard users
183 // can have working hotkeys (though possibly in
184 // odd positions). We must do all SDLK tests before
185 // any scancode tests to avoid issues in cases
186 // where the SDLK for one shortcut collides with
187 // the scancode of another.
188
189 for (int i = 0; i < KeyComboMax; i++) {
190 if (m_SpecialKeyCombos[i].enabled && event->keysym.sym == m_SpecialKeyCombos[i].keyCode) {
191 performSpecialKeyCombo(m_SpecialKeyCombos[i].keyCombo);
192 return;
193 }
194 }
195
196 for (int i = 0; i < KeyComboMax; i++) {
197 if (m_SpecialKeyCombos[i].enabled && event->keysym.scancode == m_SpecialKeyCombos[i].scanCode) {
198 performSpecialKeyCombo(m_SpecialKeyCombos[i].keyCombo);
199 return;
200 }
201 }
202 }
203
204 // Set modifier flags
205 modifiers = 0;
206 if (event->keysym.mod & KMOD_CTRL) {
207 modifiers |= MODIFIER_CTRL;
208 }
209 if (event->keysym.mod & KMOD_ALT) {
210 modifiers |= MODIFIER_ALT;
211 }
212 if (event->keysym.mod & KMOD_SHIFT) {
213 modifiers |= MODIFIER_SHIFT;
214 }
215 if (event->keysym.mod & KMOD_GUI) {
216 if (isSystemKeyCaptureActive()) {

219 }

225 // SDL defines SDL_SCANCODE_0 > SDL_SCANCODE_9, so we need to handle that manually
226 keyCode = (event->keysym.scancode - SDL_SCANCODE_1) + VK_0 + 1;
227 }
