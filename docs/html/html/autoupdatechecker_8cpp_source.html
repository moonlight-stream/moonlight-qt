81int AutoUpdateChecker::compareVersion(QVector<int>& version1, QVector<int>& version2) {
82 for (int i = 0;; i++) {
83 int v1Val = 0;
84 int v2Val = 0;
85
86 // Treat missing decimal places as 0
87 if (i < version1.count()) {
88 v1Val = version1[i];
89 }
90 if (i < version2.count()) {
91 v2Val = version2[i];
92 }
93 if (i >= version1.count() && i >= version2.count()) {
94 // Equal versions
95 return 0;
96 }
97
98 if (v1Val < v2Val) {
99 return -1;
100 }
101 else if (v1Val > v2Val) {
102 return 1;
103 }
