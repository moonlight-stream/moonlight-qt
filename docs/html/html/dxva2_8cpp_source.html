149bool DXVA2Renderer::initializeDecoder()
150{
151 HRESULT hr;
152
153 if (isDecoderBlacklisted()) {
154 return false;
155 }
156
157 hr = DXVA2CreateVideoService(m_Device.Get(), IID_IDirectXVideoDecoderService, &m_DecService);
158 if (FAILED(hr)) {
159 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
160 "DXVA2CreateVideoService(IID_IDirectXVideoDecoderService) failed: %x",
161 hr);
162 return false;
163 }
164
165 GUID* guids;
166 GUID chosenDeviceGuid;
167 UINT guidCount;
168 hr = m_DecService->GetDecoderDeviceGuids(&guidCount, &guids);
169 if (FAILED(hr)) {
170 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
171 "GetDecoderDeviceGuids() failed: %x",
172 hr);
173 return false;
174 }
175
176 UINT i;
177 for (i = 0; i < guidCount; i++) {
178 if (m_VideoFormat == VIDEO_FORMAT_H264) {
179 if (IsEqualGUID(guids[i], DXVA2_ModeH264_E) ||
180 IsEqualGUID(guids[i], DXVA2_ModeH264_F)) {
181 chosenDeviceGuid = guids[i];
182 break;
183 }
184 else if (IsEqualGUID(guids[i], DXVADDI_Intel_ModeH264_E)) {
185 chosenDeviceGuid = guids[i];
186 m_DXVAContext.workaround |= FF_DXVA2_WORKAROUND_INTEL_CLEARVIDEO;
187 break;
188 }
189 }
190 else if (m_VideoFormat == VIDEO_FORMAT_H265) {
191 if (IsEqualGUID(guids[i], DXVA2_ModeHEVC_VLD_Main)) {
192 chosenDeviceGuid = guids[i];
193 break;
194 }
195 }
196 else if (m_VideoFormat == VIDEO_FORMAT_H265_MAIN10) {
197 if (IsEqualGUID(guids[i], DXVA2_ModeHEVC_VLD_Main10)) {
198 chosenDeviceGuid = guids[i];
199 break;
200 }
201 }
202 else if (m_VideoFormat == VIDEO_FORMAT_AV1_MAIN8 || m_VideoFormat == VIDEO_FORMAT_AV1_MAIN10) {
203 if (IsEqualGUID(guids[i], DXVA2_ModeAV1_VLD_Profile0)) {
204 chosenDeviceGuid = guids[i];
205 break;
206 }
207 }
208 else if (m_VideoFormat == VIDEO_FORMAT_AV1_HIGH8_444 || m_VideoFormat == VIDEO_FORMAT_AV1_HIGH10_444) {
209 if (IsEqualGUID(guids[i], DXVA2_ModeAV1_VLD_Profile1)) {
210 chosenDeviceGuid = guids[i];
211 break;
212 }
213 }
214 }
215
216 CoTaskMemFree(guids);
217
218 if (i == guidCount) {
219 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
220 "No matching decoder device GUIDs");
221 return false;
222 }
223
224 DXVA2_ConfigPictureDecode* configs;
225 UINT configCount;
226 hr = m_DecService->GetDecoderConfigurations(chosenDeviceGuid, &m_Desc, nullptr, &configCount, &configs);
227 if (FAILED(hr)) {
228 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
229 "GetDecoderConfigurations() failed: %x",
230 hr);
231 return false;
232 }
233
234 for (i = 0; i < configCount; i++) {
235 if ((configs[i].ConfigBitstreamRaw == 1 || configs[i].ConfigBitstreamRaw == 2) &&
236 IsEqualGUID(configs[i].guidConfigBitstreamEncryption, DXVA2_NoEncrypt)) {
237 m_Config = configs[i];
238 break;
239 }
240 }
241
242 CoTaskMemFree(configs);
243
244 if (i == configCount) {
245 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
246 "No matching decoder configurations");
247 return false;
248 }
249
250 // Alignment was already taken care of
251 SDL_assert(m_Desc.SampleWidth % 16 == 0);
252 SDL_assert(m_Desc.SampleHeight % 16 == 0);
253 hr = m_DecService->CreateSurface(m_Desc.SampleWidth,
254 m_Desc.SampleHeight,
255 (UINT)m_DecSurfacesRaw.size() - 1,
256 m_Desc.Format,
257 D3DPOOL_DEFAULT,
258 0,
259 DXVA2_VideoDecoderRenderTarget,
260 m_DecSurfacesRaw.data(),
261 nullptr);
262 if (FAILED(hr)) {
263 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
264 "CreateSurface() failed: %x",
265 hr);
266 return false;
267 }
268
269 // Transfer ownership into ComPtrs
270 for (int i = 0; i < m_DecSurfaces.size(); i++) {
271 m_DecSurfaces[i].Attach(m_DecSurfacesRaw[i]);
272 }
273
274 hr = m_DecService->CreateVideoDecoder(chosenDeviceGuid, &m_Desc, &m_Config,
275 m_DecSurfacesRaw.data(),
276 (UINT)m_DecSurfacesRaw.size(),
277 &m_Decoder);
278 if (FAILED(hr)) {
279 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
280 "CreateVideoDecoder() failed: %x",
281 hr);
282 return false;
283 }
284
285 return true;
286}

288bool DXVA2Renderer::initializeRenderer()
289{
290 HRESULT hr;
291
292 hr = m_Device->GetBackBuffer(0, 0, D3DBACKBUFFER_TYPE_MONO, &m_RenderTarget);
293 if (FAILED(hr)) {
294 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
295 "GetBackBuffer() failed: %x",
296 hr);
297 return false;
298 }
299
300 D3DSURFACE_DESC renderTargetDesc;
301 m_RenderTarget->GetDesc(&renderTargetDesc);
302
303 m_DisplayWidth = renderTargetDesc.Width;
304 m_DisplayHeight = renderTargetDesc.Height;
305
306 if (!(m_DeviceQuirks & DXVA2_QUIRK_NO_VP)) {
307 hr = DXVA2CreateVideoService(m_Device.Get(), IID_IDirectXVideoProcessorService, &m_ProcService);
308
309 if (FAILED(hr)) {
310 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
311 "DXVA2CreateVideoService(IID_IDirectXVideoProcessorService) failed: %x",
312 hr);
313 return false;
314 }
315
316 DXVA2_VideoProcessorCaps caps;
317 hr = m_ProcService->GetVideoProcessorCaps(DXVA2_VideoProcProgressiveDevice, &m_Desc, renderTargetDesc.Format, &caps);
318 if (FAILED(hr)) {
319 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
320 "GetVideoProcessorCaps() failed for DXVA2_VideoProcProgressiveDevice: %x",
321 hr);
322 return false;
323 }
324
325 if (!(caps.DeviceCaps & DXVA2_VPDev_HardwareDevice)) {
326 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
327 "DXVA2_VideoProcProgressiveDevice is not hardware: %x",
328 caps.DeviceCaps);
329 return false;
330 }
331 else if (!(caps.VideoProcessorOperations & DXVA2_VideoProcess_YUV2RGB) &&
332 !(caps.VideoProcessorOperations & DXVA2_VideoProcess_YUV2RGBExtended)) {
333 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
334 "DXVA2_VideoProcProgressiveDevice can't convert YUV2RGB: %x",
335 caps.VideoProcessorOperations);
336 return false;
337 }
338 else if (!(caps.VideoProcessorOperations & DXVA2_VideoProcess_StretchX) ||
339 !(caps.VideoProcessorOperations & DXVA2_VideoProcess_StretchY)) {
340 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
341 "DXVA2_VideoProcProgressiveDevice can't stretch video: %x",
342 caps.VideoProcessorOperations);
343 return false;
344 }
345
346 if (caps.DeviceCaps & DXVA2_VPDev_EmulatedDXVA1) {
347 // DXVA2 over DXVA1 may have bad performance
348 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
349 "DXVA2_VideoProcProgressiveDevice is DXVA1");
350 }
351
352 m_ProcService->GetProcAmpRange(DXVA2_VideoProcProgressiveDevice, &m_Desc, renderTargetDesc.Format, DXVA2_ProcAmp_Brightness, &m_BrightnessRange);
353 m_ProcService->GetProcAmpRange(DXVA2_VideoProcProgressiveDevice, &m_Desc, renderTargetDesc.Format, DXVA2_ProcAmp_Contrast, &m_ContrastRange);
354 m_ProcService->GetProcAmpRange(DXVA2_VideoProcProgressiveDevice, &m_Desc, renderTargetDesc.Format, DXVA2_ProcAmp_Hue, &m_HueRange);
355 m_ProcService->GetProcAmpRange(DXVA2_VideoProcProgressiveDevice, &m_Desc, renderTargetDesc.Format, DXVA2_ProcAmp_Saturation, &m_SaturationRange);
356
357 hr = m_ProcService->CreateVideoProcessor(DXVA2_VideoProcProgressiveDevice, &m_Desc, renderTargetDesc.Format, 0, &m_Processor);
358 if (FAILED(hr)) {
359 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
360 "CreateVideoProcessor() failed for DXVA2_VideoProcProgressiveDevice: %x",
361 hr);
362 return false;
363 }
364 }
365
366 m_Device->SetRenderState(D3DRS_ZENABLE, D3DZB_FALSE);
367 m_Device->SetRenderState(D3DRS_CULLMODE, D3DCULL_NONE);
368 m_Device->SetRenderState(D3DRS_LIGHTING, FALSE);
369
370 m_Device->SetTextureStageState(0, D3DTSS_COLOROP, D3DTOP_MODULATE);
371 m_Device->SetTextureStageState(0, D3DTSS_COLORARG1, D3DTA_TEXTURE);
372 m_Device->SetTextureStageState(0, D3DTSS_COLORARG2, D3DTA_DIFFUSE);
373
374 m_Device->SetSamplerState(0, D3DSAMP_MAGFILTER, D3DTEXF_LINEAR);
375 m_Device->SetSamplerState(0, D3DSAMP_MINFILTER, D3DTEXF_LINEAR);
376
377 m_Device->SetRenderState(D3DRS_ALPHABLENDENABLE, TRUE);
378 m_Device->SetRenderState(D3DRS_SRCBLEND, D3DBLEND_SRCALPHA);
379 m_Device->SetRenderState(D3DRS_DESTBLEND, D3DBLEND_INVSRCALPHA);
380
381 m_Device->SetFVF(D3DFVF_XYZRHW | D3DFVF_TEX1);
382
383 return true;
384}

386bool DXVA2Renderer::initializeQuirksForAdapter(IDirect3D9Ex* d3d9ex, int adapterIndex)
387{
388 HRESULT hr;
389
390 SDL_assert(m_DeviceQuirks == 0);
391 SDL_assert(!m_Device);
392
393 {
394 bool ok;
395
396 m_DeviceQuirks = qEnvironmentVariableIntValue("DXVA2_QUIRK_FLAGS", &ok);
397 if (ok) {
398 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
399 "Using DXVA2 quirk override: 0x%x",
400 m_DeviceQuirks);
401 return true;
402 }
403 }
404
405 UINT adapterCount = d3d9ex->GetAdapterCount();
406 if (adapterCount > 1) {
407 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
408 "Detected multi-GPU system with %d GPUs",
409 adapterCount);
410 m_DeviceQuirks |= DXVA2_QUIRK_MULTI_GPU;
411 }
412
413 D3DCAPS9 caps;
414 hr = d3d9ex->GetDeviceCaps(adapterIndex, D3DDEVTYPE_HAL, &caps);
415 if (SUCCEEDED(hr)) {
416 D3DADAPTER_IDENTIFIER9 id;
417
418 hr = d3d9ex->GetAdapterIdentifier(adapterIndex, 0, &id);
419 if (SUCCEEDED(hr)) {
420 if (id.VendorId == 0x8086) {
421 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
422 "Avoiding IDirectXVideoProcessor API on Intel GPU");
423
424 // On Intel GPUs, we can get unwanted video "enhancements" due to post-processing
425 // effects that the GPU driver forces on us. In many cases, this makes the video
426 // actually look worse. We can avoid these by using StretchRect() instead on these
427 // platforms.
428 m_DeviceQuirks |= DXVA2_QUIRK_NO_VP;
429 }
430 else if (id.VendorId == 0x4d4f4351) { // QCOM in ASCII
431 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
432 "Avoiding IDirectXVideoProcessor API on Qualcomm GPU");
433
434 // On Qualcomm GPUs (all D3D9on12 GPUs?), the scaling quality of VideoProcessBlt()
435 // is absolutely horrible. StretchRect() is much much better.
436 m_DeviceQuirks |= DXVA2_QUIRK_NO_VP;
437 }
438 else if (id.VendorId == 0x1002 &&
439 (id.DriverVersion.HighPart > 0x1E0000 ||
440 (id.DriverVersion.HighPart == 0x1E0000 && HIWORD(id.DriverVersion.LowPart) >= 14000))) { // AMD 21.12.1 or later
441 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
442 "Using DestFormat quirk for recent AMD GPU driver");
443
444 // AMD's GPU driver doesn't correctly handle color range conversion.
445 //
446 // This used to just work because we used Rec 709 Limited which happened to be what AMD's
447 // driver defaulted to. However, AMD's driver behavior changed to default to Rec 709 Full
448 // in the 21.12.1 driver, so we must adapt to that.
449 //
450 // 30.0.13037.1003 - 21.11.3 - Limited
451 // 30.0.14011.3017 - 21.12.1 - Full
452 //
453 // To sort out this mess, we will use a quirk to tell us to populate DestFormat for AMD.
454 // For other GPUs, we'll avoid populating it as was our previous behavior.
455 m_DeviceQuirks |= DXVA2_QUIRK_SET_DEST_FORMAT;
456 }
457 }
458
459 return true;
460 }
461 else {
462 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
463 "GetDeviceCaps() failed: %x", hr);
464 }
465
466 return false;
467}

469bool DXVA2Renderer::isDecoderBlacklisted()
470{
471 ComPtr<IDirect3D9> d3d9;
472 HRESULT hr;
473 bool result = false;
474
475 if (qgetenv("DXVA2_DISABLE_DECODER_BLACKLIST") == "1") {
476 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
477 "DXVA2 decoder blacklist is disabled");
478 return false;
479 }
480
481 hr = m_Device->GetDirect3D(&d3d9);
482 if (SUCCEEDED(hr)) {
483 D3DCAPS9 caps;
484
485 hr = m_Device->GetDeviceCaps(&caps);
486 if (SUCCEEDED(hr)) {
487 D3DADAPTER_IDENTIFIER9 id;
488
489 hr = d3d9->GetAdapterIdentifier(caps.AdapterOrdinal, 0, &id);
490 if (SUCCEEDED(hr)) {
491 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
492 "Detected GPU: %s (%x:%x)",
493 id.Description,
494 id.VendorId,
495 id.DeviceId);
496 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
497 "GPU driver: %s %d.%d.%d.%d",
498 id.Driver,
499 HIWORD(id.DriverVersion.HighPart),
500 LOWORD(id.DriverVersion.HighPart),
501 HIWORD(id.DriverVersion.LowPart),
502 LOWORD(id.DriverVersion.LowPart));
503
504 if (DXUtil::isFormatHybridDecodedByHardware(m_VideoFormat, id.VendorId, id.DeviceId)) {
505 result = true;
506 }
507 // Intel drivers from before late-2017 had a bug that caused some strange artifacts
508 // when decoding HEVC. Avoid HEVC on drivers prior to build 4836 which I confirmed
509 // is not affected on my Intel HD 515. Also account for the driver version rollover
510 // that happened with the 101.1069 series.
511 // https://github.com/moonlight-stream/moonlight-qt/issues/32
512 // https://www.intel.com/content/www/us/en/support/articles/000005654/graphics-drivers.html
513 else if (id.VendorId == 0x8086 && HIWORD(id.DriverVersion.LowPart) < 100 && LOWORD(id.DriverVersion.LowPart) < 4836) {
514 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
515 "Detected buggy Intel GPU driver installed. Update your Intel GPU driver to enable HEVC!");
516 result = (m_VideoFormat & VIDEO_FORMAT_MASK_H265) != 0;
517 }
518 }
519 else {
520 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
521 "GetAdapterIdentifier() failed: %x", hr);
522 }
523 }
524 else {
525 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
526 "GetDeviceCaps() failed: %x", hr);
527 }
528 }
529 else {
530 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
531 "GetDirect3D() failed: %x", hr);
532 }
533
534 if (result) {
535 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
536 "GPU decoding for format %x is blocked due to hardware limitations",
537 m_VideoFormat);
538 }
539
540 return result;
541}

543bool DXVA2Renderer::initializeDevice(SDL_Window* window, bool enableVsync)
544{
545 SDL_SysWMinfo info;
546
547 SDL_VERSION(&info.version);
548 SDL_GetWindowWMInfo(window, &info);
549
550 ComPtr<IDirect3D9Ex> d3d9ex;
551 HRESULT hr = Direct3DCreate9Ex(D3D_SDK_VERSION, &d3d9ex);
552 if (FAILED(hr)) {
553 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
554 "Direct3DCreate9Ex() failed: %x",
555 hr);
556 return false;
557 }
558
559 int adapterIndex = SDL_Direct3D9GetAdapterIndex(SDL_GetWindowDisplayIndex(window));
560 Uint32 windowFlags = SDL_GetWindowFlags(window);
561
562 // Initialize quirks *before* calling CreateDeviceEx() to allow our below
563 // logic to avoid a hang with NahimicOSD.dll's broken full-screen handling.
564 if (!initializeQuirksForAdapter(d3d9ex.Get(), adapterIndex)) {
565 return false;
566 }
567
568 D3DCAPS9 deviceCaps;
569 d3d9ex->GetDeviceCaps(adapterIndex, D3DDEVTYPE_HAL, &deviceCaps);
570
571 D3DDISPLAYMODEEX currentMode;
572 currentMode.Size = sizeof(currentMode);
573 d3d9ex->GetAdapterDisplayModeEx(adapterIndex, &currentMode, nullptr);
574
575 D3DPRESENT_PARAMETERS d3dpp = {};
576 d3dpp.hDeviceWindow = info.info.win.window;
577 d3dpp.Flags = D3DPRESENTFLAG_VIDEO;
578
579 if (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) {
580 // Verify 10-bit A2R10G10B10 color support. This is only available
581 // as a display format in full-screen exclusive mode on DX9.
582 hr = d3d9ex->CheckDeviceType(adapterIndex,
583 D3DDEVTYPE_HAL,
584 D3DFMT_A2R10G10B10,
585 D3DFMT_A2R10G10B10,
586 FALSE);
587 if (FAILED(hr)) {
588 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
589 "GPU/driver doesn't support A2R10G10B10");
590 return false;
591 }
592 }
593
594 if ((windowFlags & SDL_WINDOW_FULLSCREEN_DESKTOP) == SDL_WINDOW_FULLSCREEN) {
595 d3dpp.Windowed = false;
596 d3dpp.BackBufferWidth = currentMode.Width;
597 d3dpp.BackBufferHeight = currentMode.Height;
598 d3dpp.FullScreen_RefreshRateInHz = currentMode.RefreshRate;
599
600 if (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) {
601 d3dpp.BackBufferFormat = currentMode.Format = D3DFMT_A2R10G10B10;
602 }
603 else {
604 d3dpp.BackBufferFormat = currentMode.Format;
605 }
606 }
607 else {
608 d3dpp.Windowed = true;
609 d3dpp.BackBufferFormat = D3DFMT_UNKNOWN;
610
611 SDL_GetWindowSize(window, (int*)&d3dpp.BackBufferWidth, (int*)&d3dpp.BackBufferHeight);
612 }
613
614 BOOL dwmEnabled;
615 DwmIsCompositionEnabled(&dwmEnabled);
616 if (d3dpp.Windowed && dwmEnabled) {
617 // If composition enabled, disable v-sync and let DWM manage things
618 // to reduce latency by avoiding double v-syncing.
619 d3dpp.PresentationInterval = D3DPRESENT_INTERVAL_IMMEDIATE;
620 d3dpp.SwapEffect = D3DSWAPEFFECT_FLIPEX;
621
622 if (enableVsync) {
623 // D3DSWAPEFFECT_FLIPEX requires at least 3 back buffers to allow us to
624 // continue while DWM is waiting to render the surface to the display.
625 // NVIDIA seems to be fine with 2, but AMD needs 3 to perform well.
626 d3dpp.BackBufferCount = 3;
627 }
628 else {
629 // With V-sync off, we need 1 more back buffer to render to while the
630 // driver/DWM are holding the others.
631 d3dpp.BackBufferCount = 4;
632 }
633
634 m_BlockingPresent = false;
635
636 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
637 "Windowed mode with DWM running");
638 }
639 else if (enableVsync) {
640 // Uncomposited desktop or full-screen exclusive mode with V-sync enabled
641 // We will enable V-sync in this scenario to avoid tearing.
642 d3dpp.PresentationInterval = D3DPRESENT_INTERVAL_ONE;
643 d3dpp.SwapEffect = D3DSWAPEFFECT_DISCARD;
644 d3dpp.BackBufferCount = 1;
645 m_BlockingPresent = true;
646
647 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
648 "V-Sync enabled");
649 }
650 else {
651 // Uncomposited desktop or full-screen exclusive mode with V-sync disabled
652 // We will allowing tearing for lowest latency.
653 d3dpp.PresentationInterval = D3DPRESENT_INTERVAL_IMMEDIATE;
654 d3dpp.SwapEffect = D3DSWAPEFFECT_DISCARD;
655 d3dpp.BackBufferCount = 1;
656 m_BlockingPresent = false;
657
658 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
659 "V-Sync disabled in tearing mode");
660 }
661
662 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
663 "Windowed: %d | Present Interval: %x",
664 d3dpp.Windowed, d3dpp.PresentationInterval);
665
666 // FFmpeg requires this attribute for doing asynchronous decoding
667 // in a separate thread with this device.
668 int deviceFlags = D3DCREATE_MULTITHREADED;
669
670 if (deviceCaps.DevCaps & D3DDEVCAPS_HWTRANSFORMANDLIGHT) {
671 deviceFlags |= D3DCREATE_HARDWARE_VERTEXPROCESSING;
672 }
673 else {
674 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
675 "No hardware vertex processing support!");
676 deviceFlags |= D3DCREATE_SOFTWARE_VERTEXPROCESSING;
677 }
678
679 hr = d3d9ex->CreateDeviceEx(adapterIndex,
680 D3DDEVTYPE_HAL,
681 d3dpp.hDeviceWindow,
682 deviceFlags,
683 &d3dpp,
684 d3dpp.Windowed ? nullptr : &currentMode,
685 &m_Device);
686 if (FAILED(hr)) {
687 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
688 "CreateDeviceEx() failed: %x",
689 hr);
690 return false;
691 }
692
693 // We must not call this for flip swapchains. It will counterintuitively
694 // increase latency by forcing our Present() to block on DWM even when
695 // using D3DPRESENT_INTERVAL_IMMEDIATE.
696 if (d3dpp.SwapEffect != D3DSWAPEFFECT_FLIPEX) {
697 hr = m_Device->SetMaximumFrameLatency(1);
698 if (FAILED(hr)) {
699 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
700 "SetMaximumFrameLatency() failed: %x",
701 hr);
702 return false;
703 }
704 }
705
706 return true;
707}

709bool DXVA2Renderer::initialize(PDECODER_PARAMETERS params)
710{
711 if (qgetenv("DXVA2_ENABLED") == "0") {
712 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
713 "DXVA2 is disabled by environment variable");
714 return false;
715 }
716 else if ((params->videoFormat & VIDEO_FORMAT_MASK_10BIT) && m_DecoderSelectionPass == 0) {
717 // Avoid using DXVA2 for HDR10. While it can render 10-bit color, it doesn't support
718 // the HDR colorspace and HDR display metadata required to enable HDR mode properly.
719 return false;
720 }
721 else if (params->videoFormat & VIDEO_FORMAT_MASK_YUV444) {
722 // It's theoretically possible to use YUV444 with D3D9, but probably not worth actually
723 // implementing because any YUV444-capable hardware supports D3D11 or Vulkan.
724 return false;
725 }
726#ifndef Q_PROCESSOR_X86
727 else if (qgetenv("DXVA2_ENABLED") != "1" && m_DecoderSelectionPass == 0) {
728 SDL_LogInfo(SDL_LOG_CATEGORY_APPLICATION,
729 "DXVA2 is disabled by default on ARM64. Set DXVA2_ENABLED=1 to override.");
730 return false;
731 }
732#endif
733
734 m_VideoFormat = params->videoFormat;
735 m_VideoWidth = params->width;
736 m_VideoHeight = params->height;
737
738 RtlZeroMemory(&m_Desc, sizeof(m_Desc));
739
740 int alignment;
741
742 // HEVC and AV1 using DXVA requires 128 pixel alignment, however this causes Intel GPUs
743 // using StretchRect() and HEVC to render draw a translucent green line at the top of
744 // the screen in full-screen mode at 720p/1080p unless we use 32 pixel alignment.
745 // This appears to work without issues on AMD and Nvidia GPUs too, so we will
746 // do it unconditionally for now.
747 // https://github.com/FFmpeg/FFmpeg/blob/a234e5cd80224c95a205c1f3e297d8c04a1374c3/libavcodec/dxva2.c#L609-L616
748 if (m_VideoFormat & VIDEO_FORMAT_MASK_H265) {
749 alignment = 32;
750 }
751 else if (m_VideoFormat & VIDEO_FORMAT_MASK_H264) {
752 alignment = 16;
753 }
754 else {
755 alignment = 128;
756 }
757
758 m_Desc.SampleWidth = FFALIGN(m_VideoWidth, alignment);
759 m_Desc.SampleHeight = FFALIGN(m_VideoHeight, alignment);
760 m_Desc.SampleFormat.VideoChromaSubsampling = DXVA2_VideoChromaSubsampling_Unknown;
761 m_Desc.SampleFormat.NominalRange = DXVA2_NominalRange_Unknown;
762 m_Desc.SampleFormat.VideoTransferMatrix = DXVA2_VideoTransferMatrix_Unknown;
763 m_Desc.SampleFormat.VideoLighting = DXVA2_VideoLighting_Unknown;
764 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_Unknown;
765 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_Unknown;
766 m_Desc.SampleFormat.SampleFormat = DXVA2_SampleProgressiveFrame;
767
768 if (m_VideoFormat & VIDEO_FORMAT_MASK_YUV444) {
769 m_Desc.Format = (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) ?
770 (D3DFORMAT)MAKEFOURCC('Y','4','1','0') :
771 (D3DFORMAT)MAKEFOURCC('A','Y','U','V');
772 }
773 else {
774 m_Desc.Format = (m_VideoFormat & VIDEO_FORMAT_MASK_10BIT) ?
775 (D3DFORMAT)MAKEFOURCC('P','0','1','0') :
776 (D3DFORMAT)MAKEFOURCC('N','V','1','2');
777 }
778
779 if (!initializeDevice(params->window, params->enableVsync)) {
780 return false;
781 }
782
783 if (!initializeDecoder()) {
784 return false;
785 }
786
787 if (!initializeRenderer()) {
788 return false;
789 }
790
791 // For some reason, using Direct3D9Ex breaks this with multi-monitor setups.
792 // When focus is lost, the window is minimized then immediately restored without
793 // input focus. This glitches out the renderer and a bunch of other stuff.
794 // Direct3D9Ex itself seems to have this minimize on focus loss behavior on its
795 // own, so just disable SDL's handling of the focus loss event.
796 SDL_SetHintWithPriority(SDL_HINT_VIDEO_MINIMIZE_ON_FOCUS_LOSS, "0", SDL_HINT_OVERRIDE);
797
798 return true;
799}

801void DXVA2Renderer::notifyOverlayUpdated(Overlay::OverlayType type)
802{
803 HRESULT hr;
804
805 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
806 bool overlayEnabled = Session::get()->getOverlayManager().isOverlayEnabled(type);
807 if (newSurface == nullptr && overlayEnabled) {
808 // The overlay is enabled and there is no new surface. Leave the old texture alone.
809 return;
810 }
811
812 SDL_AtomicLock(&m_OverlayLock);
813 ComPtr<IDirect3DTexture9> oldTexture = std::move(m_OverlayTextures[type]);
814 ComPtr<IDirect3DVertexBuffer9> oldVertexBuffer = std::move(m_OverlayVertexBuffers[type]);
815 SDL_AtomicUnlock(&m_OverlayLock);
816
817 // If the overlay is disabled, we're done
818 if (!overlayEnabled) {
819 SDL_FreeSurface(newSurface);
820 return;
821 }
822
823 // Create a dynamic texture to populate with our pixel data
824 SDL_assert(!SDL_MUSTLOCK(newSurface));
825 ComPtr<IDirect3DTexture9> newTexture;
826 hr = m_Device->CreateTexture(newSurface->w,
827 newSurface->h,
828 1,
829 D3DUSAGE_DYNAMIC,
830 D3DFMT_A8R8G8B8,
831 D3DPOOL_DEFAULT,
832 &newTexture,
833 nullptr);
834 if (FAILED(hr)) {
835 SDL_FreeSurface(newSurface);
836 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
837 "CreateTexture() failed: %x",
838 hr);
839 return;
840 }
841
842 D3DLOCKED_RECT lockedRect;
843 hr = newTexture->LockRect(0, &lockedRect, nullptr, D3DLOCK_DISCARD);
844 if (FAILED(hr)) {
845 SDL_FreeSurface(newSurface);
846 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
847 "IDirect3DTexture9::LockRect() failed: %x",
848 hr);
849 return;
850 }
851
852 // Copy (and convert, if necessary) the surface pixels to the texture
853 SDL_ConvertPixels(newSurface->w, newSurface->h, newSurface->format->format, newSurface->pixels,
854 newSurface->pitch, SDL_PIXELFORMAT_ARGB8888, lockedRect.pBits, lockedRect.Pitch);
855
856 newTexture->UnlockRect(0);
857
858 SDL_FRect renderRect = {};
859
860 if (type == Overlay::OverlayStatusUpdate) {
861 // Bottom Left
862 renderRect.x = 0;
863 renderRect.y = m_DisplayHeight - newSurface->h;
864 }
865 else if (type == Overlay::OverlayDebug) {
866 // Top left
867 renderRect.x = 0;
868 renderRect.y = 0;
869 }
870
871 renderRect.w = newSurface->w;
872 renderRect.h = newSurface->h;
873
874 // The surface is no longer required
875 SDL_FreeSurface(newSurface);
876 newSurface = nullptr;
877
878 VERTEX verts[] =
879 {
880 {renderRect.x, renderRect.y, 0, 1, 0, 0},
881 {renderRect.x, renderRect.y+renderRect.h, 0, 1, 0, 1},
882 {renderRect.x+renderRect.w, renderRect.y+renderRect.h, 0, 1, 1, 1},
883 {renderRect.x+renderRect.w, renderRect.y, 0, 1, 1, 0}
884 };
885
886 ComPtr<IDirect3DVertexBuffer9> newVertexBuffer;
887 hr = m_Device->CreateVertexBuffer(sizeof(verts),
888 D3DUSAGE_WRITEONLY,
889 D3DFVF_XYZRHW | D3DFVF_TEX1,
890 D3DPOOL_DEFAULT,
891 &newVertexBuffer,
892 nullptr);
893 if (FAILED(hr)) {
894 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
895 "CreateVertexBuffer() failed: %x",
896 hr);
897 return;
898 }
899
900 PVOID targetVertexBuffer = nullptr;
901 hr = newVertexBuffer->Lock(0, 0, &targetVertexBuffer, 0);
902 if (FAILED(hr)) {
903 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
904 "IDirect3DVertexBuffer9::Lock() failed: %x",
905 hr);
906 return;
907 }
908
909 RtlCopyMemory(targetVertexBuffer, verts, sizeof(verts));
910
911 newVertexBuffer->Unlock();
912
913 SDL_AtomicLock(&m_OverlayLock);
914 m_OverlayVertexBuffers[type] = std::move(newVertexBuffer);
915 m_OverlayTextures[type] = std::move(newTexture);
916 SDL_AtomicUnlock(&m_OverlayLock);
917}

919void DXVA2Renderer::renderOverlay(Overlay::OverlayType type)
920{
921 HRESULT hr;
922
923 if (!Session::get()->getOverlayManager().isOverlayEnabled(type)) {
924 return;
925 }
926
927 // If the overlay is being updated, just skip rendering it this frame
928 if (!SDL_AtomicTryLock(&m_OverlayLock)) {
929 return;
930 }
931
932 // Reference these objects so they don't immediately go away if the
933 // overlay update thread tries to release them.
934 ComPtr<IDirect3DTexture9> overlayTexture = m_OverlayTextures[type];
935 ComPtr<IDirect3DVertexBuffer9> overlayVertexBuffer = m_OverlayVertexBuffers[type];
936 SDL_AtomicUnlock(&m_OverlayLock);
937
938 if (overlayTexture == nullptr) {
939 return;
940 }
941
942 hr = m_Device->SetTexture(0, overlayTexture.Get());
943 if (FAILED(hr)) {
944 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
945 "SetTexture() failed: %x",
946 hr);
947 return;
948 }
949
950 hr = m_Device->SetStreamSource(0, overlayVertexBuffer.Get(), 0, sizeof(VERTEX));
951 if (FAILED(hr)) {
952 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
953 "SetStreamSource() failed: %x",
954 hr);
955 return;
956 }
957
958 hr = m_Device->DrawPrimitive(D3DPT_TRIANGLEFAN, 0, 2);
959 if (FAILED(hr)) {
960 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
961 "DrawPrimitive() failed: %x",
962 hr);
963 return;
964 }
965}

988void DXVA2Renderer::renderFrame(AVFrame *frame)
989{
990 IDirect3DSurface9* surface = reinterpret_cast<IDirect3DSurface9*>(frame->data[3]);
991 HRESULT hr;
992
993 m_Desc.SampleFormat.NominalRange = isFrameFullRange(frame) ? DXVA2_NominalRange_0_255 : DXVA2_NominalRange_16_235;
994
995 switch (frame->color_primaries) {
996 case AVCOL_PRI_BT709:
997 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_BT709;
998 break;
999 case AVCOL_PRI_BT470M:
1000 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_BT470_2_SysM;
1001 break;
1002 case AVCOL_PRI_BT470BG:
1003 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_BT470_2_SysBG;
1004 break;
1005 case AVCOL_PRI_SMPTE170M:
1006 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_SMPTE170M;
1007 break;
1008 case AVCOL_PRI_SMPTE240M:
1009 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_SMPTE240M;
1010 break;
1011 default:
1012 m_Desc.SampleFormat.VideoPrimaries = DXVA2_VideoPrimaries_Unknown;
1013 break;
1014 }
1015
1016 switch (frame->color_trc) {
1017 case AVCOL_TRC_SMPTE170M:
1018 case AVCOL_TRC_BT709:
1019 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_709;
1020 break;
1021 case AVCOL_TRC_LINEAR:
1022 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_10;
1023 break;
1024 case AVCOL_TRC_GAMMA22:
1025 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_22;
1026 break;
1027 case AVCOL_TRC_GAMMA28:
1028 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_28;
1029 break;
1030 case AVCOL_TRC_SMPTE240M:
1031 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_240M;
1032 break;
1033 case AVCOL_TRC_IEC61966_2_1:
1034 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_sRGB;
1035 break;
1036 default:
1037 m_Desc.SampleFormat.VideoTransferFunction = DXVA2_VideoTransFunc_Unknown;
1038 break;
1039 }
1040
1041 switch (getFrameColorspace(frame)) {
1042 case COLORSPACE_REC_709:
1043 m_Desc.SampleFormat.VideoTransferMatrix = DXVA2_VideoTransferMatrix_BT709;
1044 break;
1045 case COLORSPACE_REC_601:
1046 m_Desc.SampleFormat.VideoTransferMatrix = DXVA2_VideoTransferMatrix_BT601;
1047 break;
1048 default:
1049 m_Desc.SampleFormat.VideoTransferMatrix = DXVA2_VideoTransferMatrix_Unknown;
1050 break;
1051 }
1052
1053 switch (frame->chroma_location) {
1054 case AVCHROMA_LOC_LEFT:
1055 m_Desc.SampleFormat.VideoChromaSubsampling = DXVA2_VideoChromaSubsampling_Horizontally_Cosited |
1056 DXVA2_VideoChromaSubsampling_Vertically_AlignedChromaPlanes |
1057 DXVA2_VideoChromaSubsampling_ProgressiveChroma;
1058 break;
1059 case AVCHROMA_LOC_CENTER:
1060 m_Desc.SampleFormat.VideoChromaSubsampling = DXVA2_VideoChromaSubsampling_Vertically_AlignedChromaPlanes |
1061 DXVA2_VideoChromaSubsampling_ProgressiveChroma;
1062 break;
1063 case AVCHROMA_LOC_TOPLEFT:
1064 m_Desc.SampleFormat.VideoChromaSubsampling = DXVA2_VideoChromaSubsampling_Horizontally_Cosited |
1065 DXVA2_VideoChromaSubsampling_Vertically_Cosited |
1066 DXVA2_VideoChromaSubsampling_ProgressiveChroma;
1067 break;
1068 default:
1069 m_Desc.SampleFormat.VideoChromaSubsampling = DXVA2_VideoChromaSubsampling_Unknown;
1070 break;
1071 }
1072
1073 DXVA2_VideoSample sample = {};
1074 sample.Start = m_FrameIndex;
1075 sample.End = m_FrameIndex + 1;
1076 sample.SrcSurface = surface;
1077 sample.SrcRect.right = m_VideoWidth;
1078 sample.SrcRect.bottom = m_VideoHeight;
1079 sample.SampleFormat = m_Desc.SampleFormat;
1080 sample.PlanarAlpha = DXVA2_Fixed32OpaqueAlpha();
1081
1082 // Center in frame and preserve aspect ratio
1083 SDL_Rect src, dst;
1084 src.x = src.y = 0;
1085 src.w = m_VideoWidth;
1086 src.h = m_VideoHeight;
1087 dst.x = dst.y = 0;
1088 dst.w = m_DisplayWidth;
1089 dst.h = m_DisplayHeight;
1090
1091 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
1092
1093 sample.DstRect.left = dst.x;
1094 sample.DstRect.right = dst.x + dst.w;
1095 sample.DstRect.top = dst.y;
1096 sample.DstRect.bottom = dst.y + dst.h;
1097
1098 DXVA2_VideoProcessBltParams bltParams = {};
1099
1100 bltParams.TargetFrame = m_FrameIndex++;
1101 bltParams.TargetRect = sample.DstRect;
1102 bltParams.BackgroundColor.Alpha = 0xFFFF;
1103
1104 if (m_DeviceQuirks & DXVA2_QUIRK_SET_DEST_FORMAT) {
1105 bltParams.DestFormat = m_Desc.SampleFormat;
1106 }
1107 else {
1108 bltParams.DestFormat.SampleFormat = DXVA2_SampleProgressiveFrame;
1109 }
1110
1111 bltParams.ProcAmpValues.Brightness = m_BrightnessRange.DefaultValue;
1112 bltParams.ProcAmpValues.Contrast = m_ContrastRange.DefaultValue;
1113 bltParams.ProcAmpValues.Hue = m_HueRange.DefaultValue;
1114 bltParams.ProcAmpValues.Saturation = m_SaturationRange.DefaultValue;
1115
1116 bltParams.Alpha = DXVA2_Fixed32OpaqueAlpha();
1117
1118 hr = m_Device->Clear(0, nullptr, D3DCLEAR_TARGET, D3DCOLOR_ARGB(255, 0, 0, 0), 0.0f, 0);
1119 if (FAILED(hr)) {
1120 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1121 "Clear() failed: %x",
1122 hr);
1123 SDL_Event event;
1124 event.type = SDL_RENDER_TARGETS_RESET;
1125 SDL_PushEvent(&event);
1126 return;
1127 }
1128
1129 hr = m_Device->BeginScene();
1130 if (FAILED(hr)) {
1131 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1132 "BeginScene() failed: %x",
1133 hr);
1134 SDL_Event event;
1135 event.type = SDL_RENDER_TARGETS_RESET;
1136 SDL_PushEvent(&event);
1137 return;
1138 }
1139
1140 if (m_Processor) {
1141 hr = m_Processor->VideoProcessBlt(m_RenderTarget.Get(), &bltParams, &sample, 1, nullptr);
1142 if (FAILED(hr)) {
1143 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1144 "VideoProcessBlt() failed, falling back to StretchRect(): %x",
1145 hr);
1146 m_Processor.Reset();
1147 }
1148 }
1149
1150 if (!m_Processor) {
1151 // StretchRect() assumes Rec 601 on Intel and Qualcomm GPUs.
1152 SDL_assert(m_Desc.SampleFormat.VideoTransferMatrix == DXVA2_VideoTransferMatrix_BT601);
1153
1154 // This function doesn't trigger any of Intel's garbage video "enhancements"
1155 hr = m_Device->StretchRect(surface, &sample.SrcRect, m_RenderTarget.Get(), &sample.DstRect, D3DTEXF_NONE);
1156 if (FAILED(hr)) {
1157 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1158 "StretchRect() failed: %x",
1159 hr);
1160 SDL_Event event;
1161 event.type = SDL_RENDER_TARGETS_RESET;
1162 SDL_PushEvent(&event);
1163 return;
1164 }
1165 }
1166
1167 // Render overlays on top of the video stream
1168 for (int i = 0; i < Overlay::OverlayMax; i++) {
1169 renderOverlay((Overlay::OverlayType)i);
1170 }
1171
1172 hr = m_Device->EndScene();
1173 if (FAILED(hr)) {
1174 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1175 "EndScene() failed: %x",
1176 hr);
1177 SDL_Event event;
1178 event.type = SDL_RENDER_TARGETS_RESET;
1179 SDL_PushEvent(&event);
1180 return;
1181 }
1182
1183 do {
1184 // Use D3DPRESENT_DONOTWAIT if present may block in order to avoid holding the giant
1185 // lock around this D3D device for excessive lengths of time (blocking concurrent decoding tasks).
1186 hr = m_Device->PresentEx(nullptr, nullptr, nullptr, nullptr, m_BlockingPresent ? D3DPRESENT_DONOTWAIT : 0);
1187 if (hr == D3DERR_WASSTILLDRAWING) {
1188 SDL_Delay(1);
1189 }
1190 } while (hr == D3DERR_WASSTILLDRAWING);
1191 if (FAILED(hr)) {
1192 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
1193 "PresentEx() failed: %x",
1194 hr);
1195 SDL_Event event;
1196 event.type = SDL_RENDER_TARGETS_RESET;
1197 SDL_PushEvent(&event);
1198 return;
1199 }
1200}
