94bool SdlRenderer::isPixelFormatSupported(int videoFormat, AVPixelFormat pixelFormat)
95{
96 if (videoFormat & (VIDEO_FORMAT_MASK_10BIT | VIDEO_FORMAT_MASK_YUV444)) {
97 // SDL2 can't natively handle textures with these formats, but we can perform
98 // conversion on the CPU using swscale then upload them as an RGB texture.
99 const AVPixFmtDescriptor* formatDesc = av_pix_fmt_desc_get(pixelFormat);
100 if (!formatDesc) {
101 SDL_assert(formatDesc);
102 return false;
103 }
104
105 const int expectedPixelDepth = (videoFormat & VIDEO_FORMAT_MASK_10BIT) ? 10 : 8;
106 const int expectedLog2ChromaW = (videoFormat & VIDEO_FORMAT_MASK_YUV444) ? 0 : 1;
107 const int expectedLog2ChromaH = (videoFormat & VIDEO_FORMAT_MASK_YUV444) ? 0 : 1;
108
109 return formatDesc->comp[0].depth == expectedPixelDepth &&
110 formatDesc->log2_chroma_w == expectedLog2ChromaW &&
111 formatDesc->log2_chroma_h == expectedLog2ChromaH;
112 }
113 else {
114 // The formats listed below are natively supported by SDL, so it can handle
115 // YUV to RGB conversion on the GPU using pixel shaders.
116 //
117 // Remember to keep this in sync with SdlRenderer::renderFrame()!
118 switch (pixelFormat) {
119 case AV_PIX_FMT_YUV420P:
120 case AV_PIX_FMT_YUVJ420P:
121 case AV_PIX_FMT_NV12:
122 case AV_PIX_FMT_NV21:
123 return true;
124
125 default:
126 return false;
127 }
128 }
129}

131bool SdlRenderer::initialize(PDECODER_PARAMETERS params)
132{
133 Uint32 rendererFlags = SDL_RENDERER_ACCELERATED;
134
135 m_VideoFormat = params->videoFormat;
136 m_SwFrameMapper.setVideoFormat(m_VideoFormat);
137
138 if (params->videoFormat & VIDEO_FORMAT_MASK_10BIT) {
139 // SDL doesn't support rendering HDR yet
140 return false;
141 }
142
143 SDL_SysWMinfo info;
144 SDL_VERSION(&info.version);
145 if (!SDL_GetWindowWMInfo(params->window, &info)) {
146 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
147 "SDL_GetWindowWMInfo() failed: %s",
148 SDL_GetError());
149 return false;
150 }
151
152 // Only set SDL_RENDERER_PRESENTVSYNC if we know we'll get tearing otherwise.
153 // Since we don't use V-Sync to pace our frame rate, we want non-blocking
154 // presents to reduce video latency.
155 switch (info.subsystem) {
156 case SDL_SYSWM_WINDOWS:
157 // DWM is always tear-free except in full-screen exclusive mode
158 if ((SDL_GetWindowFlags(params->window) & SDL_WINDOW_FULLSCREEN_DESKTOP) == SDL_WINDOW_FULLSCREEN) {
159 if (params->enableVsync) {
160 rendererFlags |= SDL_RENDERER_PRESENTVSYNC;
161 }
162 }
163 break;
164 case SDL_SYSWM_WAYLAND:
165 // Wayland is always tear-free in all modes
166 break;
167 default:
168 // For other subsystems, just set SDL_RENDERER_PRESENTVSYNC if asked
169 if (params->enableVsync) {
170 rendererFlags |= SDL_RENDERER_PRESENTVSYNC;
171 }
172 break;
173 }
174
175#ifdef Q_OS_WIN32
176 // We render on a different thread than the main thread which is handling window
177 // messages. Without D3DCREATE_MULTITHREADED, this can cause a deadlock by blocking
178 // on a window message being processed while the main thread is blocked waiting for
179 // the render thread to finish.
180 SDL_SetHintWithPriority(SDL_HINT_RENDER_DIRECT3D_THREADSAFE, "1", SDL_HINT_OVERRIDE);
181#endif
182
183 m_Renderer = SDL_CreateRenderer(params->window, -1, rendererFlags);
184 if (!m_Renderer) {
185 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
186 "SDL_CreateRenderer() failed: %s",
187 SDL_GetError());
188 }
189
190 // SDL_CreateRenderer() can end up having to recreate our window (SDL_RecreateWindow())
191 // to ensure it's compatible with the renderer's OpenGL context. If that happens, we
192 // can get spurious SDL_WINDOWEVENT events that will cause us to (again) recreate our
193 // renderer. This can lead to an infinite to renderer recreation, so discard all
194 // SDL_WINDOWEVENT events after SDL_CreateRenderer().
195 Session* session = Session::get();
196 if (session != nullptr) {
197 // If we get here during a session, we need to synchronize with the event loop
198 // to ensure we don't drop any important events.
199 session->flushWindowEvents();
200 }
201 else {
202 // If we get here prior to the start of a session, just pump and flush ourselves.
203 SDL_PumpEvents();
204 SDL_FlushEvent(SDL_WINDOWEVENT);
205 }
206
207 if (!m_Renderer) {
208 m_InitFailureReason = InitFailureReason::NoSoftwareSupport;
209 return false;
210 }
211
212#ifdef Q_OS_WIN32
213 // For some reason, using Direct3D9Ex breaks this with multi-monitor setups.
214 // When focus is lost, the window is minimized then immediately restored without
215 // input focus. This glitches out the renderer and a bunch of other stuff.
216 // Direct3D9Ex itself seems to have this minimize on focus loss behavior on its
217 // own, so just disable SDL's handling of the focus loss event.
218 SDL_SetHintWithPriority(SDL_HINT_VIDEO_MINIMIZE_ON_FOCUS_LOSS, "0", SDL_HINT_OVERRIDE);
219#endif
220
221 return true;
222}

224void SdlRenderer::renderOverlay(Overlay::OverlayType type)
225{
226 if (Session::get()->getOverlayManager().isOverlayEnabled(type)) {
227 // If a new surface has been created for updated overlay data, convert it into a texture.
228 // NB: We have to do this conversion at render-time because we can only interact
229 // with the renderer on a single thread.
230 SDL_Surface* newSurface = Session::get()->getOverlayManager().getUpdatedOverlaySurface(type);
231 if (newSurface != nullptr) {
232 if (m_OverlayTextures[type] != nullptr) {
233 SDL_DestroyTexture(m_OverlayTextures[type]);
234 }
235
236 if (type == Overlay::OverlayStatusUpdate) {
237 // Bottom Left
238 SDL_Rect viewportRect;
239 SDL_RenderGetViewport(m_Renderer, &viewportRect);
240 m_OverlayRects[type].x = 0;
241 m_OverlayRects[type].y = viewportRect.h - newSurface->h;
242 }
243 else if (type == Overlay::OverlayDebug) {
244 // Top left
245 m_OverlayRects[type].x = 0;
246 m_OverlayRects[type].y = 0;
247 }
248
249 m_OverlayRects[type].w = newSurface->w;
250 m_OverlayRects[type].h = newSurface->h;
251
252 m_OverlayTextures[type] = SDL_CreateTextureFromSurface(m_Renderer, newSurface);
253 SDL_FreeSurface(newSurface);
254 }
255
256 // If we have an overlay texture, render it too
257 if (m_OverlayTextures[type] != nullptr) {
258 SDL_RenderCopy(m_Renderer, m_OverlayTextures[type], nullptr, &m_OverlayRects[type]);
259 }
260 }

268void SdlRenderer::renderFrame(AVFrame* frame)
269{
270 int err;
271 AVFrame* swFrame = nullptr;
272
273 if (frame->hw_frames_ctx != nullptr && frame->format != AV_PIX_FMT_CUDA) {
274#ifdef HAVE_CUDA
275ReadbackRetry:
276#endif
277 // If we are acting as the frontend for a hardware
278 // accelerated decoder, we'll need to read the frame
279 // back to render it.
280
281 // Map or copy this hwframe to a swframe that we can work with
282 frame = swFrame = m_SwFrameMapper.getSwFrameFromHwFrame(frame);
283 if (swFrame == nullptr) {
284 return;
285 }
286 }
287
288 // Recreate the texture if the frame format or size changes
289 if (hasFrameFormatChanged(frame)) {
290#ifdef HAVE_CUDA
291 if (m_CudaGLHelper != nullptr) {
292 delete m_CudaGLHelper;
293 m_CudaGLHelper = nullptr;
294 }
295#endif
296
297 if (m_Texture != nullptr) {
298 SDL_DestroyTexture(m_Texture);
299 m_Texture = nullptr;
300 }
301 }
302
303 if (m_Texture == nullptr) {
304 Uint32 sdlFormat;
305
306 // Remember to keep this in sync with SdlRenderer::isPixelFormatSupported()!
307 m_NeedsYuvToRgbConversion = false;
308 switch (frame->format)
309 {
310 case AV_PIX_FMT_YUV420P:
311 case AV_PIX_FMT_YUVJ420P:
312 sdlFormat = SDL_PIXELFORMAT_YV12;
313 break;
314 case AV_PIX_FMT_CUDA:
315 case AV_PIX_FMT_NV12:
316 sdlFormat = SDL_PIXELFORMAT_NV12;
317 break;
318 case AV_PIX_FMT_NV21:
319 sdlFormat = SDL_PIXELFORMAT_NV21;
320 break;
321 default:
322 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
323 "Performing color conversion on CPU due to lack of SDL support for format: %s",
324 av_get_pix_fmt_name((AVPixelFormat)frame->format));
325 sdlFormat = SDL_PIXELFORMAT_XRGB8888;
326 m_NeedsYuvToRgbConversion = true;
327 break;
328 }
329
330 if (m_NeedsYuvToRgbConversion) {
331 m_RgbFrame->width = frame->width;
332 m_RgbFrame->height = frame->height;
333 m_RgbFrame->format = AV_PIX_FMT_BGR0;
334
335 sws_freeContext(m_SwsContext);
336
337#if LIBSWSCALE_VERSION_INT >= AV_VERSION_INT(6, 1, 100)
338 m_SwsContext = sws_alloc_context();
339 if (!m_SwsContext) {
340 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
341 "sws_alloc_context() failed");
342 goto Exit;
343 }
344
345 AVDictionary *options { nullptr };
346 av_dict_set_int(&options, "srcw", frame->width, 0);
347 av_dict_set_int(&options, "srch", frame->height, 0);
348 av_dict_set_int(&options, "src_format", frame->format, 0);
349 av_dict_set_int(&options, "dstw", m_RgbFrame->width, 0);
350 av_dict_set_int(&options, "dsth", m_RgbFrame->height, 0);
351 av_dict_set_int(&options, "dst_format", m_RgbFrame->format, 0);
352 av_dict_set_int(&options, "threads", std::min(SDL_GetCPUCount(), 4), 0); // Up to 4 threads
353
354 err = av_opt_set_dict(m_SwsContext, &options);
355 av_dict_free(&options);
356 if (err < 0) {
357 char string[AV_ERROR_MAX_STRING_SIZE];
358 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
359 "av_opt_set_dict() failed: %s",
360 av_make_error_string(string, sizeof(string), err));
361 goto Exit;
362 }
363
364 err = sws_init_context(m_SwsContext, nullptr, nullptr);
365 if (err < 0) {
366 char string[AV_ERROR_MAX_STRING_SIZE];
367 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
368 "sws_init_context() failed: %s",
369 av_make_error_string(string, sizeof(string), err));
370 goto Exit;
371 }
372#else
373 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
374 "CPU color conversion is slow on FFmpeg 4.x. Update FFmpeg for better performance.");
375
376 m_SwsContext = sws_getContext(frame->width, frame->height, (AVPixelFormat)frame->format,
377 m_RgbFrame->width, m_RgbFrame->height, (AVPixelFormat)m_RgbFrame->format,
378 0, nullptr, nullptr, nullptr);
379 if (!m_SwsContext) {
380 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
381 "sws_getContext() failed");
382 goto Exit;
383 }
384#endif
385 }
386 else {
387 // SDL will perform YUV conversion on the GPU
388 switch (getFrameColorspace(frame))
389 {
390 case COLORSPACE_REC_709:
391 SDL_assert(!isFrameFullRange(frame));
392 SDL_SetYUVConversionMode(SDL_YUV_CONVERSION_BT709);
393 break;
394 case COLORSPACE_REC_601:
395 if (isFrameFullRange(frame)) {
396 // SDL's JPEG mode is Rec 601 Full Range
397 SDL_SetYUVConversionMode(SDL_YUV_CONVERSION_JPEG);
398 }
399 else {
400 SDL_SetYUVConversionMode(SDL_YUV_CONVERSION_BT601);
401 }
402 break;
403 default:
404 break;
405 }
406 }
407
408 m_Texture = SDL_CreateTexture(m_Renderer,
409 sdlFormat,
410 SDL_TEXTUREACCESS_STREAMING,
411 frame->width,
412 frame->height);
413 if (!m_Texture) {
414 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
415 "SDL_CreateTexture() failed: %s",
416 SDL_GetError());
417 goto Exit;
418 }
419
420#ifdef HAVE_CUDA
421 if (frame->format == AV_PIX_FMT_CUDA) {
422 SDL_assert(m_CudaGLHelper == nullptr);
423 m_CudaGLHelper = new CUDAGLInteropHelper(((AVHWFramesContext*)frame->hw_frames_ctx->data)->device_ctx);
424
425 SDL_GL_BindTexture(m_Texture, nullptr, nullptr);
426 if (!m_CudaGLHelper->registerBoundTextures()) {
427 // If we can't register textures, fall back to normal read-back rendering
428 delete m_CudaGLHelper;
429 m_CudaGLHelper = nullptr;
430 }
431 SDL_GL_UnbindTexture(m_Texture);
432 }
433#endif
434 }
435
436 if (frame->format == AV_PIX_FMT_CUDA) {
437#ifdef HAVE_CUDA
438 if (m_CudaGLHelper == nullptr || !m_CudaGLHelper->copyCudaFrameToTextures(frame)) {
439 goto ReadbackRetry;
440 }
441#else
442 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
443 "Got CUDA frame, but not built with CUDA support!");
444 goto Exit;
445#endif
446 }
447 else if (frame->format == AV_PIX_FMT_YUV420P || frame->format == AV_PIX_FMT_YUVJ420P) {
448 SDL_UpdateYUVTexture(m_Texture, nullptr,
449 frame->data[0],
450 frame->linesize[0],
451 frame->data[1],
452 frame->linesize[1],
453 frame->data[2],
454 frame->linesize[2]);
455 }
456 else if (!m_NeedsYuvToRgbConversion) {
457#if SDL_VERSION_ATLEAST(2, 0, 15)
458 // SDL_UpdateNVTexture is not supported on all renderer backends,
459 // (notably not DX9), so we must have a fallback in case it's not
460 // supported and for earlier versions of SDL.
461 if (SDL_UpdateNVTexture(m_Texture,
462 nullptr,
463 frame->data[0],
464 frame->linesize[0],
465 frame->data[1],
466 frame->linesize[1]) != 0)
467#endif
468 {
469 char* pixels;
470 int texturePitch;
471
472 err = SDL_LockTexture(m_Texture, nullptr, (void**)&pixels, &texturePitch);
473 if (err < 0) {
474 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
475 "SDL_LockTexture() failed: %s",
476 SDL_GetError());
477 goto Exit;
478 }
479
480 // If the planar pitches match, we can use a single memcpy() to transfer
481 // the data. If not, we'll need to do separate memcpy() calls for each
482 // line to ensure the pitch doesn't get screwed up.
483
484 if (frame->linesize[0] == texturePitch) {
485 memcpy(pixels,
486 frame->data[0],
487 frame->linesize[0] * frame->height);
488 }
489 else {
490 int pitch = SDL_min(frame->linesize[0], texturePitch);
491 for (int i = 0; i < frame->height; i++) {
492 memcpy(pixels + (texturePitch * i),
493 frame->data[0] + (frame->linesize[0] * i),
494 pitch);
495 }
496 }
497
498 if (frame->linesize[1] == texturePitch) {
499 memcpy(pixels + (texturePitch * frame->height),
500 frame->data[1],
501 frame->linesize[1] * frame->height / 2);
502 }
503 else {
504 int pitch = SDL_min(frame->linesize[1], texturePitch);
505 for (int i = 0; i < frame->height / 2; i++) {
506 memcpy(pixels + (texturePitch * (frame->height + i)),
507 frame->data[1] + (frame->linesize[1] * i),
508 pitch);
509 }
510 }
511
512 SDL_UnlockTexture(m_Texture);
513 }
514 }
515 else {
516 // We have a pixel format that SDL doesn't natively support, so we must use
517 // swscale to convert the YUV frame into an RGB frame to upload to the GPU.
518 uint8_t* pixels;
519 int texturePitch;
520
521 err = SDL_LockTexture(m_Texture, nullptr, (void**)&pixels, &texturePitch);
522 if (err < 0) {
523 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
524 "SDL_LockTexture() failed: %s",
525 SDL_GetError());
526 goto Exit;
527 }
528
529 // Create a buffer to wrap our locked texture buffer
530 m_RgbFrame->buf[0] = av_buffer_create(pixels, m_RgbFrame->height * texturePitch, ffNoopFree, nullptr, 0);
531 m_RgbFrame->data[0] = pixels;
532 m_RgbFrame->linesize[0] = texturePitch;
533
534#if LIBSWSCALE_VERSION_INT >= AV_VERSION_INT(6, 1, 100)
535 // Perform multi-threaded color conversion into the locked texture buffer
536 err = sws_scale_frame(m_SwsContext, m_RgbFrame, frame);
537#else
538 // Perform a single-threaded color conversion using the legacy swscale API
539 err = sws_scale(m_SwsContext, frame->data, frame->linesize, 0, frame->height,
540 m_RgbFrame->data, m_RgbFrame->linesize);
541#endif
542
543 av_buffer_unref(&m_RgbFrame->buf[0]);
544 SDL_UnlockTexture(m_Texture);
545
546 if (err < 0) {
547 char string[AV_ERROR_MAX_STRING_SIZE];
548 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
549 "sws_scale_frame() failed: %s",
550 av_make_error_string(string, AV_ERROR_MAX_STRING_SIZE, err));
551 goto Exit;
552 }
553 }
554
555 SDL_RenderClear(m_Renderer);
556
557 // Calculate the video region size, scaling to fill the output size while
558 // preserving the aspect ratio of the video stream.
559 SDL_Rect src, dst;
560 src.x = src.y = 0;
561 src.w = frame->width;
562 src.h = frame->height;
563 dst.x = dst.y = 0;
564 SDL_GetRendererOutputSize(m_Renderer, &dst.w, &dst.h);
565 StreamUtils::scaleSourceToDestinationSurface(&src, &dst);
566
567 // Ensure the viewport is set to the desired video region
568 SDL_RenderSetViewport(m_Renderer, &dst);
569
570 // Draw the video content itself
571 SDL_RenderCopy(m_Renderer, m_Texture, nullptr, nullptr);
572
573 // Reset the viewport to the full window for overlay rendering
574 SDL_RenderSetViewport(m_Renderer, nullptr);
575
576 // Draw the overlays
577 for (int i = 0; i < Overlay::OverlayMax; i++) {
578 renderOverlay((Overlay::OverlayType)i);
579 }
580
581 SDL_RenderPresent(m_Renderer);
582
583Exit:
584 if (swFrame != nullptr) {
585 av_frame_free(&swFrame);
586 }
587}

589bool SdlRenderer::testRenderFrame(AVFrame* frame)
590{
591 // If we are acting as the frontend for a hardware
592 // accelerated decoder, we'll need to read the frame
593 // back to render it. Test that this can be done
594 // for the given frame successfully.
595 if (frame->hw_frames_ctx != nullptr) {
596#ifdef HAVE_MMAL
597 // FFmpeg for Raspberry Pi has NEON-optimized routines that allow
598 // us to use av_hwframe_transfer_data() to convert from SAND frames
599 // to standard fully-planar YUV. Unfortunately, the combination of
600 // doing this conversion on the CPU and the slow GL texture upload
601 // performance on the Pi make the performance of this approach
602 // unacceptable. Don't use this copyback path on the Pi by default
603 // to ensure we fall back to H.264 (with the MMAL renderer) in X11
604 // rather than using HEVC+copyback and getting terrible performance.
605 if (qgetenv("RPI_ALLOW_COPYBACK_RENDER") != "1") {
606 return false;
607 }
608#endif
609
610 AVFrame* swFrame = m_SwFrameMapper.getSwFrameFromHwFrame(frame);
611 if (swFrame == nullptr) {
612 return false;
613 }
614
615 av_frame_free(&swFrame);
616 }
617 else if (!isPixelFormatSupported(m_VideoFormat, (AVPixelFormat)frame->format)) {
618 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
619 "Swframe pixel format unsupported: %d",
620 frame->format);
621 return false;
622 }
623
624 return true;
625}
