195void SystemProperties::refreshDisplaysInternal()
196{
197 if (SDL_InitSubSystem(SDL_INIT_VIDEO) != 0) {
198 SDL_LogError(SDL_LOG_CATEGORY_APPLICATION,
199 "SDL_InitSubSystem(SDL_INIT_VIDEO) failed: %s",
200 SDL_GetError());
201 return;
202 }
203
204 monitorNativeResolutions.clear();
205
206 SDL_DisplayMode bestMode;
207 for (int displayIndex = 0; displayIndex < SDL_GetNumVideoDisplays(); displayIndex++) {
208 SDL_DisplayMode desktopMode;
209 SDL_Rect safeArea;
210
211 if (StreamUtils::getNativeDesktopMode(displayIndex, &desktopMode, &safeArea)) {
212 if (desktopMode.w <= 8192 && desktopMode.h <= 8192) {
213 monitorNativeResolutions.insert(displayIndex, QRect(0, 0, desktopMode.w, desktopMode.h));
214 monitorSafeAreaResolutions.insert(displayIndex, QRect(0, 0, safeArea.w, safeArea.h));
215 }
216 else {
217 SDL_LogWarn(SDL_LOG_CATEGORY_APPLICATION,
218 "Skipping resolution over 8K: %dx%d",
219 desktopMode.w, desktopMode.h);
220 }
221
222 // Start at desktop mode and work our way up
223 bestMode = desktopMode;
224 for (int i = 0; i < SDL_GetNumDisplayModes(displayIndex); i++) {
225 SDL_DisplayMode mode;
226 if (SDL_GetDisplayMode(displayIndex, i, &mode) == 0) {
227 if (mode.w == desktopMode.w && mode.h == desktopMode.h) {
228 if (mode.refresh_rate > bestMode.refresh_rate) {
229 bestMode = mode;
230 }
231 }
232 }
233 }
234
235 // Try to normalize values around our our standard refresh rates.
236 // Some displays/OSes report values that are slightly off.
237 if (bestMode.refresh_rate >= 58 && bestMode.refresh_rate <= 62) {
238 monitorRefreshRates.append(60);
239 }
240 else if (bestMode.refresh_rate >= 28 && bestMode.refresh_rate <= 32) {
241 monitorRefreshRates.append(30);
242 }
243 else {
244 monitorRefreshRates.append(bestMode.refresh_rate);
245 }
246 }
247 }
248
249 SDL_QuitSubSystem(SDL_INIT_VIDEO);
